27/// Get CPU features, loading from a cache if available.
28pub fn get_cpu_features() -> Flags {
29    use core::sync::atomic::AtomicU32;
30    static CACHE: AtomicU32 = AtomicU32::new(0);
31    get_or_init_flags_cache(&CACHE, load_x86_features)
32}
33
34/// Read from cpuid and translate to a `Flags` instance, using `cpu_flags`.
35///
36/// Implementation is taken from [std-detect][std-detect].
37///
38/// [std-detect]: https://github.com/rust-lang/stdarch/blob/690b3a6334d482874163bd6fcef408e0518febe9/crates/std_detect/src/detect/os/x86.rs#L142
39fn load_x86_features() -> Flags {
40    let mut value = Flags::empty();
41
42    if cfg!(target_env = "sgx") {
43        // doesn't support this because it is untrusted data
44        return Flags::empty();
45    }
46
47    // Calling `__cpuid`/`__cpuid_count` from here on is safe because the CPU
48    // has `cpuid` support.
49
50    // 0. EAX = 0: Basic Information:
51    // - EAX returns the "Highest Function Parameter", that is, the maximum leaf
52    //   value for subsequent calls of `cpuinfo` in range [0, 0x8000_0000].
53    // - The vendor ID is stored in 12 u8 ascii chars, returned in EBX, EDX, and ECX
54    //   (in that order)
55    let mut vendor_id = [0u8; 12];
56    let max_basic_leaf;
57    unsafe {
58        let CpuidResult { eax, ebx, ecx, edx } = __cpuid(0);
59        max_basic_leaf = eax;
60        vendor_id[0..4].copy_from_slice(&ebx.to_ne_bytes());
61        vendor_id[4..8].copy_from_slice(&edx.to_ne_bytes());
62        vendor_id[8..12].copy_from_slice(&ecx.to_ne_bytes());
63    }
64
65    if max_basic_leaf < 1 {
66        // Earlier Intel 486, CPUID not implemented
67        return value;
68    }
69
70    // EAX = 1, ECX = 0: Queries "Processor Info and Feature Bits";
71    // Contains information about most x86 features.
72    let CpuidResult { ecx, edx, .. } = unsafe { __cpuid(0x0000_0001_u32) };
73    let proc_info_ecx = Flags::from_bits(ecx);
74    let proc_info_edx = Flags::from_bits(edx);
75
76    // EAX = 7: Queries "Extended Features";
77    // Contains information about bmi,bmi2, and avx2 support.
78    let mut extended_features_ebx = Flags::empty();
79    let mut extended_features_edx = Flags::empty();
80    let mut extended_features_eax_leaf_1 = Flags::empty();
81    if max_basic_leaf >= 7 {
82        let CpuidResult { ebx, edx, .. } = unsafe { __cpuid(0x0000_0007_u32) };
83        extended_features_ebx = Flags::from_bits(ebx);
84        extended_features_edx = Flags::from_bits(edx);
85
86        let CpuidResult { eax, .. } = unsafe { __cpuid_count(0x0000_0007_u32, 0x0000_0001_u32) };
87        extended_features_eax_leaf_1 = Flags::from_bits(eax)
88    }
89
90    // EAX = 0x8000_0000, ECX = 0: Get Highest Extended Function Supported
91    // - EAX returns the max leaf value for extended information, that is,
92    //   `cpuid` calls in range [0x8000_0000; u32::MAX]:
93    let extended_max_basic_leaf = unsafe { __cpuid(0x8000_0000_u32) }.eax;
94
95    // EAX = 0x8000_0001, ECX=0: Queries "Extended Processor Info and Feature Bits"
96    let mut extended_proc_info_ecx = Flags::empty();
97    if extended_max_basic_leaf >= 1 {
98        let CpuidResult { ecx, .. } = unsafe { __cpuid(0x8000_0001_u32) };
99        extended_proc_info_ecx = Flags::from_bits(ecx);
100    }
101
102    let mut enable = |regflags: Flags, regbit, flag| {
103        if regflags.test_nth(regbit) {
104            value.insert(flag);
105        }
106    };
107
108    enable(proc_info_ecx, 0, cpu_flags::SSE3);
109    enable(proc_info_ecx, 29, cpu_flags::F16C);
110    enable(proc_info_edx, 25, cpu_flags::SSE);
111    enable(proc_info_edx, 26, cpu_flags::SSE2);
112    enable(extended_features_ebx, 9, cpu_flags::ERMSB);
113    enable(extended_features_eax_leaf_1, 31, cpu_flags::MOVRS);
114
115    // `XSAVE` and `AVX` support:
116    let cpu_xsave = proc_info_ecx.test_nth(26);
117    if cpu_xsave {
118        // 0. Here the CPU supports `XSAVE`.
119
120        // 1. Detect `OSXSAVE`, that is, whether the OS is AVX enabled and
121        //    supports saving the state of the AVX/AVX2 vector registers on
122        //    context-switches, see:
123        //
124        // - [intel: is avx enabled?][is_avx_enabled],
125        // - [mozilla: sse.cpp][mozilla_sse_cpp].
126        //
127        // [is_avx_enabled]: https://software.intel.com/en-us/blogs/2011/04/14/is-avx-enabled
128        // [mozilla_sse_cpp]: https://hg.mozilla.org/mozilla-central/file/64bab5cbb9b6/mozglue/build/SSE.cpp#l190
129        let cpu_osxsave = proc_info_ecx.test_nth(27);
130
131        if cpu_osxsave {
132            // 2. The OS must have signaled the CPU that it supports saving and
133            // restoring the:
134            //
135            // * SSE -> `XCR0.SSE[1]`
136            // * AVX -> `XCR0.AVX[2]`
137            // * AVX-512 -> `XCR0.AVX-512[7:5]`.
138            // * AMX -> `XCR0.AMX[18:17]`
139            //
140            // by setting the corresponding bits of `XCR0` to `1`.
141            //
142            // This is safe because the CPU supports `xsave` and the OS has set `osxsave`.
143            let xcr0 = unsafe { _xgetbv(0) };
144            // Test `XCR0.SSE[1]` and `XCR0.AVX[2]` with the mask `0b110 == 6`:
145            let os_avx_support = xcr0 & 6 == 6;
146            // Test `XCR0.AVX-512[7:5]` with the mask `0b1110_0000 == 0xe0`:
147            let os_avx512_support = xcr0 & 0xe0 == 0xe0;
148
149            // Only if the OS and the CPU support saving/restoring the AVX
150            // registers we enable `xsave` support:
151            if os_avx_support {
152                // See "13.3 ENABLING THE XSAVE FEATURE SET AND XSAVE-ENABLED
153                // FEATURES" in the "Intel® 64 and IA-32 Architectures Software
154                // Developer’s Manual, Volume 1: Basic Architecture":
155                //
156                // "Software enables the XSAVE feature set by setting
157                // CR4.OSXSAVE[bit 18] to 1 (e.g., with the MOV to CR4
158                // instruction). If this bit is 0, execution of any of XGETBV,
159                // XRSTOR, XRSTORS, XSAVE, XSAVEC, XSAVEOPT, XSAVES, and XSETBV
160                // causes an invalid-opcode exception (#UD)"
161
162                // FMA (uses 256-bit wide registers):
163                enable(proc_info_ecx, 12, cpu_flags::FMA);
164
165                // For AVX-512 the OS also needs to support saving/restoring
166                // the extended state, only then we enable AVX-512 support:
167                if os_avx512_support {
168                    enable(extended_features_edx, 23, cpu_flags::AVX512FP16);
169                    enable(extended_features_eax_leaf_1, 5, cpu_flags::AVX512BF16);
170                }
171            }
172        }
173    }
174
175    // As Hygon Dhyana originates from AMD technology and shares most of the architecture with
176    // AMD's family 17h, but with different CPU Vendor ID("HygonGenuine")/Family series number
177    // (Family 18h).
178    //
179    // For CPUID feature bits, Hygon Dhyana(family 18h) share the same definition with AMD
180    // family 17h.
181    //
182    // Related AMD CPUID specification is https://www.amd.com/system/files/TechDocs/25481.pdf
183    // (AMD64 Architecture Programmer's Manual, Appendix E).
184    // Related Hygon kernel patch can be found on
185    // http://lkml.kernel.org/r/5ce86123a7b9dad925ac583d88d2f921040e859b.1538583282.git.puwen@hygon.cn
186    if vendor_id == *b"AuthenticAMD" || vendor_id == *b"HygonGenuine" {
187        // These features are available on AMD arch CPUs:
188        enable(extended_proc_info_ecx, 16, cpu_flags::FMA4);
189    }
190
191    value
192}

194#[cfg(test)]
195mod tests {
196    extern crate std;
197    use std::is_x86_feature_detected;
198
199    use super::*;
200
201    #[test]
202    fn check_matches_std() {
203        let features = get_cpu_features();
204        for i in 0..cpu_flags::ALL.len() {
205            let flag = cpu_flags::ALL[i];
206            let name = cpu_flags::NAMES[i];
207
208            let std_detected = match flag {
209                cpu_flags::SSE3 => is_x86_feature_detected!("sse3"),
210                cpu_flags::F16C => is_x86_feature_detected!("f16c"),
211                cpu_flags::SSE => is_x86_feature_detected!("sse"),
212                cpu_flags::SSE2 => is_x86_feature_detected!("sse2"),
213                cpu_flags::ERMSB => is_x86_feature_detected!("ermsb"),
214                cpu_flags::MOVRS => continue, // only very recent support in std
215                cpu_flags::FMA => is_x86_feature_detected!("fma"),
216                cpu_flags::FMA4 => continue, // not yet supported in std
217                cpu_flags::AVX512FP16 => is_x86_feature_detected!("avx512fp16"),
218                cpu_flags::AVX512BF16 => is_x86_feature_detected!("avx512bf16"),
219                _ => panic!("untested CPU flag {name}"),
220            };
221
222            assert_eq!(
223                std_detected,
224                features.contains(flag),
225                "different flag {name}. flags: {features:?}"
226            );
227        }
228    }
