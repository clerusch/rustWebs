35fn katexit2(item: TokenStream2) -> TokenStream2 {
36    if let Ok(mut item) = syn::parse2(item.clone()) {
37        match item {
38            syn::Item::Const(syn::ItemConst { ref mut attrs, .. })
39            | syn::Item::Enum(syn::ItemEnum { ref mut attrs, .. })
40            | syn::Item::ExternCrate(syn::ItemExternCrate { ref mut attrs, .. })
41            | syn::Item::Fn(syn::ItemFn { ref mut attrs, .. })
42            | syn::Item::ForeignMod(syn::ItemForeignMod { ref mut attrs, .. })
43            | syn::Item::Impl(syn::ItemImpl { ref mut attrs, .. })
44            | syn::Item::Macro(syn::ItemMacro { ref mut attrs, .. })
45            | syn::Item::Mod(syn::ItemMod { ref mut attrs, .. })
46            | syn::Item::Static(syn::ItemStatic { ref mut attrs, .. })
47            | syn::Item::Struct(syn::ItemStruct { ref mut attrs, .. })
48            | syn::Item::Trait(syn::ItemTrait { ref mut attrs, .. })
49            | syn::Item::TraitAlias(syn::ItemTraitAlias { ref mut attrs, .. })
50            | syn::Item::Type(syn::ItemType { ref mut attrs, .. })
51            | syn::Item::Union(syn::ItemUnion { ref mut attrs, .. })
52            | syn::Item::Use(syn::ItemUse { ref mut attrs, .. }) => {
53                attrs.push(syn::parse_quote! { #[doc = #KATEX_HEADER] });
