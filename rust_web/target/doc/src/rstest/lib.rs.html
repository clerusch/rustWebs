2//! This crate will help you to write simpler tests by leveraging a software testing concept called
3//! [test fixtures](https://en.wikipedia.org/wiki/Test_fixture#Software). A fixture is something
4//! that you can use in your tests to encapsulate a test's dependencies.
5//!
6//! The general idea is to have smaller tests that only describe the thing you're testing while you
7//! hide the auxiliary utilities your tests make use of somewhere else.
8//! For instance, if you have an application that has many tests with users, shopping baskets, and
9//! products, you'd have to create a user, a shopping basket, and product every single time in
10//! every test which becomes unwieldy quickly. In order to cut down on that repetition, you can
11//! instead use fixtures to declare that you need those objects for your function and the fixtures
12//! will take care of creating those by themselves. Focus on the important stuff in your tests!
13//!
14//! In `rstest` a fixture is a function that can return any kind of valid Rust type. This
15//! effectively means that your fixtures are not limited by the kind of data they can return.
16//! A test can consume an arbitrary number of fixtures at the same time.
17//!
18//! ## What
19//!
20//! The `rstest` crate defines the following procedural macros:
21//!
22//! - [`[rstest]`](macro@rstest): Declare that a test or a group of tests that may take
23//!   [fixtures](attr.rstest.html#injecting-fixtures),
24//!   [input table](attr.rstest.html#test-parametrized-cases) or
25//!   [list of values](attr.rstest.html#values-lists).
26//! - [`[fixture]`](macro@fixture): To mark a function as a fixture.
27//!
28//! ## Why
29//!
30//! Very often in Rust we write tests like this
31//!
32//! ```
33//! #[test]
34//! fn should_process_two_users() {
35//!     let mut repository = create_repository();
36//!     repository.add("Bob", 21);
37//!     repository.add("Alice", 22);
38//!
39//!     let processor = string_processor();
40//!     processor.send_all(&repository, "Good Morning");
41//!
42//!     assert_eq!(2, processor.output.find("Good Morning").count());
43//!     assert!(processor.output.contains("Bob"));
44//!     assert!(processor.output.contains("Alice"));
45//! }
46//! ```
47//!
48//! By making use of [`[rstest]`](macro@rstest) we can isolate the dependencies `empty_repository` and
49//! `string_processor` by passing them as fixtures:
50//!
51//! ```
52//! # use rstest::*;
53//! #[rstest]
54//! fn should_process_two_users(mut empty_repository: impl Repository,
55//!                             string_processor: FakeProcessor) {
56//!     empty_repository.add("Bob", 21);
57//!     empty_repository.add("Alice", 22);
58//!
59//!     string_processor.send_all("Good Morning");
60//!
61//!     assert_eq!(2, string_processor.output.find("Good Morning").count());
62//!     assert!(string_processor.output.contains("Bob"));
63//!     assert!(string_processor.output.contains("Alice"));
64//! }
65//! ```
66//!
67//! ... or if you use `"Alice"` and `"Bob"` in other tests, you can isolate `alice_and_bob` fixture
68//! and use it directly:
69//!
70//! ```
71//! # use rstest::*;
72//! # trait Repository { fn add(&mut self, name: &str, age: u8); }
73//! # struct Rep;
74//! # impl Repository for Rep { fn add(&mut self, name: &str, age: u8) {} }
75//! # #[fixture]
76//! # fn empty_repository() -> Rep {
77//! #     Rep
78//! # }
79//! #[fixture]
80//! fn alice_and_bob(mut empty_repository: impl Repository) -> impl Repository {
81//!     empty_repository.add("Bob", 21);
82//!     empty_repository.add("Alice", 22);
83//!     empty_repository
84//! }
85//!
86//! #[rstest]
87//! fn should_process_two_users(alice_and_bob: impl Repository,
88//!                             string_processor: FakeProcessor) {
89//!     string_processor.send_all("Good Morning");
90//!
91//!     assert_eq!(2, string_processor.output.find("Good Morning").count());
92//!     assert!(string_processor.output.contains("Bob"));
93//!     assert!(string_processor.output.contains("Alice"));
94//! }
95//! ```
96//! ### Features
97//!
98//! - `async-timeout`: `timeout` for `async` tests (Default enabled)
99//! - `crate-name`: Import `rstest` package with different name (Default enabled)
100//!
101//! ## Injecting fixtures as function arguments
102//!
103//! `rstest` functions can receive fixtures by using them as input arguments.
104//! A function decorated with [`[rstest]`](attr.rstest.html#injecting-fixtures)
105//! will resolve each argument name by call the fixture function.
106//! Fixtures should be annotated with the [`[fixture]`](macro@fixture) attribute.
107//!
108//! Fixtures will be resolved like function calls by following the standard resolution rules.
109//! Therefore, an identically named fixture can be use in different context.
110//!
111//! ```
112//! # use rstest::*;
113//! # trait Repository { }
114//! # #[derive(Default)]
115//! # struct DataSet {}
116//! # impl Repository for DataSet { }
117//! mod empty_cases {
118//! # use rstest::*;
119//! # trait Repository { }
120//! # #[derive(Default)]
121//! # struct DataSet {}
122//! # impl Repository for DataSet { }
123//!     use super::*;
124//!
125//!     #[fixture]
126//!     fn repository() -> impl Repository {
127//!         DataSet::default()
128//!     }
129//!
130//!     #[rstest]
131//!     fn should_do_nothing(repository: impl Repository) {
132//!         //.. test impl ..
133//!     }
134//! }
135//!
136//! mod non_trivial_case {
137//! # use rstest::*;
138//! # trait Repository { }
139//! # #[derive(Default)]
140//! # struct DataSet {}
141//! # impl Repository for DataSet { }
142//!     use super::*;
143//!
144//!     #[fixture]
145//!     fn repository() -> impl Repository {
146//!         let mut ds = DataSet::default();
147//!         // Fill your dataset with interesting case
148//!         ds
149//!     }
150//!
151//!     #[rstest]
152//!     fn should_notify_all_entries(repository: impl Repository) {
153//!         //.. test impl ..
154//!     }
155//! }
156//!
157//! ```
158//!
159//! Last but not least, fixtures can be injected like we saw in `alice_and_bob` example.
160//!
161//! ## Creating parametrized tests
162//!
163//! You can use also [`[rstest]`](attr.rstest.html#test-parametrized-cases) to create
164//! simple table-based tests. Let's see the classic Fibonacci example:
165//!
166//! ```
167//! use rstest::rstest;
168//!
169//! #[rstest]
170//! #[case(0, 0)]
171//! #[case(1, 1)]
172//! #[case(2, 1)]
173//! #[case(3, 2)]
174//! #[case(4, 3)]
175//! #[case(5, 5)]
176//! #[case(6, 8)]
177//! fn fibonacci_test(#[case] input: u32,#[case] expected: u32) {
178//!     assert_eq!(expected, fibonacci(input))
179//! }
180//!
181//! fn fibonacci(input: u32) -> u32 {
182//!     match input {
183//!         0 => 0,
184//!         1 => 1,
185//!         n => fibonacci(n - 2) + fibonacci(n - 1)
186//!     }
187//! }
188//! ```
189//! This will generate a bunch of tests, one for every `#[case(a, b)]`.
190//!
191//! ## Creating a test for each combinations of given values
192//!
193//! In some cases you need to test your code for each combinations of some input values. In this
194//! cases [`[rstest]`](attr.rstest.html#values-lists) give you the ability to define a list
195//! of values (rust expressions) to use for an arguments.
196//!
197//! ```
198//! # use rstest::rstest;
199//! # #[derive(PartialEq, Debug)]
200//! # enum State { Init, Start, Processing, Terminated }
201//! # #[derive(PartialEq, Debug)]
202//! # enum Event { Error, Fatal }
203//! # impl State { fn process(self, event: Event) -> Self { self } }
204//!
205//! #[rstest]
206//! fn should_terminate(
207//!     #[values(State::Init, State::Start, State::Processing)]
208//!     state: State,
209//!     #[values(Event::Error, Event::Fatal)]
210//!     event: Event
211//! ) {
212//!     assert_eq!(State::Terminated, state.process(event))
213//! }
214//! ```
215//!
216//! This will generate a test for each combination of `state` and `event`.
217//!
218//! ## Magic Conversion
219//!
220//! If you need a value where its type implement `FromStr()` trait you
221//! can use a literal string to build it.
222//!
223//! ```
224//! # use rstest::rstest;
225//! # use std::net::SocketAddr;
226//! #[rstest]
227//! #[case("1.2.3.4:8080", 8080)]
228//! #[case("127.0.0.1:9000", 9000)]
229//! fn check_port(#[case] addr: SocketAddr, #[case] expected: u16) {
230//!     assert_eq!(expected, addr.port());
231//! }
232//! ```
233//! You can use this feature also in value list and in fixture default value.
234//!
235//! # Optional features
236//!
237//! `rstest` Enable all features by default. You can disable them if you need to
238//! speed up compilation.
239//!
240//! - **`async-timeout`** *(enabled by default)* â€” Implement timeout for async
241//!   tests.
242//!
243//! # Rust version compatibility
244//!
245//! The minimum supported Rust version is 1.67.1.
246//!

253/// Define a fixture that you can use in all `rstest`'s test arguments. You should just mark your
254/// function as `#[fixture]` and then use it as a test's argument. Fixture functions can also
255/// use other fixtures.
256///
257/// Let's see a trivial example:
258///
259/// ```
260/// use rstest::*;
261///
262/// #[fixture]
263/// fn twenty_one() -> i32 { 21 }
264///
265/// #[fixture]
266/// fn two() -> i32 { 2 }
267///
268/// #[fixture]
269/// fn injected(twenty_one: i32, two: i32) -> i32 { twenty_one * two }
270///
271/// #[rstest]
272/// fn the_test(injected: i32) {
273///     assert_eq!(42, injected)
274/// }
275/// ```
276///
277/// If the fixture function is an [`async` function](#async) your fixture become an `async`
278/// fixture.
279///
280/// # Default values
281///
282/// If you need to define argument default value you can use `#[default(expression)]`
283/// argument's attribute:
284///
285/// ```
286/// use rstest::*;
287///
288/// #[fixture]
289/// fn injected(
290///     #[default(21)]
291///     twenty_one: i32,
292///     #[default(1 + 1)]
293///     two: i32
294/// ) -> i32 { twenty_one * two }
295///
296/// #[rstest]
297/// fn the_test(injected: i32) {
298///     assert_eq!(42, injected)
299/// }
300/// ```
301/// The `expression` could be any valid rust expression, even an `async` block if you need.
302/// Moreover, if the type implements `FromStr` trait you can use a literal string to build it.
303///
304/// ```
305/// # use rstest::*;
306/// # use std::net::SocketAddr;
307/// # struct DbConnection {}
308/// #[fixture]
309/// fn db_connection(
310///     #[default("127.0.0.1:9000")]
311///     addr: SocketAddr
312/// ) -> DbConnection {
313///     // create connection
314/// # DbConnection{}
315/// }
316/// ```
317///
318/// # Async
319///
320/// If you need you can write `async` fixtures to use in your `async` tests. Simply use `async`
321/// keyword for your function and the fixture become an `async` fixture.
322///
323/// ```
324/// use rstest::*;
325///
326/// #[fixture]
327/// async fn async_fixture() -> i32 { 42 }
328///
329///
330/// #[rstest]
331/// async fn the_test(#[future] async_fixture: i32) {
332///     assert_eq!(42, async_fixture.await)
333/// }
334/// ```
335/// The `#[future]` argument attribute helps to remove the `impl Future<Output = T>` boilerplate.
336/// In this case the macro expands it in:
337///
338/// ```
339/// # use rstest::*;
340/// # use std::future::Future;
341/// # #[fixture]
342/// # async fn async_fixture() -> i32 { 42 }
343/// #[rstest]
344/// async fn the_test(async_fixture: impl std::future::Future<Output = i32>) {
345///     assert_eq!(42, async_fixture.await)
346/// }
347/// ```
348/// If you need, you can use `#[future]` attribute also with an implicit lifetime reference
349/// because the macro will replace the implicit lifetime with an explicit one.
350///
351/// # Rename
352///
353/// Sometimes you want to have long and descriptive name for your fixture but you prefer to use a much
354/// shorter name for argument that represent it in your fixture or test. You can rename the fixture
355/// using `#[from(short_name)]` attribute like following example:
356///
357/// ## Destructuring
358///
359/// It's possible to destructure the fixture type but, in this case, your're forced to use renaming syntax
360/// because it's not possible to guess the fixture name from this syntax:
361///
362/// ```
363/// use rstest::*;
364/// #[fixture]
365/// fn two_values() -> (u32, u32) { (42, 24) }
366///
367/// #[rstest]
368/// fn the_test(#[from(two_values)] (first, _): (u32, u32)) {
369///     assert_eq!(42, first)
370/// }
371/// ```
372///
373/// ```
374/// use rstest::*;
375///
376/// #[fixture]
377/// fn long_and_boring_descriptive_name() -> i32 { 42 }
378///
379/// #[rstest]
380/// fn the_test(#[from(long_and_boring_descriptive_name)] short: i32) {
381///     assert_eq!(42, short)
382/// }
383/// ```
384///
385/// This feature can also be useful when you don't want to declare the `use` of a fixture or simple
386/// use the fixture's path:
387///
388/// ```
389/// use rstest::*;
390///
391/// # mod magic_numbers {
392/// #     use rstest::*;
393/// #     #[fixture]
394/// #     pub fn fortytwo() -> i32 { 42 }
395/// # }
396/// #[rstest]
397/// fn the_test(#[from(magic_numbers::fortytwo)] x: i32) {
398///     assert_eq!(42, x)
399/// }
400/// ```
401///
402/// # `#[once]` Fixture
403///
404/// Especially in integration tests there are cases where you need a fixture that is called just once
405/// for every tests. `rstest` provides `#[once]` attribute for these cases.
406///
407/// If you mark your fixture with this attribute, then `rstest` will compute a static reference to your
408/// fixture result and return this reference to all your tests that need this fixture.
409///
410/// In follow example all tests share the same reference to the `42` static value.
411///
412/// ```
413/// use rstest::*;
414///
415/// #[fixture]
416/// #[once]
417/// fn once_fixture() -> i32 { 42 }
418///
419/// // Take care!!! You need to use a reference to the fixture value
420///
421/// #[rstest]
422/// #[case(1)]
423/// #[case(2)]
424/// fn cases_tests(once_fixture: &i32, #[case] v: i32) {
425///     // Take care!!! You need to use a reference to the fixture value
426///     assert_eq!(&42, once_fixture)
427/// }
428///
429/// #[rstest]
430/// fn single(once_fixture: &i32) {
431///     assert_eq!(&42, once_fixture)
432/// }
433/// ```
434///
435/// There are some limitations when you use `#[once]` fixture. `rstest` forbid to use once fixture
436/// for:
437///
438/// - `async` function
439/// - Generic function (both with generic types or use `impl` trait)
440///
441/// Take care that the `#[once]` fixture value will **never be dropped**.
442///
443/// # Partial Injection
444///
445/// You can also partially inject fixture dependency using `#[with(v1, v2, ..)]` attribute:
446///
447/// ```
448/// use rstest::*;
449///
450/// #[fixture]
451/// fn base() -> i32 { 1 }
452///
453/// #[fixture]
454/// fn first(base: i32) -> i32 { 1 * base }
455///
456/// #[fixture]
457/// fn second(base: i32) -> i32 { 2 * base }
458///
459/// #[fixture]
460/// fn injected(first: i32, #[with(3)] second: i32) -> i32 { first * second }
461///
462/// #[rstest]
463/// fn the_test(injected: i32) {
464///     assert_eq!(-6, injected)
465/// }
466/// ```
467/// Note that injected value can be an arbitrary rust expression. `#[with(v1, ..., vn)]`
468/// attribute will inject `v1, ..., vn` expression as fixture arguments: all remaining arguments
469/// will be resolved as fixtures.
470///
471/// Sometimes the return type cannot be inferred so you must define it: For the few times you may
472/// need to do it, you can use the `#[default(type)]`, `#[partial_n(type)]` function attribute
473/// to define it:
474///
475/// ```
476/// use rstest::*;
477/// # use std::fmt::Debug;
478///
479/// #[fixture]
480/// pub fn i() -> u32 {
481///     42
482/// }
483///
484/// #[fixture]
485/// pub fn j() -> i32 {
486///     -42
487/// }
488///
489/// #[fixture]
490/// #[default(impl Iterator<Item=(u32, i32)>)]
491/// #[partial_1(impl Iterator<Item=(I,i32)>)]
492/// pub fn fx<I, J>(i: I, j: J) -> impl Iterator<Item=(I, J)> {
493///     std::iter::once((i, j))
494/// }
495///
496/// #[rstest]
497/// fn resolve_by_default(mut fx: impl Iterator<Item=(u32, i32)>) {
498///     assert_eq!((42, -42), fx.next().unwrap())
499/// }
500///
501/// #[rstest]
502/// fn resolve_partial(#[with(42.0)] mut fx: impl Iterator<Item=(f32, i32)>) {
503///     assert_eq!((42.0, -42), fx.next().unwrap())
504/// }
505/// ```
506/// `partial_i` is the fixture used when you inject the first `i` arguments in test call.
507///

572/// The attribute that you should use for your tests. Your
573/// annotated function's arguments can be
574/// [injected](attr.rstest.html#injecting-fixtures) with
575/// [`[fixture]`](macro@fixture)s, provided by
576/// [parametrized cases](attr.rstest.html#test-parametrized-cases)
577/// or by [value lists](attr.rstest.html#values-lists).
578///
579/// `rstest` attribute can be applied to _any_ function and you can customize its
580/// parameters by using function and arguments attributes.
581///
582/// Your test function can use generics, `impl` or `dyn` and like any kind of rust tests:
583///
584/// - return results
585/// - marked by `#[should_panic]` attribute
586///
587/// In the function signature, where you define your tests inputs, you can also destructuring
588/// the values like any other rust function.
589///
590/// If the test function is an [`async` function](#async) `rstest` will run all tests as `async`
591/// tests. You can use it just with `async-std` and you should include `attributes` in
592/// `async-std`'s features.
593///
594/// In your test function you can:
595///
596/// - [injecting fixtures](#injecting-fixtures)
597/// - Generate [parametrized test cases](#test-parametrized-cases)
598/// - Generate tests for each combination of [value lists](#values-lists)
599///
600/// ## Injecting Fixtures
601///
602/// The simplest case is write a test that can be injected with
603/// [`[fixture]`](macro@fixture)s. You can just declare all used fixtures by passing
604/// them as a function's arguments. This can help your test to be neat
605/// and make your dependency clear.
606///
607/// ```
608/// use rstest::*;
609///
610/// #[fixture]
611/// fn injected() -> i32 { 42 }
612///
613/// #[rstest]
614/// fn the_test(injected: i32) {
615///     assert_eq!(42, injected)
616/// }
617/// ```
618///
619/// [`[rstest]`](macro@rstest) procedural macro will desugar it to something that isn't
620/// so far from
621///
622/// ```
623/// #[test]
624/// fn the_test() {
625///     let injected=injected();
626///     assert_eq!(42, injected)
627/// }
628/// ```
629///
630/// If you want to use long and descriptive names for your fixture but prefer to use
631/// shorter names inside your tests you use rename feature described in
632/// [fixture rename](attr.fixture.html#rename):
633///
634/// ```
635/// use rstest::*;
636///
637/// #[fixture]
638/// fn long_and_boring_descriptive_name() -> i32 { 42 }
639///
640/// #[rstest]
641/// fn the_test(#[from(long_and_boring_descriptive_name)] short: i32) {
642///     assert_eq!(42, short)
643/// }
644/// ```
645///
646/// The use of `#[from(...)]` attribute is mandatory if you need to destructure the value:
647///
648/// ```
649/// use rstest::*;
650///
651/// #[fixture]
652/// fn tuple() -> (u32, f32) { (42, 42.0) }
653///
654/// #[rstest]
655/// fn the_test(#[from(tuple)] (u, _): (u32, f32)) {
656///     assert_eq!(42, u)
657/// }
658/// ```
659///
660/// Sometimes is useful to have some parameters in your fixtures but your test would
661/// override the fixture's default values in some cases. Like in
662/// [fixture partial injection](attr.fixture.html#partial-injection) you use `#[with]`
663/// attribute to indicate some fixture's arguments also in `rstest`.
664///
665/// ```
666/// # struct User(String, u8);
667/// # impl User { fn name(&self) -> &str {&self.0} }
668/// use rstest::*;
669///
670/// #[fixture]
671/// fn user(
672///     #[default("Alice")] name: impl AsRef<str>,
673///     #[default(22)] age: u8
674/// ) -> User { User(name.as_ref().to_owned(), age) }
675///
676/// #[rstest]
677/// fn check_user(#[with("Bob")] user: User) {
678///     assert_eq("Bob", user.name())
679/// }
680/// ```
681///
682/// ## Test Parametrized Cases
683///
684/// If you would execute your test for a set of input data cases
685/// you can define the arguments to use and the cases list. Let see
686/// the classical Fibonacci example. In this case we would give the
687/// `input` value and the `expected` result for a set of cases to test.
688///
689/// ```
690/// use rstest::rstest;
691///
692/// #[rstest]
693/// #[case(0, 0)]
694/// #[case(1, 1)]
695/// #[case(2, 1)]
696/// #[case(3, 2)]
697/// #[case(4, 3)]
698/// fn fibonacci_test(#[case] input: u32,#[case] expected: u32) {
699///     assert_eq!(expected, fibonacci(input))
700/// }
701///
702/// fn fibonacci(input: u32) -> u32 {
703///     match input {
704///         0 => 0,
705///         1 => 1,
706///         n => fibonacci(n - 2) + fibonacci(n - 1)
707///     }
708/// }
709/// ```
710///
711/// `rstest` will produce 5 independent tests and not just one that
712/// check every case. Every test can fail independently and `cargo test`
713/// will give follow output:
714///
715/// ```text
716/// running 5 tests
717/// test fibonacci_test::case_1 ... ok
718/// test fibonacci_test::case_2 ... ok
719/// test fibonacci_test::case_3 ... ok
720/// test fibonacci_test::case_4 ... ok
721/// test fibonacci_test::case_5 ... ok
722///
723/// test result: ok. 5 passed; 0 failed; 0 ignored; 0 measured; 0 filtered out
724/// ```
725///
726/// The cases input values can be arbitrary Rust expressions that return the
727/// argument type.
728///
729/// ```
730/// use rstest::rstest;
731///
732/// fn sum(a: usize, b: usize) -> usize { a + b }
733///
734/// #[rstest]
735/// #[case("foo", 3)]
736/// #[case(String::from("foo"), 2 + 1)]
737/// #[case(format!("foo"), sum(2, 1))]
738/// fn test_len(#[case] s: impl AsRef<str>,#[case] len: usize) {
739///     assert_eq!(s.as_ref().len(), len);
740/// }
741/// ```
742/// ### Feature flagged cases
743///
744/// In case you want certain test cases to only be present if a certain feature is
745/// enabled, use `#[cfg_attr(feature = â€¦, case(â€¦))]`:
746///
747/// ```
748/// use rstest::rstest;
749///
750/// #[rstest]
751/// #[case(2, 2)]
752/// #[cfg_attr(feature = "frac", case(4/2, 2))]
753/// #[case(4/2, 2)]
754/// fn it_works(#[case] a: u32, #[case] b: u32) {
755///     assert!(a == b);
756/// }
757/// ```
758///
759/// This also works with [`rstest_reuse`](https://crates.io/crates/rstest_reuse).
760///
761/// ### Magic Conversion
762///
763/// You can use the magic conversion feature every time you would define a variable
764/// where its type define `FromStr` trait: test will parse the string to build the value.
765///
766/// ```
767/// # use rstest::rstest;
768/// # use std::path::PathBuf;
769/// # fn count_words(path: PathBuf) -> usize {0}
770/// #[rstest]
771/// #[case("resources/empty", 0)]
772/// #[case("resources/divine_comedy", 101.698)]
773/// fn test_count_words(#[case] path: PathBuf, #[case] expected: usize) {
774///     assert_eq!(expected, count_words(path))
775/// }
776/// ```
777///
778/// ### Optional case description
779///
780/// Optionally you can give a _description_ to every case simple by follow `case`
781/// with `::my_case_description` where `my_case_description` should be a a valid
782/// Rust ident.
783///
784/// ```
785/// # use rstest::*;
786/// #[rstest]
787/// #[case::zero_base_case(0, 0)]
788/// #[case::one_base_case(1, 1)]
789/// #[case(2, 1)]
790/// #[case(3, 2)]
791/// fn fibonacci_test(#[case] input: u32,#[case] expected: u32) {
792///     assert_eq!(expected, fibonacci(input))
793/// }
794///
795/// # fn fibonacci(input: u32) -> u32 {
796/// #     match input {
797/// #         0 => 0,
798/// #         1 => 1,
799/// #         n => fibonacci(n - 2) + fibonacci(n - 1)
800/// #     }
801/// # }
802/// ```
803///
804/// Output will be
805/// ```text
806/// running 4 tests
807/// test fibonacci_test::case_1_zero_base_case ... ok
808/// test fibonacci_test::case_2_one_base_case ... ok
809/// test fibonacci_test::case_3 ... ok
810/// test fibonacci_test::case_4 ... ok
811///
812/// test result: ok. 4 passed; 0 failed; 0 ignored; 0 measured; 0 filtered out
813/// ```
814///
815/// ### Use specific `case` attributes
816///
817/// Every function's attributes that preceding a `#[case]` attribute will
818/// be used in this test case and all function's attributes that follow the
819/// last `#[case]` attribute will mark all test cases.
820///
821/// This feature can be use to mark just some cases as `should_panic`
822/// and choose to have a fine grain on expected panic messages.
823///
824/// In follow example we run 3 tests where the first pass without any
825/// panic, in the second we catch a panic but we don't care about the message
826/// and in the third one we also check the panic message.
827///
828/// ```
829/// use rstest::rstest;
830///
831/// #[rstest]
832/// #[case::no_panic(0)]
833/// #[should_panic]
834/// #[case::panic(1)]
835/// #[should_panic(expected="expected")]
836/// #[case::panic_with_message(2)]
837/// fn attribute_per_case(#[case] val: i32) {
838///     match val {
839///         0 => assert!(true),
840///         1 => panic!("No catch"),
841///         2 => panic!("expected"),
842///         _ => unreachable!(),
843///     }
844/// }
845/// ```
846///
847/// Output:
848///
849/// ```text
850/// running 3 tests
851/// test attribute_per_case::case_1_no_panic ... ok
852/// test attribute_per_case::case_3_panic_with_message ... ok
853/// test attribute_per_case::case_2_panic ... ok
854///
855/// test result: ok. 3 passed; 0 failed; 0 ignored; 0 measured; 0 filtered out
856/// ```
857///
858/// To mark all your tests as `#[should_panic]` use:
859///
860/// ```
861/// # use rstest::rstest;
862/// #[rstest]
863/// #[case(1)]
864/// #[case(2)]
865/// #[case(3)]
866/// #[should_panic]
867/// fn fail(#[case] v: u32) { assert_eq!(0, v) }
868/// ```
869///
870/// ## Values Lists
871///
872/// Another useful way to write a test and execute it for some values
873/// is to use the values list syntax. This syntax can be useful both
874/// for a plain list and for testing all combination of input arguments.
875///
876/// ```
877/// # use rstest::*;
878/// # fn is_valid(input: &str) -> bool { true }
879///
880/// #[rstest]
881/// fn should_be_valid(
882///     #[values("John", "alice", "My_Name", "Zigy_2001")]
883///     input: &str
884/// ) {
885///     assert!(is_valid(input))
886/// }
887/// ```
888///
889/// or
890///
891/// ```
892/// # use rstest::*;
893/// # fn valid_user(name: &str, age: u8) -> bool { true }
894///
895/// #[rstest]
896/// fn should_accept_all_corner_cases(
897///     #[values("J", "A", "A________________________________________21")]
898///     name: &str,
899///     #[values(14, 100)]
900///     age: u8
901/// ) {
902///     assert!(valid_user(name, age))
903/// }
904/// ```
905/// where `cargo test` output is
906///
907/// ```text
908/// test should_accept_all_corner_cases::name_1___J__::age_2_100 ... ok
909/// test should_accept_all_corner_cases::name_2___A__::age_1_14 ... ok
910/// test should_accept_all_corner_cases::name_2___A__::age_2_100 ... ok
911/// test should_accept_all_corner_cases::name_3___A________________________________________21__::age_2_100 ... ok
912/// test should_accept_all_corner_cases::name_3___A________________________________________21__::age_1_14 ... ok
913/// test should_accept_all_corner_cases::name_1___J__::age_1_14 ... ok
914///
915/// test result: ok. 6 passed; 0 failed; 0 ignored; 0 measured; 0 filtered out; finished in 0.00s
916/// ```
917/// Note that the test names contains the given expression sanitized into
918/// a valid Rust identifier name. This should help to identify which case fails.
919///
920///
921/// Also value list implements the magic conversion feature: every time the value type
922/// implements `FromStr` trait you can use a literal string to define it.
923///
924/// ```
925/// # use rstest::rstest;
926/// # use std::net::SocketAddr;
927/// #[rstest]
928/// fn given_port(#[values("1.2.3.4:8000", "4.3.2.1:8000", "127.0.0.1:8000")] addr: SocketAddr) {
929///     assert_eq!(8000, addr.port())
930/// }
931/// ```
932///
933/// ## Destructuring inputs
934///
935/// Both paramtrized case and values can be destructured:
936///
937/// ```
938/// # use rstest::*;
939/// struct S {
940///     first: u32,
941///     second: u32,
942/// }
943///
944/// struct T(i32);
945///
946/// #[rstest]
947/// #[case(S{first: 21, second: 42})]
948/// fn some_test(#[case] S{first, second} : S, #[values(T(-1), T(1))] T(t): T) {
949///     assert_eq!(1, t * t);
950///     assert_eq!(2 * first, second);
951/// }
952/// ```
953///
954/// ## Files path as input arguments
955///
956/// If you need to create a test for each file in a given location you can use
957/// `#[files("glob path syntax")]` attribute to generate a test for each file that
958/// satisfy the given glob path.
959///
960/// ```
961/// # use rstest::rstest;
962/// # use std::path::{Path, PathBuf};
963/// # fn check_file(path: &Path) -> bool { true };
964/// #[rstest]
965/// fn for_each_file(#[files("src/**/*.rs")] #[exclude("test")] path: PathBuf) {
966///     assert!(check_file(&path))
967/// }
968/// ```
969///
970/// The default behavior is to ignore the files that start with `"."`, but you can
971/// modify this by use `#[include_dot_files]` attribute. The `files` attribute can be
972/// used more than once on the same variable, and you can also create some custom
973/// exclusion rules with the `#[exclude("regex")]` attributes that filter out all
974/// paths that verify the regular expression.
975///
976/// Sometime is useful to have test files in a workspace folder to share them between the
977/// crates in your workspace. You can do that by use the usual parent folders `..` in
978/// the glob path. In this case the test names will be the relative path from the crate root
979/// where the parent folder components are replaced by `_UP`: for instance if you have a
980/// `valid_call.yaml` in the folder `../test_cases` (from your crate root) a test name could be
981/// `path_1__UP_test_cases_valid_call_yaml`.
982///
983/// If you want to change the base path for the test files. You can do that by using
984/// `#[base_dir = "..."]` attribute. The `base_dir` is resolved relative to the crate root
985/// (similar to the `files` attribute without `base_dir`). If you want to use an absolute
986/// path you can use `#[base_dir = "/path/to/your/files"]`.
987///
988/// Both `files` and `base_dir` attributes can use environment variables using the syntax
989/// `$VAR` or `${VAR}`. If the environment variable is not set, the attribute will cause
990/// an error. This can be ignored using the `#[ignore_missing_env_vars]` attribute. A
991/// default value can be provided for the environment variable using the syntax
992/// `${VAR:-default}` (similar to bash). This can be really useful when you want to override
993/// the cargo root on some environments: use `#[base_dir = "BASE_TEST_DIR:-"]` do the trick.
994///
995/// Finally, often you would to recompile tests sources when file the folders or the
996/// environment variables changed. In this case you should provide a `build.rs` script file
997/// that tell to the compiler what to look in order to recompile the tests. For instance
998/// follow a simple example:
999///
1000/// ```ignore
1001/// pub fn main() {
1002///     println!("cargo::rerun-if-changed=tests/resources");
1003///     println!("cargo::rerun-if-env-changed=BASE_TEST_DIR");
1004/// }
1005/// ```
1006///
1007/// ## Use Parametrize definition in more tests
1008///
1009/// If you need to use a test list for more than one test you can use
1010/// [`rstest_reuse`](https://crates.io/crates/rstest_reuse) crate.
1011/// With this helper crate you can define a template and use it everywhere.
1012///
1013/// ```rust,ignore
1014/// use rstest::rstest;
1015/// use rstest_reuse::{self, *};
1016///
1017/// #[template]
1018/// #[rstest]
1019/// #[case(2, 2)]
1020/// #[case(4/2, 2)]
1021/// fn two_simple_cases(#[case] a: u32, #[case] b: u32) {}
1022///
1023/// #[apply(two_simple_cases)]
1024/// fn it_works(#[case] a: u32,#[case] b: u32) {
1025///     assert_eq!(a, b);
1026/// }
1027/// ```
1028///
1029/// See [`rstest_reuse`](https://crates.io/crates/rstest_reuse) for more details.
1030///
1031/// ## Async
1032///
1033/// `rstest` provides out of the box `async` support. Just mark your
1034/// test function as `async` and it'll use `#[async-std::test]` to
1035/// annotate it. This feature can be really useful to build async
1036/// parametric tests using a tidy syntax:
1037///
1038/// ```
1039/// use rstest::*;
1040/// # async fn async_sum(a: u32, b: u32) -> u32 { a + b }
1041///
1042/// #[rstest]
1043/// #[case(5, 2, 3)]
1044/// #[should_panic]
1045/// #[case(42, 40, 1)]
1046/// async fn my_async_test(#[case] expected: u32, #[case] a: u32, #[case] b: u32) {
1047///     assert_eq!(expected, async_sum(a, b).await);
1048/// }
1049/// ```
1050///
1051/// Currently only `async-std` is supported out of the box. But if you need to use
1052/// another runtime that provide it's own test attribute (i.e. `tokio::test` or
1053/// `actix_rt::test`) you can use it in your `async` test like described in
1054/// [Inject Test Attribute](attr.rstest.html#inject-test-attribute).
1055///
1056/// To use this feature, you need to enable `attributes` in the `async-std`
1057/// features list in your `Cargo.toml`:
1058///
1059/// ```toml
1060/// async-std = { version = "1.13", features = ["attributes"] }
1061/// ```
1062///
1063/// If your test input is an async value (fixture or test parameter) you can use `#[future]`
1064/// attribute to remove `impl Future<Output = T>` boilerplate and just use `T`:
1065///
1066/// ```
1067/// use rstest::*;
1068/// #[fixture]
1069/// async fn base() -> u32 { 42 }
1070///
1071/// #[rstest]
1072/// #[case(21, async { 2 })]
1073/// #[case(6, async { 7 })]
1074/// async fn my_async_test(#[future] base: u32, #[case] expected: u32, #[future] #[case] div: u32) {
1075///     assert_eq!(expected, base.await / div.await);
1076/// }
1077/// ```
1078///
1079/// As you noted you should `.await` all _future_ values and this some times can be really boring.
1080/// In this case you can use `#[future(awt)]` to _awaiting_ an input or annotating your function
1081/// with `#[awt]` attributes to globally `.await` all your _future_ inputs. Previous code can be
1082/// simplified like follow:
1083///
1084/// ```
1085/// use rstest::*;
1086/// # #[fixture]
1087/// # async fn base() -> u32 { 42 }
1088///
1089/// #[rstest]
1090/// #[case(21, async { 2 })]
1091/// #[case(6, async { 7 })]
1092/// #[awt]
1093/// async fn global(#[future] base: u32, #[case] expected: u32, #[future] #[case] div: u32) {
1094///     assert_eq!(expected, base / div);
1095/// }
1096///
1097/// #[rstest]
1098/// #[case(21, async { 2 })]
1099/// #[case(6, async { 7 })]
1100/// async fn single(#[future] base: u32, #[case] expected: u32, #[future(awt)] #[case] div: u32) {
1101///     assert_eq!(expected, base.await / div);
1102/// }
1103/// ```
1104///
1105/// ### Default timeout
1106///
1107/// You can set a default timeout for test using the `RSTEST_TIMEOUT` environment variable.
1108/// The value is in seconds and is evaluated on test compile time.///
1109///
1110/// ### Test `#[timeout()]`
1111///
1112/// You can define an execution timeout for your tests with `#[timeout(<duration>)]` attribute. Timeout
1113/// works both for sync and async tests and is runtime agnostic. `#[timeout(<duration>)]` take an
1114/// expression that should return a `std::time::Duration`. Follow a simple async example:
1115///
1116/// ```rust
1117/// use rstest::*;
1118/// use std::time::Duration;
1119///
1120/// async fn delayed_sum(a: u32, b: u32,delay: Duration) -> u32 {
1121///     async_std::task::sleep(delay).await;
1122///     a + b
1123/// }
1124///
1125/// #[rstest]
1126/// #[timeout(Duration::from_millis(80))]
1127/// async fn single_pass() {
1128///     assert_eq!(4, delayed_sum(2, 2, ms(10)).await);
1129/// }
1130/// ```
1131/// In this case test pass because the delay is just 10 milliseconds and timeout is
1132/// 80 milliseconds.
1133///
1134/// You can use `timeout` attribute like any other attribute in your tests, and you can
1135/// override a group timeout with a test specific one. In the follow example we have
1136/// 3 tests where first and third use 100 milliseconds but the second one use 10 milliseconds.
1137/// Another valuable point in this example is to use an expression to compute the
1138/// duration.
1139///
1140/// ```rust
1141/// # use rstest::*;
1142/// # use std::time::Duration;
1143/// #
1144/// # async fn delayed_sum(a: u32, b: u32,delay: Duration) -> u32 {
1145/// #     async_std::task::sleep(delay).await;
1146/// #     a + b
1147/// # }
1148/// fn ms(ms: u32) -> Duration {
1149///     Duration::from_millis(ms.into())
1150/// }
1151///
1152/// #[rstest]
1153/// #[case::pass(ms(1), 4)]
1154/// #[timeout(ms(10))]
1155/// #[case::fail_timeout(ms(60), 4)]
1156/// #[case::fail_value(ms(1), 5)]
1157/// #[timeout(ms(100))]
1158/// async fn group_one_timeout_override(#[case] delay: Duration, #[case] expected: u32) {
1159///     assert_eq!(expected, delayed_sum(2, 2, delay).await);
1160/// }
1161/// ```
1162///
1163/// If you want to use `timeout` for `async` test you need to use `async-timeout`
1164/// feature (enabled by default).
1165///
1166/// ## Inject Test Attribute
1167///
1168/// If you would like to use another `test` attribute for your test you can simply
1169/// indicate it in your test function's attributes. For instance if you want
1170/// to test some async function with use `actix_rt::test` attribute you can just write:
1171///
1172/// ```
1173/// use rstest::*;
1174/// use actix_rt;
1175/// use std::future::Future;
1176///
1177/// #[rstest]
1178/// #[case(2, async { 4 })]
1179/// #[case(21, async { 42 })]
1180/// #[actix_rt::test]
1181/// async fn my_async_test(#[case] a: u32, #[case] #[future] result: u32) {
1182///     assert_eq!(2 * a, result.await);
1183/// }
1184/// ```
1185/// Just the attributes that ends with `test` (last path segment) can be injected:
1186/// in this case the `#[actix_rt::test]` attribute will replace the standard `#[test]`
1187/// attribute.
1188///
1189/// Some test attributes allow to inject arguments into the test function, in a similar way to rstest.
1190/// This can lead to compile errors when rstest is not able to resolve the additional arguments.
1191/// To avoid this, see [Ignoring Arguments](attr.rstest.html#ignoring-arguments).
1192///
1193/// ## Local lifetime and `#[by_ref]` attribute
1194///
1195/// In some cases you may want to use a local lifetime for some arguments of your test.
1196/// In these cases you can use the `#[by_ref]` attribute then use the reference instead
1197/// the value.
1198///
1199/// ```rust
1200/// # use std::cell::Cell;
1201/// # use rstest::*;
1202/// # #[derive(Debug, Clone, Copy, PartialEq, Eq)]
1203/// enum E<'a> {
1204///     A(bool),
1205///     B(&'a Cell<E<'a>>),
1206/// }
1207///
1208/// fn make_e_from_bool<'a>(_bump: &'a (), b: bool) -> E<'a> {
1209///     E::A(b)
1210/// }
1211///
1212/// #[fixture]
1213/// fn bump() -> () {}
1214///
1215/// #[rstest]
1216/// #[case(true, E::A(true))]
1217/// fn it_works<'a>(#[by_ref] bump: &'a (), #[case] b: bool, #[case] expected: E<'a>) {
1218///     let actual = make_e_from_bool(&bump, b);
1219///     assert_eq!(actual, expected);
1220/// }
1221/// ```
1222///
1223/// You can use `#[by_ref]` attribute for all arguments of your test and not just for fixture
1224/// but also for cases, values and files.
1225///
1226/// ## Putting all Together
1227///
1228/// All these features can be used together with a mixture of fixture variables,
1229/// fixed cases and bunch of values. For instance, you might need two
1230/// test cases which test for panics, one for a logged in user and one for a guest user.
1231///
1232/// ```rust
1233/// # enum User { Guest, Logged, }
1234/// # impl User { fn logged(_n: &str, _d: &str, _w: &str, _s: &str) -> Self { Self::Logged } }
1235/// # struct Item {}
1236/// # trait Repository { fn find_items(&self, user: &User, query: &str) -> Result<Vec<Item>, String> { Err("Invalid query error".to_owned()) } }
1237/// # #[derive(Default)] struct InMemoryRepository {}
1238/// # impl Repository for InMemoryRepository {}
1239///
1240/// use rstest::*;
1241///
1242/// #[fixture]
1243/// fn repository() -> InMemoryRepository {
1244///     let mut r = InMemoryRepository::default();
1245///     // fill repository with some data
1246///     r
1247/// }
1248///
1249/// #[fixture]
1250/// fn alice() -> User {
1251///     User::logged("Alice", "2001-10-04", "London", "UK")
1252/// }
1253///
1254/// #[rstest]
1255/// #[case::authorized_user(alice())] // We can use `fixture` also as standard function
1256/// #[case::guest(User::Guest)]   // We can give a name to every case : `guest` in this case
1257/// #[should_panic(expected = "Invalid query error")] // We would test a panic
1258/// fn should_be_invalid_query_error(
1259///     repository: impl Repository,
1260///     #[case] user: User,
1261///     #[values("     ", "^%$some#@invalid!chars", ".n.o.d.o.t.s.")] query: &str,
1262///     query: &str
1263/// ) {
1264///     repository.find_items(&user, query).unwrap();
1265/// }
1266/// ```
1267///
1268/// ## Ignoring Arguments
1269///
1270/// Sometimes, you may want to inject and use fixtures not managed by rstest
1271/// (e.g. db connection pools for sqlx tests).
1272///
1273/// In these cases, you can use the `#[ignore]` attribute to ignore the additional
1274/// parameter and let another crate take care of it:
1275///
1276/// ```rust, ignore
1277/// use rstest::*;
1278/// use sqlx::*;
1279///
1280/// #[fixture]
1281/// fn my_fixture() -> i32 { 42 }
1282///
1283/// #[rstest]
1284/// #[sqlx::test]
1285/// async fn test_db(my_fixture: i32, #[ignore] pool: PgPool) {
1286///     assert_eq!(42, injected);
1287///     // do stuff with the connection pool
1288/// }
1289/// ```
1290///
1291///
1292/// ## Trace Input Arguments
1293///
1294/// Sometimes can be very helpful to print all test's input arguments. To
1295/// do it you can use the `#[trace]` function attribute that you can apply
1296/// to all cases or just to some of them.
1297///
1298/// ```
1299/// use rstest::*;
1300///
1301/// #[fixture]
1302/// fn injected() -> i32 { 42 }
1303///
1304/// #[rstest]
1305/// #[trace]
1306/// fn the_test(injected: i32) {
1307///     assert_eq!(42, injected)
1308/// }
1309/// ```
1310///
1311/// Will print an output like
1312///
1313/// ```bash
1314/// Testing started at 14.12 ...
1315/// ------------ TEST ARGUMENTS ------------
1316/// injected = 42
1317/// -------------- TEST START --------------
1318///
1319///
1320/// Expected :42
1321/// Actual   :43
1322/// ```
1323/// But
1324/// ```
1325/// # use rstest::*;
1326/// #[rstest]
1327/// #[case(1)]
1328/// #[trace]
1329/// #[case(2)]
1330/// fn the_test(#[case] v: i32) {
1331///     assert_eq!(0, v)
1332/// }
1333/// ```
1334/// will trace just `case_2` input arguments.
1335///
1336/// If you want to trace input arguments but skip some of them that don't
1337/// implement the `Debug` trait, you can also use the
1338/// `#[notrace]` argument attribute to skip them:
1339///
1340/// ```
1341/// # use rstest::*;
1342/// # struct Xyz;
1343/// # struct NoSense;
1344/// #[rstest]
1345/// #[trace]
1346/// fn the_test(injected: i32, #[notrace] xyz: Xyz, #[notrace] have_no_sense: NoSense) {
1347///     assert_eq!(42, injected)
1348/// }
1349/// ```
1350/// # Old _compact_ syntax
1351///
1352/// `rstest` support also a syntax where all options and configuration can be write as
1353/// `rstest` attribute arguments. This syntax is a little less verbose but make
1354/// composition harder: for instance try to add some cases to a `rstest_reuse` template
1355/// is really hard.
1356///
1357/// So we'll continue to maintain the old syntax for a long time but we strongly encourage
1358/// to switch your test in the new form.
1359///
1360/// Anyway, here we recall this syntax and rewrite the previous example in the _compact_ form.
1361///
1362/// ```text
1363/// rstest(
1364///     arg_1,
1365///     ...,
1366///     arg_n[,]
1367///     [::attribute_1[:: ... [::attribute_k]]]
1368/// )
1369/// ```
1370/// Where:
1371///
1372/// - `arg_i` could be one of the follow
1373///   - `ident` that match to one of function arguments for parametrized cases
1374///   - `case[::description](v1, ..., vl)` a test case
1375///   - `fixture(v1, ..., vl) [as argument_name]` where fixture is the injected
1376///     fixture and argument_name (default use fixture) is one of function arguments
1377///     that and `v1, ..., vl` is a partial list of fixture's arguments
1378///   - `ident => [v1, ..., vl]` where `ident` is one of function arguments and
1379///     `v1, ..., vl` is a list of values for ident
1380/// - `attribute_j` a test attribute like `trace` or `notrace`
1381///
1382/// ## Fixture Arguments
1383///
1384/// ```
1385/// # struct User(String, u8);
1386/// # impl User { fn name(&self) -> &str {&self.0} }
1387/// # use rstest::*;
1388/// #
1389/// # #[fixture]
1390/// # fn user(
1391/// #     #[default("Alice")] name: impl AsRef<str>,
1392/// #     #[default(22)] age: u8
1393/// # ) -> User { User(name.as_ref().to_owned(), age) }
1394/// #
1395/// #[rstest(user("Bob"))]
1396/// fn check_user(user: User) {
1397///     assert_eq("Bob", user.name())
1398/// }
1399/// ```
1400///
1401/// ## Fixture Rename
1402/// ```
1403/// # use rstest::*;
1404/// #[fixture]
1405/// fn long_and_boring_descriptive_name() -> i32 { 42 }
1406///
1407/// #[rstest(long_and_boring_descriptive_name as short)]
1408/// fn the_test(short: i32) {
1409///     assert_eq!(42, short)
1410/// }
1411/// ```
1412///
1413/// ## Parametrized
1414///
1415/// ```
1416/// # use rstest::*;
1417/// #[rstest(input, expected,
1418///     case::zero_base_case(0, 0),
1419///     case::one_base_case(1, 1),
1420///     case(2, 1),
1421///     case(3, 2),
1422///     #[should_panic]
1423///     case(4, 42)
1424/// )]
1425/// fn fibonacci_test(input: u32, expected: u32) {
1426///     assert_eq!(expected, fibonacci(input))
1427/// }
1428///
1429/// # fn fibonacci(input: u32) -> u32 {
1430/// #     match input {
1431/// #         0 => 0,
1432/// #         1 => 1,
1433/// #         n => fibonacci(n - 2) + fibonacci(n - 1)
1434/// #     }
1435/// # }
1436/// ```
1437///
1438/// ## Values Lists
1439///
1440/// ```
1441/// # use rstest::*;
1442/// # fn is_valid(input: &str) -> bool { true }
1443///
1444/// #[rstest(
1445///     input => ["John", "alice", "My_Name", "Zigy_2001"]
1446/// )]
1447/// fn should_be_valid(input: &str) {
1448///     assert!(is_valid(input))
1449/// }
1450/// ```
1451///

1458/// #[rstest(::trace::notrace(xzy, have_no_sense))]
1459/// fn the_test(injected: i32, xyz: Xyz, have_no_sense: NoSense) {
1460///     assert_eq!(42, injected)
1461/// }
