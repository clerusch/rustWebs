37/// Low-level translation interface for definitions and declarations.
38///
39/// This trait allows you to walk over the structure of `Program`
40/// element by element. Each `visit_` method defines the callback
41/// for what to do when a particular element is encountered.
42///
43/// Any unimplemented `visit_` methods have a default which traverses
44/// all substructures in the element. Since this traversal does not happen
45/// automatically, if you override a `visit_` method, you must either
46/// traverse the element manually, or call the corresponding `walk_` method
47/// which does the default traversal.
48///
49/// Also note that the declarations are garuanteed to be visited in the
50/// following order by type: `Include`, `Def`, `QReg`, `CReg`, `Stmt`.
51/// The error type is provided to allow you to fail out at any time by
52/// returning an `Err` variant.
53///
54/// For example, to get the names of all gate definitions in a program,
55/// you might do the following:
56/// ```ignore
57/// struct DefFinder;
58///
59/// impl ProgramVisitor for DefFinder {
60///     fn visit_gate_def(
61///         &mut self,
62///         name: &Span<Symbol>,
63///         params: &[Span<Symbol>],
64///         args: &[Span<Symbol>],
65///         body: &[Span<Stmt>]
66///     ) {
67///         println!("Found definition of `{}`.", name.inner);
68///         self.walk_gate_def(name, params, args, body);
69///     }
70/// }
71///
72/// fn main() {
73///     let program = ...; // aquire a program from somewhere
74///     DefFinder.visit_program(&program);
75/// }
76/// ```
77#[allow(unused_variables)]
78pub trait ProgramVisitor {
79    type Error;
80
81    fn visit_program(&mut self, program: &Program) -> Result<(), Self::Error> {
82        self.walk_program(program)
83    }
84
85    fn walk_program(&mut self, program: &Program) -> Result<(), Self::Error> {
86        for decl in &program.decls {
87            if matches!(&*decl.inner, Decl::Include { .. }) {
88                self.visit_decl(decl)?;
89            }
90        }
91
92        for decl in &program.decls {
93            let decl = decl;
94            if matches!(&*decl.inner, Decl::Def { .. }) {
95                self.visit_decl(decl)?;
96            }
97        }
98
99        for decl in &program.decls {
100            if matches!(&*decl.inner, Decl::QReg { .. }) {
101                self.visit_decl(decl)?;
102            }
103        }
104
105        for decl in &program.decls {
106            if matches!(&*decl.inner, Decl::CReg { .. }) {
107                self.visit_decl(decl)?;
108            }
109        }
110
111        for decl in &program.decls {
112            if matches!(&*decl.inner, Decl::Stmt(..)) {
113                self.visit_decl(decl)?;
114            }
115        }
116
117        Ok(())
118    }
119
120    fn visit_decl(&mut self, decl: &Span<Decl>) -> Result<(), Self::Error> {
121        self.walk_decl(decl)
122    }
123
124    fn walk_decl(&mut self, decl: &Span<Decl>) -> Result<(), Self::Error> {
125        match &*decl.inner {
126            Decl::Include { file } => self.visit_include(file),
127            Decl::QReg { reg } => self.visit_qreg(reg),
128            Decl::CReg { reg } => self.visit_creg(reg),
129            Decl::Def {
130                name,
131                params,
132                args,
133                body,
134            } => match body {
135                None => self.visit_opaque_def(name, &params, &args),
136                Some(body) => self.visit_gate_def(name, &params, &args, &body),
137            },
138            Decl::Stmt(stmt) => self.visit_stmt(stmt),
139        }
140    }
141
142    fn visit_include(&mut self, file: &Span<Symbol>) -> Result<(), Self::Error> {
143        Ok(())
144    }
145
146    fn visit_qreg(&mut self, reg: &Span<Reg>) -> Result<(), Self::Error> {
147        Ok(())
148    }
149
150    fn visit_creg(&mut self, reg: &Span<Reg>) -> Result<(), Self::Error> {
151        Ok(())
152    }
153
154    fn visit_opaque_def(
155        &mut self,
156        name: &Span<Symbol>,
157        params: &[Span<Symbol>],
158        args: &[Span<Symbol>],
159    ) -> Result<(), Self::Error> {
160        Ok(())
161    }
162
163    fn visit_gate_def(
164        &mut self,
165        name: &Span<Symbol>,
166        params: &[Span<Symbol>],
167        args: &[Span<Symbol>],
168        body: &[Span<Stmt>],
169    ) -> Result<(), Self::Error> {
170        self.walk_gate_def(name, params, args, body)
171    }
172
173    fn walk_gate_def(
174        &mut self,
175        name: &Symbol,
176        params: &[Span<Symbol>],
177        args: &[Span<Symbol>],
178        body: &[Span<Stmt>],
179    ) -> Result<(), Self::Error> {
180        for stmt in body {
181            self.visit_stmt(stmt)?
182        }
183        Ok(())
184    }
185
186    fn visit_stmt(&mut self, stmt: &Span<Stmt>) -> Result<(), Self::Error> {
187        self.walk_stmt(stmt)
188    }
189
190    fn walk_stmt(&mut self, stmt: &Span<Stmt>) -> Result<(), Self::Error> {
191        match &*stmt.inner {
192            Stmt::Barrier { regs } => self.visit_barrier(&regs),
193            Stmt::Measure { from, to } => self.visit_measure(from, to),
194            Stmt::Reset { reg } => self.visit_reset(reg),
195            Stmt::CX { copy, xor } => self.visit_cx(copy, xor),
196            Stmt::U {
197                theta,
198                phi,
199                lambda,
200                reg,
201            } => self.visit_u(theta, phi, lambda, reg),
202            Stmt::Gate { name, params, args } => self.visit_gate(name, &params, &args),
203            Stmt::Conditional { reg, val, then } => self.visit_conditional(reg, val, then),
204        }
205    }
206
207    fn visit_barrier(&mut self, regs: &[Span<Reg>]) -> Result<(), Self::Error> {
208        Ok(())
209    }
210
211    fn visit_measure(&mut self, from: &Span<Reg>, to: &Span<Reg>) -> Result<(), Self::Error> {
212        Ok(())
213    }
214
215    fn visit_reset(&mut self, reg: &Span<Reg>) -> Result<(), Self::Error> {
216        Ok(())
217    }
218
219    fn visit_cx(&mut self, copy: &Span<Reg>, xor: &Span<Reg>) -> Result<(), Self::Error> {
220        Ok(())
221    }
222
223    fn visit_u(
224        &mut self,
225        theta: &Span<Expr>,
226        phi: &Span<Expr>,
227        lambda: &Span<Expr>,
228        reg: &Span<Reg>,
229    ) -> Result<(), Self::Error> {
230        Ok(())
231    }
232
233    fn visit_gate(
234        &mut self,
235        name: &Span<Symbol>,
236        params: &[Span<Expr>],
237        args: &[Span<Reg>],
238    ) -> Result<(), Self::Error> {
239        Ok(())
240    }
241
242    fn visit_conditional(
243        &mut self,
244        reg: &Span<Reg>,
245        val: &Span<u64>,
246        then: &Span<Stmt>,
247    ) -> Result<(), Self::Error> {
248        self.walk_conditional(reg, val, then)
249    }
250
251    fn walk_conditional(
252        &mut self,
253        reg: &Span<Reg>,
254        val: &Span<u64>,
255        then: &Span<Stmt>,
256    ) -> Result<(), Self::Error> {
257        self.visit_stmt(then)
258    }
259}

261/// Companion trait to `ProgramVisitor`.
262///
263/// This visits an `Expr` piece by piece and uses the required
264/// callbacks to compute a return value.
265/// The type of the return value is given by `ExprVisitor::Output`.
266///
267/// The default `visit_` methods just traverse the structure and
268/// call the callbacks where necessary, so if you override them,
269/// you must do the traversal yourself.
270///
271/// For example, if you wanted to compute the maximum integer
272/// value that appeared in an expression, you could do:
273/// ```ignore
274/// struct MaxFinder;
275///
276/// impl ExprVisitor for MaxFinder {
277///     type Output = u64;
278///
279///     fn binop(&mut self, _: Binop, a: u64, b: u64) -> u64 { a.max(b) }
280///     fn unop(&mut self, _: Unop, a: u64) -> u64 { a }
281///     fn lookup(&mut self, _: &Symbol) -> u64 { 0 }
282///     fn pi(&mut self) -> u64 { 0 }
283///     fn int(&mut self, val: u64) -> u64 { val }
284///     fn real(&mut self, val: f32) -> u64 { 0 }
285/// }
286///
287/// fn main() {
288///     let expr = ...; // acquire an expression from somewhere.
289///     println!("max = {}", MaxFinder.visit_expr(&expr));
290/// }
291/// ```
292pub trait ExprVisitor {
293    type Output;
294
295    fn binop(&mut self, op: Binop, a: Self::Output, b: Self::Output) -> Self::Output;
296    fn unop(&mut self, op: Unop, a: Self::Output) -> Self::Output;
297    fn lookup(&mut self, var: &Symbol) -> Self::Output;
298    fn pi(&mut self) -> Self::Output;
299    fn int(&mut self, val: u64) -> Self::Output;
300    fn real(&mut self, val: f32) -> Self::Output;
301
302    fn visit_expr(&mut self, expr: &Span<Expr>) -> Self::Output {
303        match &*expr.inner {
304            Expr::Pi => self.pi(),
305            Expr::Int(val) => self.int(*val),
306            Expr::Real(val) => self.real(*val),
307            Expr::Var(symbol) => self.visit_var(symbol),
308            Expr::Add(a, b) => self.visit_binop(Binop::Add, a, b),
309            Expr::Sub(a, b) => self.visit_binop(Binop::Sub, a, b),
310            Expr::Mul(a, b) => self.visit_binop(Binop::Mul, a, b),
311            Expr::Div(a, b) => self.visit_binop(Binop::Div, a, b),
312            Expr::Pow(a, b) => self.visit_binop(Binop::Pow, a, b),
313            Expr::Sin(a) => self.visit_unop(Unop::Sin, a),
314            Expr::Cos(a) => self.visit_unop(Unop::Cos, a),
315            Expr::Tan(a) => self.visit_unop(Unop::Tan, a),
316            Expr::Exp(a) => self.visit_unop(Unop::Exp, a),
317            Expr::Ln(a) => self.visit_unop(Unop::Ln, a),
318            Expr::Sqrt(a) => self.visit_unop(Unop::Sqrt, a),
319            Expr::Neg(a) => self.visit_unop(Unop::Neg, a),
320        }
321    }
322
323    fn visit_var(&mut self, var: &Symbol) -> Self::Output {
324        self.lookup(var)
325    }
326
327    fn visit_binop(&mut self, op: Binop, a: &Span<Expr>, b: &Span<Expr>) -> Self::Output {
328        let a = self.visit_expr(a);
329        let b = self.visit_expr(b);
330        self.binop(op, a, b)
331    }
332
333    fn visit_unop(&mut self, op: Unop, a: &Span<Expr>) -> Self::Output {
334        let a = self.visit_expr(a);
335        self.unop(op, a)
336    }
337}

343// These are the possible errors that can occur
344// when evaluating a parameter. Unfortunately, there
345// are still some we don't catch, for example
346// certain uses of constants that don't fit in a i64.
347// Additionally, a lot of errors will just show up as
348// `ApproximateFail(NaN)`
349#[derive(Debug)]
350enum EvalError {
351    DivideByZero,
352    ApproximateFail(f32),
353    OverflowError,
354}

356impl<'a> ExprVisitor for FrameEvaluator<'a> {
357    type Output = Result<Value, EvalError>;
358
359    fn pi(&mut self) -> Self::Output {
360        Ok(Value::PI)
361    }
362
363    fn int(&mut self, val: u64) -> Self::Output {
364        Ok(Value::int(val as i64))
365    }
366
367    fn real(&mut self, val: f32) -> Self::Output {
368        Value::from_float(val).ok_or(EvalError::ApproximateFail(val))
369    }
370
371    fn unop(&mut self, op: Unop, a: Self::Output) -> Self::Output {
372        let a = a?;
373        match op {
374            Unop::Neg => a.neg_internal(),
375            Unop::Sin => a.sin_internal(),
376            Unop::Cos => a.cos_internal(),
377            Unop::Tan => a.tan_internal(),
378            Unop::Exp => a.exp_internal(),
379            Unop::Ln => a.ln_internal(),
380            Unop::Sqrt => a.sqrt_internal(),
381        }
382    }
383
384    fn binop(&mut self, op: Binop, a: Self::Output, b: Self::Output) -> Self::Output {
385        let (a, b) = (a?, b?);
386        match op {
387            Binop::Add => a.add_internal(b),
388            Binop::Sub => a.sub_internal(b),
389            Binop::Mul => a.mul_internal(b),
390            Binop::Div => a.div_internal(b),
391            Binop::Pow => a.pow_internal(b),
392        }
393    }
394
395    fn lookup(&mut self, var: &Symbol) -> Self::Output {
396        Ok(self.frame.params[var])
397    }
398}

415/// High-level translation interface.
416///
417/// This structure is used to turn a program into a linear
418/// list of gates. It uses a value that implements `GateWriter`
419/// to output primitive gates to some medium.
420///
421/// This can be used by constructing one with `Linearize::new`
422/// with your chosen `GateWriter`, and then using the `ProgramVisitor`
423/// impl to call `.visit_program` on your program. The `depth` argument
424/// to `Linearize::new` determines how many layers of definitions
425/// it will expand.
426///
427/// Note that it is assumed you have type-checked your program first.
428/// If you haven't you may get garbage output / random panics. If
429/// you have type-checked it, you shouldn't get any panics.
430///
431/// Example:
432/// ```ignore
433/// struct GatePrinter;
434///
435/// impl GateWriter for GatePrinter {
436///     type Error = std::convert::Infallible;
437/// 
438///     fn initialize(&mut self, _: &[Symbol], _: &[Symbol]) -> Result<(), Self::Error> {
439///         Ok(())
440///     }
441///
442///     fn write_cx(&mut self, copy: usize, xor: usize) -> Result<(), Self::Error> {
443///         println!("cx {copy} {xor}");
444///         Ok(())
445///     }
446///
447///     fn write_u(&mut self, theta: Value, phi: Value, lambda: Value, reg: usize) -> Result<(), Self::Error> {
448///         println!("u({theta}, {phi}, {lambda}) {reg}");
449///         Ok(())
450///     }
451///
452///     fn write_opaque(&mut self, name: &Symbol, _: &[Value], _: &[usize]) -> Result<(), Self::Error> {
453///         println!("opaque gate {}", name);
454///         Ok(())
455///     }
456///
457///     fn write_barrier(&mut self, _: &[usize]) -> Result<(), Self::Error> {
458///         Ok(())
459///     }
460///
461///     fn write_measure(&mut self, from: usize, to: usize) -> Result<(), Self::Error> {
462///         println!("measure {} -> {}", from, to);
463///         Ok(())
464///     }
465///
466///     fn write_reset(&mut self, reg: usize) -> Result<(), Self::Error> {
467///         println!("reset {reg}");
468///         Ok(())
469///     }
470///
471///     fn start_conditional(&mut self, reg: usize, count: usize, value: usize) -> Result<(), Self::Error> {
472///         println!("if ({reg}:{count} == {value}) {{");
473///         Ok(())
474///     }
475///
476///     fn end_conditional(&mut self) -> Result<(), Self::Error> {
477///         println!("}}");
478///         Ok(())
479///     }
480/// }
481///
482/// fn main() {
483///     let program = ...; // acquire a program from somewhere.
484///     program.type_check().unwrap(); // make sure to type check.
485///     let mut l = Linearize::new(GatePrinter, usize::MAX);
486///     l.visit_program(&program).unwrap();
487/// }
488/// ```
489pub struct Linearize<T> {
490    next_qid: u64,
491    next_cid: u64,
492    depth: usize,
493    defs: HashMap<Symbol, Rc<Definition>>,
494    stack: Vec<Frame>,
495    frame: Frame,
496    initialized: bool,
497    writer: T,
498}

500/// Output format from `Linearize`.
501///
502/// This trait is used by `Linearize` to actually output the
503/// linearized program. It contains various methods to output
504/// a primitive operation. Qubits and bits are referenced by
505/// consecutive integers starting from zero, and parameters
506/// are provided as `Value`s.
507///
508/// The only non-obvious methods are `initialize`, which provides
509/// the names of qubits and bits to the backend, and is called
510/// exactly once, before any other function. `start/end_conditional`
511/// is called before and after any statements that are intended
512/// to be conditional on a classical value.
513pub trait GateWriter: Sized {
514    type Error: std::error::Error + 'static;
515
516    fn initialize(&mut self, qubits: &[Symbol], bits: &[Symbol]) -> Result<(), Self::Error>;
517    fn write_cx(&mut self, copy: usize, xor: usize) -> Result<(), Self::Error>;
518    fn write_u(&mut self, theta: Value, phi: Value, lambda: Value, reg: usize) -> Result<(), Self::Error>;
519    fn write_opaque(&mut self, name: &Symbol, params: &[Value], args: &[usize]) -> Result<(), Self::Error>;
520    fn write_barrier(&mut self, regs: &[usize]) -> Result<(), Self::Error>;
521    fn write_measure(&mut self, from: usize, to: usize) -> Result<(), Self::Error>;
522    fn write_reset(&mut self, reg: usize) -> Result<(), Self::Error>;
523    fn start_conditional(&mut self, reg: usize, count: usize, val: u64) -> Result<(), Self::Error>;
524    fn end_conditional(&mut self) -> Result<(), Self::Error>;
525}

527impl<T: GateWriter> Linearize<T> {
528    pub fn new(writer: T, depth: usize) -> Linearize<T> {
529        Linearize {
530            next_qid: 0,
531            next_cid: 0,
532            depth,
533            defs: HashMap::new(),
534            stack: Vec::new(),
535            frame: Frame {
536                name: Symbol::new("<toplevel>"),
537                def_name: Symbol::new("<toplevel>"),
538                call: None,
539                qregs: HashMap::new(),
540                cregs: HashMap::new(),
541                params: HashMap::new(),
542            },
543            initialized: false,
544            writer,
545        }
546    }
547
548    // Check that these args are distinct.
549    fn assert_different(
550        &mut self,
551        a: usize,
552        b: usize,
553        aspan: FileSpan,
554        bspan: FileSpan,
555    ) -> Result<(), LinearizeError> {
556        if a == b {
557            self.err(Err(LinearizeErrorKind::OverlappingRegs {
558                a: aspan,
559                b: bspan,
560            }))
561        } else {
562            Ok(())
563        }
564    }
565
566    // Convert an error kind into an actual error
567    // by recording a backtrace.
568    fn err<V>(&self, e: Result<V, LinearizeErrorKind>) -> Result<V, LinearizeError> {
569        e.map_err(|kind| {
570            let mut stack: Vec<_> = self
571                .stack
572                .iter()
573                .filter_map(|frame| Some((frame.name.clone(), frame.call?)))
574                .collect();
575
576            if let Some(call) = self.frame.call {
577                stack.push((self.frame.name.clone(), call));
578            }
579
580            LinearizeError { stack, kind }
581        })
582    }
583
584    // Convert a register reference to a qubit number and a size.
585    // The qubit number corresponds to the base of this reference,
586    // and the remaining qubits in the register are the next
587    // consecutive size bits.
588    fn resolve_qreg(&self, reg: &Reg) -> (u64, u64) {
589        match reg.index {
590            None => self.frame.qregs[&reg.name],
591            Some(idx) => {
592                let base = self.frame.qregs[&reg.name].0;
593                (base + idx, 1)
594            }
595        }
596    }
597
598    fn resolve_creg(&self, reg: &Reg) -> (u64, u64) {
599        match reg.index {
600            None => self.frame.cregs[&reg.name],
601            Some(idx) => {
602                let base = self.frame.cregs[&reg.name].0;
603                (base + idx, 1)
604            }
605        }
606    }
607}

609impl<T: GateWriter> ProgramVisitor for Linearize<T> {
610    type Error = LinearizeError;
611
612    // Since ProgramVisitor guarantees that all the definitions
613    // and register declarations are visited first, this is ok
614    // to do at the same time as everything else.
615    fn visit_opaque_def(
616        &mut self,
617        name: &Span<Symbol>,
618        params: &[Span<Symbol>],
619        args: &[Span<Symbol>],
620    ) -> Result<(), Self::Error> {
621        let def = Definition {
622            args: args.iter().map(|s| s.to_symbol()).collect(),
623            params: params.iter().map(|s| s.to_symbol()).collect(),
624            gates: None,
625        };
626
627        self.defs.insert(name.to_symbol(), Rc::new(def));
628
629        Ok(())
630    }
631
632    fn visit_gate_def(
633        &mut self,
634        name: &Span<Symbol>,
635        params: &[Span<Symbol>],
636        args: &[Span<Symbol>],
637        gates: &[Span<Stmt>],
638    ) -> Result<(), Self::Error> {
639        let def = Definition {
640            args: args.iter().map(|s| s.to_symbol()).collect(),
641            params: params.iter().map(|s| s.to_symbol()).collect(),
642            gates: Some(gates.to_vec()),
643        };
644
645        self.defs.insert(name.to_symbol(), Rc::new(def));
646
647        Ok(())
648    }
649
650    fn visit_qreg(&mut self, reg: &Span<Reg>) -> Result<(), Self::Error> {
651        // Allocate some registers to the current frame
652        // and bump `next_qid` by the appropriate amount,
653        // so that the qubits are layed out consecutively.
654        let size = reg.index.unwrap_or(1);
655        self.frame
656            .qregs
657            .insert(reg.name.to_symbol(), (self.next_qid, size));
658        self.next_qid += size;
659
660        Ok(())
661    }
662
663    fn visit_creg(&mut self, reg: &Span<Reg>) -> Result<(), Self::Error> {
664        let size = reg.index.unwrap_or(1);
665        self.frame
666            .cregs
667            .insert(reg.name.to_symbol(), (self.next_cid, size));
668        self.next_cid += size;
669
670        Ok(())
671    }
672
673    fn visit_stmt(&mut self, stmt: &Span<Stmt>) -> Result<(), Self::Error> {
674        // Since this is first called after all register declarations,
675        // now is a good time to initialize the backend.
676        if !self.initialized {
677            let mut qubits = vec![Symbol::new(""); self.next_qid as usize];
678            let mut bits = vec![Symbol::new(""); self.next_cid as usize];
679
680            for (name, (base, size)) in &self.frame.qregs {
681                for offset in 0..*size {
682                    let n = Symbol::new(format!("{name}[{offset}]"));
683                    qubits[(*base + offset) as usize] = n;
684                }
685            }
686
687            for (name, (base, size)) in &self.frame.cregs {
688                for offset in 0..*size {
689                    let n = Symbol::new(format!("{name}[{offset}]"));
690                    bits[(*base + offset) as usize] = n;
691                }
692            }
693
694            let err = self.writer.initialize(&qubits, &bits).map_err(|error| {
695                LinearizeErrorKind::WriterError {
696                    error: Box::new(error)
697                }
698            });
699            self.err(err)?;
700
701            self.initialized = true;
702        }
703
704        self.walk_stmt(stmt)
705    }

707    fn visit_barrier(&mut self, regs: &[Span<Reg>]) -> Result<(), Self::Error> {
708        // Get all regs referenced by these expressions
709        // regardless of size of matching.
710        let mut args = HashSet::new();
711        for reg in regs {
712            let (base, size) = self.resolve_qreg(reg);
713            for offset in 0..size {
714                args.insert((base + offset) as usize);
715            }
716        }
717
718        let args = args.drain().collect::<Vec<_>>();
719        let err = self.writer.write_barrier(&args).map_err(|error| {
720            LinearizeErrorKind::WriterError {
721                error: Box::new(error)
722            }
723        });
724        self.err(err)?;
725
726        Ok(())
727    }
728
729    fn visit_measure(&mut self, from: &Span<Reg>, to: &Span<Reg>) -> Result<(), Self::Error> {
730        let (fbase, fsize) = self.resolve_qreg(from);
731        let (tbase, tsize) = self.resolve_creg(to);
732
733        if fsize == tsize {
734            // Many - many
735            for offset in 0..fsize {
736                let err = self.writer
737                    .write_measure((fbase + offset) as usize, (tbase + offset) as usize)
738                    .map_err(|error| {
739                        LinearizeErrorKind::WriterError {
740                            error: Box::new(error)
741                        }
742                    });
743                self.err(err)?;
744            }
745        } else if fsize == 1 {
746            // One - many
747            for offset in 0..tsize {
748                let err = self.writer
749                    .write_measure(fbase as usize, (tbase + offset) as usize)
750                    .map_err(|error| {
751                        LinearizeErrorKind::WriterError {
752                            error: Box::new(error)
753                        }
754                    });
755                self.err(err)?;
756            }
757        } else {
758            // Many - one
759            for offset in 0..fsize {
760                let err = self.writer
761                    .write_measure((fbase + offset) as usize, tbase as usize)
762                    .map_err(|error| {
763                        LinearizeErrorKind::WriterError {
764                            error: Box::new(error)
765                        }
766                    });
767                self.err(err)?;
768            }
769        }
770
771        Ok(())
772    }

774    fn visit_reset(&mut self, reg: &Span<Reg>) -> Result<(), Self::Error> {
775        let (base, size) = self.resolve_qreg(reg);
776        for offset in 0..size {
777            let err = self.writer.write_reset((base + offset) as usize)
778            .map_err(|error| {
779                LinearizeErrorKind::WriterError {
780                    error: Box::new(error)
781                }
782            });
783            self.err(err)?;
784        }
785
786        Ok(())
787    }

789    fn visit_cx(&mut self, copy: &Span<Reg>, xor: &Span<Reg>) -> Result<(), Self::Error> {
790        let (cbase, csize) = self.resolve_qreg(copy);
791        let (xbase, xsize) = self.resolve_qreg(xor);
792
793        // Same as with visit_measure but this time they have to be
794        // distinct as the CX of two identical gates is not well defined.
795        if csize == xsize {
796            for offset in 0..csize {
797                self.assert_different(
798                    (cbase + offset) as usize,
799                    (xbase + offset) as usize,
800                    copy.span,
801                    xor.span,
802                )?;
803                let err = self.writer
804                    .write_cx((cbase + offset) as usize, (xbase + offset) as usize)
805                    .map_err(|error| {
806                        LinearizeErrorKind::WriterError {
807                            error: Box::new(error)
808                        }
809                    });
810                self.err(err)?;
811            }
812        } else if csize == 1 {
813            for offset in 0..xsize {
814                self.assert_different(
815                    cbase as usize,
816                    (xbase + offset) as usize,
817                    copy.span,
818                    xor.span,
819                )?;
820                let err = self.writer
821                    .write_cx(cbase as usize, (xbase + offset) as usize)
822                    .map_err(|error| {
823                        LinearizeErrorKind::WriterError {
824                            error: Box::new(error)
825                        }
826                    });
827                self.err(err)?;
828            }
829        } else {
830            for offset in 0..csize {
831                self.assert_different(
832                    (cbase + offset) as usize,
833                    xbase as usize,
834                    copy.span,
835                    xor.span,
836                )?;
837                let err = self.writer
838                    .write_cx((cbase + offset) as usize, xbase as usize)
839                    .map_err(|error| {
840                        LinearizeErrorKind::WriterError {
841                            error: Box::new(error)
842                        }
843                    });
844                self.err(err)?;
845            }
846        }
847
848        Ok(())
849    }

851    fn visit_u(
852        &mut self,
853        theta: &Span<Expr>,
854        phi: &Span<Expr>,
855        lambda: &Span<Expr>,
856        reg: &Span<Reg>,
857    ) -> Result<(), Self::Error> {
858        // Evaluate the parameters in this frame.
859        let mut eval = FrameEvaluator { frame: &self.frame };
860
861        let theta = self.err(eval.visit_expr(theta).map_err(|e| match e {
862            // Add the span information to the raw errors.
863            EvalError::DivideByZero => LinearizeErrorKind::DivideByZero { span: theta.span },
864            EvalError::ApproximateFail(value) => LinearizeErrorKind::ApproximateFail {
865                span: theta.span,
866                value,
867            },
868            EvalError::OverflowError => LinearizeErrorKind::NumericalOverflow { span: theta.span },
869        }))?;
870        let phi = self.err(eval.visit_expr(phi).map_err(|e| match e {
871            EvalError::DivideByZero => LinearizeErrorKind::DivideByZero { span: phi.span },
872            EvalError::ApproximateFail(value) => LinearizeErrorKind::ApproximateFail {
873                span: phi.span,
874                value,
875            },
876            EvalError::OverflowError => LinearizeErrorKind::NumericalOverflow { span: phi.span },
877        }))?;
878        let lambda = self.err(eval.visit_expr(lambda).map_err(|e| match e {
879            EvalError::DivideByZero => LinearizeErrorKind::DivideByZero { span: lambda.span },
880            EvalError::ApproximateFail(value) => LinearizeErrorKind::ApproximateFail {
881                span: lambda.span,
882                value,
883            },
884            EvalError::OverflowError => LinearizeErrorKind::NumericalOverflow { span: lambda.span },
885        }))?;
886
887        // Now do a unitary for each referenced qubit.
888        let (base, size) = self.resolve_qreg(reg);
889        for offset in 0..size {
890            let err = self.writer
891                .write_u(theta, phi, lambda, (base + offset) as usize)
892                .map_err(|error| {
893                    LinearizeErrorKind::WriterError {
894                        error: Box::new(error)
895                    }
896                });
897            self.err(err)?;
898        }
899
900        Ok(())
901    }

903    fn visit_gate(
904        &mut self,
905        name: &Span<Symbol>,
906        params: &[Span<Expr>],
907        args: &[Span<Reg>],
908    ) -> Result<(), Self::Error> {
909        let def = self.defs[name.as_symbol()].clone();
910
911        // We may need to push a new stack frame.
912        let mut frame = Frame {
913            name: self.frame.def_name.to_symbol(),
914            def_name: name.to_symbol(),
915            call: Some(name.span),
916            cregs: HashMap::new(),
917            qregs: HashMap::new(),
918            params: HashMap::new(),
919        };
920
921        // First, evaluate all the parameters.
922        let mut eval = FrameEvaluator { frame: &self.frame };
923
924        let mut values = Vec::new();
925        for (param, name) in params.iter().zip(&def.params) {
926            let value = self.err(eval.visit_expr(param).map_err(|e| match e {
927                EvalError::DivideByZero => LinearizeErrorKind::DivideByZero { span: param.span },
928                EvalError::ApproximateFail(value) => LinearizeErrorKind::ApproximateFail {
929                    span: param.span,
930                    value,
931                },
932                EvalError::OverflowError => {
933                    LinearizeErrorKind::NumericalOverflow { span: param.span }
934                }
935            }))?;
936
937            // If this is opaque, we will need the parameters as a list.
938            if def.gates.is_none() {
939                values.push(value);
940            } else {
941                // Otherwise, just add them to the new stack frame.
942                frame.params.insert(name.clone(), value);
943            }
944        }
945
946        // Find the maximum register size. Since these are matched,
947        // this is also the size of all non-single registers.
948        let size = args
949            .iter()
950            .map(|a| self.resolve_qreg(a).1)
951            .max()
952            .unwrap_or(0);
953        // Record which registers are fixed, and which are looped over.
954        let fixed = args
955            .iter()
956            .map(|a| self.resolve_qreg(a).1 == 1)
957            .collect::<Vec<_>>();
958        // Record the base addresses of all registers.
959        let mut argsn = args
960            .iter()
961            .map(|a| self.resolve_qreg(a).0 as usize)
962            .collect::<Vec<_>>();
963
964        if def.gates.is_some() {
965            // Push the current frame onto the stack and replace it with the new one.
966            self.stack.push(std::mem::replace(&mut self.frame, frame));
967        }
968
969        for _ in 0..size {
970            // If we have an opaque gate, we must have all the qubits distinct.
971            if def.gates.is_none() {
972                for i in 0..argsn.len() {
973                    for j in 0..i {
974                        self.assert_different(argsn[i], argsn[j], args[i].span, args[j].span)?;
975                    }
976                }
977            }
978
979            match &def.gates {
980                Some(gates) if self.stack.len() < self.depth => {
981                    // This gate has a body, so process it. First insert
982                    // all of the arguments as quantum registers of size one.
983                    for (name, arg) in def.args.iter().zip(&argsn) {
984                        self.frame.qregs.insert(name.clone(), (*arg as u64, 1));
985                    }
986
987                    // Then recurse on the body.
988                    for stmt in gates {
989                        self.visit_stmt(stmt)?;
990                    }
991                }
992                _ => {
993                    // If this is opaque (or we passed the depth limit), write it out straight away.
994                    let err = self.writer.write_opaque(name, &values, &argsn).map_err(|error| {
995                        LinearizeErrorKind::WriterError {
996                            error: Box::new(error)
997                        }
998                    });
999                    self.err(err)?;
1000                }
1001            }
1002
1003            // Advance all the large registers in lock-step.
1004            for i in 0..argsn.len() {
1005                if !fixed[i] {
1006                    argsn[i] += 1;
1007                }
1008            }
1009        }
1010
1011        // If we had a new stack frame, pop it off.
1012        if def.gates.is_some() {
1013            self.frame = self.stack.pop().unwrap();
1014        }
1015
1016        Ok(())
1017    }
1018
1019    fn visit_conditional(
1020        &mut self,
1021        reg: &Span<Reg>,
1022        val: &Span<u64>,
1023        then: &Span<Stmt>,
1024    ) -> Result<(), Self::Error> {
1025        let (base, size) = self.frame.qregs[&reg.name];
1026        let err =self.writer
1027            .start_conditional(base as usize, size as usize, **val).map_err(|error| {
1028                LinearizeErrorKind::WriterError {
1029                    error: Box::new(error)
1030                }
1031            });
1032        self.err(err)?;
1033        self.visit_stmt(then)?;
1034        let err = self.writer.end_conditional().map_err(|error| {
1035            LinearizeErrorKind::WriterError {
1036                error: Box::new(error)
1037            }
1038        });
1039        self.err(err)?;
1040        Ok(())
1041    }

1044/// An error that occured while linearizing a program.
1045/// This error contains both the call stack leading up
1046/// to this error, as well as the kind of error that occured.
1047/// `stack` is layed out as a list of function names and call
1048/// for those functions, from oldest to youngest (i.e the
1049/// current stack frame is the end of `stack`).
1050#[derive(Debug, Error)]
1051#[error("linearization error")]
1052pub struct LinearizeError {
1053    pub stack: Vec<(Symbol, FileSpan)>,
1054    #[source]
1055    pub kind: LinearizeErrorKind,
1056}
1057
1058/// The type of error that occurred while linearizing.
1059#[derive(Debug, Error)]
1060pub enum LinearizeErrorKind {
1061    /// A division by zero happened while computing this parameter.
1062    #[error("division by zero")]
1063    DivideByZero { span: FileSpan },
1064    /// A value could not be converted from float to rational
1065    /// while computing this parameter.
1066    #[error("float approximation fail")]
1067    ApproximateFail { span: FileSpan, value: f32 },
1068    /// This expression had an overflow.
1069    #[error("numerical overflow")]
1070    NumericalOverflow { span: FileSpan },
1071    /// A `CX` gate or opaque gate was called with non-distinct arguments.
1072    #[error("overlapping arguments")]
1073    OverlappingRegs {
1074        /// This argument overlaps with the other.
1075        a: FileSpan,
1076        /// This argument overlaps with the other.
1077        b: FileSpan,
1078    },
1079    #[error("{error}")]
1080    WriterError {
1081        error: Box<dyn std::error::Error>
1082    }
1083}
1084
1085#[cfg(feature = "ariadne")]
1086impl LinearizeError {
1087    pub fn to_report(&self) -> Report {
1088        let mut base = match self.kind {
1089            LinearizeErrorKind::DivideByZero { span } => {
1090                Report::build(ReportKind::Error, span.file, span.start)
1091                    .with_message("Division by zero in parameter evaluation")
1092                    .with_label(Label::new(span)
1093                        .with_message("This expression had an error.")
1094                        .with_color(ariadne::Color::Cyan))
1095            },
1096            LinearizeErrorKind::NumericalOverflow { span } => {
1097                Report::build(ReportKind::Error, span.file, span.start)
1098                    .with_message("Numerical overflow in parameter evaluation")
1099                    .with_label(Label::new(span)
1100                        .with_message("This expression had an error.")
1101                        .with_color(ariadne::Color::Cyan))
1102            },
1103            LinearizeErrorKind::ApproximateFail { value, span } => {
1104                Report::build(ReportKind::Error, span.file, span.start)
1105                    .with_message("Could not approximate float in parameter evaluation")
1106                    .with_label(Label::new(span)
1107                        .with_message(format!("Part of this expression evaluated to {}, which cannot be represented.", value))
1108                        .with_color(ariadne::Color::Cyan))
1109                    .with_note(concat!(
1110                        "Parameter values are represented in the form `a + bπ` for `a` and `b` rationals to preserve accuracy.",
1111                        "Not all floats can be represented in this form, however this usually indicates an error in your calculations as unrepresentable",
1112                        "values are not physically meaningful (they are either very large or invalid)."
1113                    ))
1114            },
1115            LinearizeErrorKind::OverlappingRegs { a, b } => {
1116                Report::build(ReportKind::Error, a.file, a.start)
1117                    .with_message("Overlapping arguments to primitive gate")
1118                    .with_label(Label::new(a)
1119                        .with_message("The register is first referenced here.")
1120                        .with_color(ariadne::Color::Cyan)
1121                        .with_order(0))
1122                    .with_label(Label::new(b)
1123                        .with_message("This overlaps with the previous reference.")
1124                        .with_color(ariadne::Color::Cyan)
1125                        .with_order(1))
1126            },
1127            LinearizeErrorKind::WriterError { ref error } => {
1128                Report::build(ReportKind::Error, 0usize, 0usize)
1129                    .with_message(format!("{}", error))
1130            }
1131        };
1132
1133        let mut cols = ariadne::ColorGenerator::new();
1134        for (i, (name, call)) in self.stack.iter().enumerate().rev() {
1135            let kind = match i {
1136                0 => "Top level frame".to_string(),
1137                _ => format!("Frame {i}"),
1138            };
1139
1140            let rest = if name.as_str() == "<toplevel>" {
1141                String::new()
1142            } else {
1143                format!(" in `{name}`")
1144            };
1145
1146            base = base.with_label(
1147                Label::new(*call)
1148                    .with_message(format!("{kind} originated here{rest}."))
1149                    .with_color(cols.next()),
1150            );
1151        }
1152
1153        base.finish()
1154    }
