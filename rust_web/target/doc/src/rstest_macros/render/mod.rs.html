38pub(crate) fn single(mut test: ItemFn, mut info: RsTestInfo) -> TokenStream {
39    test.apply_arguments(&mut info.arguments, &mut ());
40
41    let resolver = resolver::fixtures::get(&info.arguments, info.data.fixtures());
42
43    let args = test.sig.inputs.iter().cloned().collect::<Vec<_>>();
44    let attrs = std::mem::take(&mut test.attrs);
45    let asyncness = test.sig.asyncness;

53        asyncness,
54        Some(&test),
55        resolver,
56        &info,
57        &test.sig.generics,
58    )

61pub(crate) fn parametrize(mut test: ItemFn, info: RsTestInfo) -> TokenStream {
62    let mut arguments_info = info.arguments.clone();
63    test.apply_arguments(&mut arguments_info, &mut ());
64
65    let resolver_fixtures = resolver::fixtures::get(&info.arguments, info.data.fixtures());
66
67    let rendered_cases = cases_data(&info, test.sig.ident.span())
68        .map(|(name, attrs, resolver)| {
69            TestCaseRender::new(name, attrs, (resolver, &resolver_fixtures))
70        })
71        .map(|case| case.render(&test, &info))
72        .collect();
73
74    test_group(test, rendered_cases)
75}

77impl ValueList {
78    fn render(
79        &self,
80        test: &ItemFn,
81        resolver: &dyn Resolver,
82        attrs: &[syn::Attribute],
83        info: &RsTestInfo,
84    ) -> TokenStream {
85        let span = test.sig.ident.span();
86        let test_cases = self
87            .argument_data(resolver, info)
88            .map(|(name, r)| TestCaseRender::new(Ident::new(&name, span), attrs, r))
89            .map(|test_case| test_case.render(test, info));
90
91        quote! { #(#test_cases)* }
92    }
93
94    fn argument_data<'a>(
95        &'a self,
96        resolver: &'a dyn Resolver,
97        info: &'a RsTestInfo,
98    ) -> impl Iterator<Item = (String, Box<(&'a dyn Resolver, (Pat, Expr))>)> + 'a {
99        let max_len = self.values.len();
100        self.values.iter().enumerate().map(move |(index, value)| {
101            let description = sanitize_ident(&value.description());
102            let arg = info.arguments.inner_pat(&self.arg);
103
104            let arg_name = arg
105                .maybe_ident()
106                .expect("BUG: Here all arguments should be PatIdent types")
107                .to_string();
108
109            let name = format!(
110                "{}_{:0len$}_{description:.64}",
111                arg_name,
112                index + 1,
113                len = max_len.display_len()
114            );
115            let resolver_this = (arg.clone(), value.expr.clone());
116            (name, Box::new((resolver, resolver_this)))
117        })
118    }
119}

121fn _matrix_recursive<'a>(
122    test: &ItemFn,
123    list_values: &'a [&'a ValueList],
124    resolver: &dyn Resolver,
125    attrs: &'a [syn::Attribute],
126    info: &RsTestInfo,
127) -> TokenStream {
128    if list_values.is_empty() {
129        return Default::default();
130    }
131    let vlist = list_values[0];
132    let list_values = &list_values[1..];
133
134    if list_values.is_empty() {
135        let mut attrs = attrs.to_vec();
136        attrs.push(parse_quote!(
137            #[allow(non_snake_case)]
138        ));
139        vlist.render(test, resolver, &attrs, info)
140    } else {
141        let span = test.sig.ident.span();
142        let modules = vlist
143            .argument_data(resolver, info)
144            .map(move |(name, resolver)| {
145                _matrix_recursive(test, list_values, &resolver, attrs, info)
146                    .wrap_by_mod(&Ident::new(&name, span))
147            });
148
149        quote! { #(
150            #[allow(non_snake_case)]
151            #modules
152        )* }
153    }
154}

156pub(crate) fn matrix(mut test: ItemFn, mut info: RsTestInfo) -> TokenStream {
157    test.apply_arguments(&mut info.arguments, &mut ());
158    let span = test.sig.ident.span();
159
160    let cases = cases_data(&info, span).collect::<Vec<_>>();
161
162    let resolver = resolver::fixtures::get(&info.arguments, info.data.fixtures());
163    let rendered_cases = if cases.is_empty() {
164        let list_values = info.data.list_values().collect::<Vec<_>>();
165        _matrix_recursive(&test, &list_values, &resolver, &[], &info)
166    } else {
167        cases
168            .into_iter()
169            .map(|(case_name, attrs, case_resolver)| {
170                let list_values = info.data.list_values().collect::<Vec<_>>();
171                _matrix_recursive(
172                    &test,
173                    &list_values,
174                    &(case_resolver, &resolver),
175                    attrs,
176                    &info,
177                )
178                .wrap_by_mod(&case_name)
179            })
180            .collect()
181    };
182
183    test_group(test, rendered_cases)
184}

194fn render_exec_call(fn_path: Path, args: &[Expr], is_async: bool) -> TokenStream {
195    if is_async {
196        quote! {#fn_path(#(#args),*).await}
197    } else {
198        quote! {#fn_path(#(#args),*)}
199    }
200}

202fn render_test_call(
203    fn_path: Path,
204    args: &[Expr],
205    timeout: Option<Expr>,
206    is_async: bool,
207) -> TokenStream {
208    let timeout = timeout.map(|x| quote! {#x}).or_else(|| {
209        std::env::var("RSTEST_TIMEOUT")
210            .ok()
211            .map(|to| quote! { std::time::Duration::from_secs( (#to).parse().unwrap()) })
212    });
213    let rstest_path = crate_name();
214    match (timeout, is_async) {
215        (Some(to_expr), true) => quote! {
216            use #rstest_path::timeout::*;
217            execute_with_timeout_async(move || #fn_path(#(#args),*), #to_expr).await
218        },
219        (Some(to_expr), false) => quote! {
220            use #rstest_path::timeout::*;
221            execute_with_timeout_sync(move || #fn_path(#(#args),*), #to_expr)
222        },
223        _ => render_exec_call(fn_path, args, is_async),
224    }
225}

231/// Render a single test case:
232///
233/// * `name` - Test case name
234/// * `testfn_name` - The name of test function to call
235/// * `args` - The arguments of the test function
236/// * `attrs` - The expected test attributes
237/// * `output` - The expected test return type
238/// * `asyncness` - The `async` fn token
239/// * `test_impl` - If you want embed test function (should be the one called by `testfn_name`)
240/// * `resolver` - The resolver used to resolve injected values
241/// * `info` - `RsTestInfo` that's expose the requested test behavior
242/// * `generic_types` - The generic types used in signature
243///
244// Ok I need some refactoring here but now that not a real issue
245#[allow(clippy::too_many_arguments)]
246fn single_test_case(
247    name: &Ident,
248    testfn_name: &Ident,
249    args: &[FnArg],
250    attrs: &[Attribute],
251    output: &ReturnType,
252    asyncness: Option<Async>,
253    test_impl: Option<&ItemFn>,
254    resolver: impl Resolver,
255    info: &RsTestInfo,
256    generics: &syn::Generics,
257) -> TokenStream {
258    let (attrs, trace_me): (Vec<_>, Vec<_>) =
259        attrs.iter().cloned().partition(|a| !attr_is(a, "trace"));
260    let mut attributes = info.attributes.clone();
261    if !trace_me.is_empty() {
262        attributes.add_trace(format_ident!("trace"));
263    }
264
265    let generics_types = generics_types_ident(generics).cloned().collect::<Vec<_>>();
266    let args = info
267        .arguments
268        .replace_fn_args_with_related_inner_pat(args.iter().cloned())
269        .collect::<Vec<_>>();
270
271    let (injectable_args, ignored_args): (Vec<_>, Vec<_>) =
272        args.iter().partition(|arg| match arg.maybe_pat() {
273            Some(pat) => !info.arguments.is_ignore(pat),
274            None => true,
275        });
276
277    let inject = inject::resolve_arguments(injectable_args.into_iter(), &resolver, &generics_types);
278
279    let args = args
280        .iter()
281        .filter_map(MaybePat::maybe_pat)
282        .cloned()
283        .collect::<Vec<_>>();
284    let trace_args = trace_arguments(args.iter(), &attributes);
285
286    let is_async = asyncness.is_some();
287    let (attrs, timeouts): (Vec<_>, Vec<_>) =
288        attrs.iter().cloned().partition(|a| !attr_is(a, "timeout"));
289
290    let timeout = timeouts
291        .into_iter()
292        .last()
293        .map(|attribute| attribute.parse_args::<Expr>().unwrap());
294
295    // If no injected attribute provided use the default one
296    let test_attr = if attrs
297        .iter()
298        .any(|a| attr_ends_with(a, &parse_quote! {test}))
299    {
300        None
301    } else {
302        Some(resolve_default_test_attr(is_async))
303    };
304
305    let args = args
306        .iter()
307        .map(|arg| (arg, info.arguments.is_by_refs(arg)))
308        .filter_map(|(a, by_refs)| a.maybe_ident().map(|id| (id, by_refs)))
309        .map(|(arg, by_ref)| {
310            if by_ref {
311                parse_quote! { &#arg }
312            } else {
313                parse_quote! { #arg }
314            }
315        })
316        .collect::<Vec<_>>();
317
318    let execute = render_test_call(testfn_name.clone().into(), &args, timeout, is_async);
319    let lifetimes = generics.lifetimes();
320
321    quote! {
322        #test_attr
323        #(#attrs)*
324        #asyncness fn #name<#(#lifetimes,)*>(#(#ignored_args,)*) #output {
325            #test_impl
326            #inject
327            #trace_args
328            #execute
329        }
330    }
331}

333fn trace_arguments<'a>(
334    args: impl Iterator<Item = &'a Pat>,
335    attributes: &RsTestAttributes,
336) -> Option<TokenStream> {
337    let mut statements = args
338        .filter(|&arg| attributes.trace_me(arg))
339        .map(|arg| {
340            let s: Stmt = parse_quote! {
341                println!("{} = {:?}", stringify!(#arg), #arg);
342            };
343            s
344        })
345        .peekable();
346    if statements.peek().is_some() {
347        Some(quote! {
348            println!("{:-^40}", " TEST ARGUMENTS ");
349            #(#statements)*
350            println!("{:-^40}", " TEST START ");
351        })
352    } else {
353        None
354    }
355}

357struct TestCaseRender<'a> {
358    name: Ident,
359    attrs: &'a [syn::Attribute],
360    resolver: Box<dyn Resolver + 'a>,
361}
362
363impl<'a> TestCaseRender<'a> {
364    pub fn new<R: Resolver + 'a>(name: Ident, attrs: &'a [syn::Attribute], resolver: R) -> Self {
365        TestCaseRender {
366            name,
367            attrs,
368            resolver: Box::new(resolver),
369        }
370    }
371
372    fn render(self, testfn: &ItemFn, info: &RsTestInfo) -> TokenStream {
373        let args = testfn.sig.inputs.iter().cloned().collect::<Vec<_>>();
374        let mut attrs = testfn.attrs.clone();
375        attrs.extend(self.attrs.iter().cloned());
376        let asyncness = testfn.sig.asyncness;
377
378        single_test_case(
379            &self.name,
380            &testfn.sig.ident,
381            &args,
382            &attrs,
383            &testfn.sig.output,
384            asyncness,
385            None,
386            self.resolver,
387            info,
388            &testfn.sig.generics,
389        )
390    }
391}

393fn test_group(mut test: ItemFn, rendered_cases: TokenStream) -> TokenStream {
394    let fname = &test.sig.ident;
395    test.attrs = vec![];
396
397    quote! {
398        #[cfg(test)]
399        #test
400
401        #[cfg(test)]
402        mod #fname {
403            use super::*;
404
405            #rendered_cases
406        }
407    }
408}

420fn format_case_name(case: &TestCase, index: usize, display_len: usize) -> String {
421    let description = case
422        .description
423        .as_ref()
424        .map(|d| format!("_{d}"))
425        .unwrap_or_default();
426    format!("case_{index:0display_len$}{description}")
427}

429fn cases_data(
430    info: &RsTestInfo,
431    name_span: Span,
432) -> impl Iterator<Item = (Ident, &[syn::Attribute], HashMap<Pat, &syn::Expr>)> {
433    let display_len = info.data.cases().count().display_len();
434    info.data.cases().enumerate().map({
435        move |(n, case)| {
436            let resolver_case = info
437                .data
438                .case_args()
439                .cloned()
440                .map(|arg| info.arguments.inner_pat(&arg).clone())
441                .zip(case.args.iter())
442                .collect::<HashMap<_, _>>();
443            (
444                Ident::new(&format_case_name(case, n + 1, display_len), name_span),
445                case.attrs.as_slice(),
446                resolver_case,
447            )
448        }
449    })
