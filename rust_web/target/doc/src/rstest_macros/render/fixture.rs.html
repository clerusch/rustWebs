13fn wrap_return_type_as_static_ref(rt: ReturnType) -> ReturnType {
14    match rt {
15        syn::ReturnType::Type(_, t) => parse_quote! {
16           -> &'static #t
17        },
18        o => o,
19    }
20}

22fn wrap_call_impl_with_call_once_impl(call_impl: TokenStream, rt: &ReturnType) -> TokenStream {
23    match rt {
24        syn::ReturnType::Type(_, t) => parse_quote! {
25            static CELL: std::sync::OnceLock<#t> =
26                std::sync::OnceLock::new();
27            CELL.get_or_init(|| #call_impl )
28        },
29        _ => parse_quote! {
30            static CELL: std::sync::Once = std::sync::Once::new();
31            CELL.call_once(|| #call_impl );
32        },
33    }
34}

36pub(crate) fn render(mut fixture: ItemFn, info: FixtureInfo) -> TokenStream {
37    let mut arguments = info.arguments.clone();
38    fixture.apply_arguments(&mut arguments, &mut ());
39    let name = &fixture.sig.ident;
40    let asyncness = &fixture.sig.asyncness.clone();
41    let inner_args = info
42        .arguments
43        .replace_fn_args_with_related_inner_pat(fixture.sig.inputs.iter().cloned())
44        .collect::<Vec<_>>();
45    let args_ident = inner_args
46        .iter()
47        .filter_map(MaybeIdent::maybe_ident)
48        .cloned()
49        .collect::<Vec<_>>();
50    let orig_attrs = &fixture.attrs;
51    let generics = &fixture.sig.generics;
52    let mut default_output = info
53        .attributes
54        .extract_default_type()
55        .unwrap_or_else(|| fixture.sig.output.clone());
56    let default_generics =
57        generics_clean_up(&fixture.sig.generics, std::iter::empty(), &default_output);
58    let default_where_clause = &default_generics.where_clause;
59    let where_clause = &fixture.sig.generics.where_clause;
60    let mut output = fixture.sig.output.clone();
61    let visibility = &fixture.vis;
62    let resolver = (
63        resolver::fixtures::get(&info.arguments, info.data.fixtures()),
64        resolver::values::get(info.data.values()),
65    );
66    let generics_idents = generics
67        .type_params()
68        .map(|tp| &tp.ident)
69        .cloned()
70        .collect::<Vec<_>>();
71    let inject = inject::resolve_arguments(inner_args.iter(), &resolver, &generics_idents);
72
73    let partials = (1..=inner_args.len()).map(|n| {
74        render_partial_impl(
75            &inner_args,
76            &fixture.sig.output,
77            &fixture.sig.generics,
78            fixture.sig.asyncness.as_ref(),
79            n,
80            &resolver,
81            &info,
82        )
83    });
84
85    let args = args_ident
86        .iter()
87        .map(|arg| parse_quote! { #arg })
88        .collect::<Vec<_>>();
89    let call_get = render_exec_call(parse_quote! { Self::get }, &args, asyncness.is_some());
90    let mut call_impl = render_exec_call(parse_quote! { #name }, &args, asyncness.is_some());
91
92    if info.arguments.is_once() {
93        call_impl = wrap_call_impl_with_call_once_impl(call_impl, &output);
94        output = wrap_return_type_as_static_ref(output);
95        default_output = wrap_return_type_as_static_ref(default_output);
96    }
97
98    quote! {
99        #[allow(non_camel_case_types)]
100        #visibility struct #name {}
101
102        impl #name {
103            #(#orig_attrs)*
104            #[allow(unused_mut)]
105            pub #asyncness fn get #generics (#(#inner_args),*) #output #where_clause {
106                #call_impl
107            }
108
109            pub #asyncness fn default #default_generics () #default_output #default_where_clause {
110                #inject
111                #call_get
112            }
113
114            #(#partials)*
115        }
116
117        #[allow(dead_code)]
118        #fixture
119    }
120}

122fn render_partial_impl(
123    args: &[FnArg],
124    output: &ReturnType,
125    generics: &Generics,
126    asyncness: Option<&Async>,
127    n: usize,
128    resolver: &impl Resolver,
129    info: &FixtureInfo,
130) -> TokenStream {
131    let mut output = info
132        .attributes
133        .extract_partial_type(n)
134        .unwrap_or_else(|| output.clone());
135
136    if info.arguments.is_once() {
137        output = wrap_return_type_as_static_ref(output);
138    }
139
140    let generics = generics_clean_up(generics, args.iter().take(n), &output);
141    let where_clause = &generics.where_clause;
142    let genercs_idents = generics
143        .type_params()
144        .map(|tp| &tp.ident)
145        .cloned()
146        .collect::<Vec<_>>();
147    let inject = inject::resolve_arguments(args.iter().skip(n), resolver, &genercs_idents);
148
149    let sign_args = args.iter().take(n);
150    let fixture_args = args
151        .iter()
152        .filter_map(MaybeIdent::maybe_ident)
153        .map(|arg| parse_quote! {#arg})
154        .collect::<Vec<_>>();
155    let name = Ident::new(&format!("partial_{n}"), Span::call_site());
156
157    let call_get = render_exec_call(
158        parse_quote! { Self::get },
159        &fixture_args,
160        asyncness.is_some(),
161    );
162
163    quote! {
164        #[allow(unused_mut)]
165        pub #asyncness fn #name #generics (#(#sign_args),*) #output #where_clause {
166            #inject
167            #call_get
168        }
169    }
170}

206    fn parse_fixture<S: AsRef<str>>(code: S) -> (ItemFn, FixtureOutput) {
207        let item_fn = parse_str::<ItemFn>(code.as_ref()).unwrap();

210        (item_fn, parse2(tokens).unwrap())

220    fn select_method<S: AsRef<str>>(impl_code: ItemImpl, name: S) -> Option<syn::ImplItemFn> {
221        impl_code
222            .items
223            .into_iter()
224            .filter_map(|ii| match ii {
225                syn::ImplItem::Fn(f) => Some(f),
226                _ => None,
227            })
228            .find(|f| f.sig.ident == name.as_ref())
229    }

458    fn clean_generics(#[case] code: &str, #[case] expected: Vec<&str>) {
459        let (item_fn, out) = parse_fixture(code);
460        let n_args = item_fn.sig.inputs.iter().count();
461
462        let mut signatures = vec![select_method(out.core_impl.clone(), "default").unwrap().sig];
463        signatures.extend((1..=n_args).map(|n| {
464            select_method(out.core_impl.clone(), format!("partial_{}", n))
465                .unwrap()
466                .sig
467        }));
468
469        let expected = expected
470            .into_iter()
471            .map(parse_str::<ItemFn>)
472            .map(|f| f.unwrap().sig)
473            .collect::<Vec<_>>();
474
475        assert_eq!(expected, signatures);
476    }
