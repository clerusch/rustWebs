27impl FilesGlobReferences {
28    /// Replace environment variables in a string.
29    fn replace_env_vars(
30        &self,
31        attr: &LitStrAttr,
32        env: impl EnvironmentResolver,
33    ) -> Result<String, syn::Error> {
34        let re = Regex::new(r"\$([a-zA-Z_][a-zA-Z0-9_]+)|\$\{([^}]+)}")
35            .expect("Could not build the regex");
36
37        let ignore_missing = self.ignore_missing_env_vars;
38
39        let path = attr.value();
40        let haystack = path.as_str();
41        let mut result = String::with_capacity(attr.value().len());
42        let mut last_match = 0;
43
44        for caps in re.captures_iter(haystack) {
45            let match_all = caps.get(0).expect("The regex should have matched");
46
47            // Match the name of the variable and its default value (if any).
48            let (var_name, default) = if let Some(m) = caps.get(1) {
49                // In the first case ($VAR), the default value is None.
50                (m.as_str(), None)
51            } else {
52                // In the second case we have to split the variable name and the default value
53                // if there's a `:-` separator.
54                let m = caps.get(2).expect("The regex should have matched");
55
56                let (var_name, default) =
57                    if let Some((var_name, default)) = m.as_str().split_once(":-") {
58                        (var_name, Some(default))
59                    } else {
60                        (m.as_str(), None)
61                    };
62
63                var_name
64                    .chars()
65                    .all(|c| c.is_alphanumeric() || c == '_')
66                    .then_some((var_name, default))
67                    .ok_or_else(|| {
68                        attr.error(&format!(
69                            "The variable \"{}\" name does not match [a-zA-Z0-9_]+",
70                            m.as_str()
71                        ))
72                    })?
73            };
74
75            let v = env.get(var_name);
76            let replacement = match (v.as_deref(), default) {
77                (Some(""), Some(default)) => default.to_string(),
78                (Some(val), _) => val.to_string(),
79                (None, Some(default)) => default.to_string(),
80                (None, _) if ignore_missing => String::new(),
81                (None, _) => {
82                    return Err(attr.error(&format!(
83                        "Could not find the environment variable {:?}",
84                        var_name
85                    )))
86                }
87            };
88
89            result.push_str(&haystack[last_match..match_all.start()]);
90            result.push_str(&replacement);
91            last_match = match_all.end();
92        }
93
94        result.push_str(&haystack[last_match..]);
95
96        Ok(result)
97    }
98
99    /// Return the base_dir attribute if set, with variables replaced.
100    fn base_dir(&self) -> Result<Option<PathBuf>, syn::Error> {
101        self.base_dir
102            .as_ref()
103            .map(|attr| {
104                self.replace_env_vars(attr, StdEnvironmentResolver)
105                    .map(PathBuf::from)
106            })
107            .transpose()
108    }
109
110    /// Return the tuples attribute, path string if they are valid relative paths
111    fn paths(&self, base_dir: &PathBuf) -> Result<Vec<(&LitStrAttr, String)>, syn::Error> {
112        self.glob
113            .iter()
114            .map(|attr| {
115                let path = self.replace_env_vars(attr, StdEnvironmentResolver)?;
116                RelativePath::from_path(&path)
117                    .map_err(|e| attr.error(&format!("Invalid glob path: {e}")))
118                    .map(|p| p.to_logical_path(base_dir))
119                    .map(|p| (attr, p.to_string_lossy().into_owned()))
120            })
121            .collect::<Result<Vec<_>, _>>()
122    }

178impl LitStrAttr {
179    fn parse_meta_name_value(attr: &Attribute) -> Result<Self, syn::Error> {
180        let meta = attr.meta.require_name_value()?;
181        if let syn::Expr::Lit(expr) = &meta.value {
182            if let syn::Lit::Str(value) = expr.lit.clone() {
183                return Ok(Self {
184                    attr: attr.clone(),
185                    value,
186                });
187            }
188        }
189
190        Err(syn::Error::new_spanned(
191            attr,
192            "expected a string literal value",
193        ))
194    }
195
196    fn value(&self) -> String {
197        self.value.value()
198    }
199}

261impl ValueFilesExtractor {
262    pub(crate) fn take(self) -> Result<Vec<(Ident, FilesGlobReferences)>, ErrorsVec> {
263        if self.errors.is_empty() {
264            Ok(self.files)
265        } else {
266            Err(self.errors.into())
267        }
268    }
269
270    fn collect_errors<T: Default>(&mut self, result: Result<T, syn::Error>) -> T {
271        match result {
272            Ok(v) => v,
273            Err(e) => {
274                self.errors.push(e);
275                T::default()
276            }
277        }
278    }

280    fn extract_argument_attrs<'a, B: 'a + std::fmt::Debug>(
281        &mut self,
282        node: &mut FnArg,
283        is_valid_attr: fn(&syn::Attribute) -> bool,
284        build: impl Fn(syn::Attribute) -> syn::Result<B> + 'a,
285    ) -> Vec<B> {
286        self.collect_errors(
287            extract_argument_attrs(node, is_valid_attr, build).collect::<Result<Vec<_>, _>>(),
288        )
289    }

292        self.extract_argument_attrs(node, |a| attr_is(a, "files"), |attr| attr.try_into())
293    }

296        self.extract_argument_attrs(node, |a| attr_is(a, "exclude"), Exclude::try_from)
297    }

306                    .map_err(|_| attr.error("Use #[include_dot_files] to include dot files"))?;

342impl VisitMut for ValueFilesExtractor {
343    fn visit_fn_arg_mut(&mut self, node: &mut FnArg) {
344        let name = node.maybe_ident().cloned();
345        if matches!(node, FnArg::Receiver(_)) || name.is_none() {
346            return;
347        }
348        let name = name.unwrap();
349        let files = self.extract_files(node);
350        let excludes = self.extract_exclude(node);
351        let include_dot_files = self.extract_include_dot_files(node);
352        if !include_dot_files.is_empty() {
353            include_dot_files.iter().skip(1).for_each(|attr| {
354                self.errors
355                    .push(attr.error("Cannot use #[include_dot_files] more than once"))
356            })
357        }
358        let ignore_missing_env_vars = self.extract_ignore_missing_env_vars(node);
359        if !ignore_missing_env_vars.is_empty() {
360            ignore_missing_env_vars.iter().skip(1).for_each(|attr| {
361                self.errors
362                    .push(attr.error("Cannot use #[ignore_missing_env_vars] more than once"))
363            })
364        }
365        let base_dir = self.extract_base_dir(node);
366        if base_dir.len() > 1 {
367            base_dir.iter().skip(1).for_each(|attr| {
368                self.errors
369                    .push(attr.error(r#"Cannot use #[base_dir = "..."] more than once"#))
370            })
371        }
372        if !files.is_empty() {
373            self.files.push((
374                name,
375                FilesGlobReferences::new(
376                    files,
377                    excludes,
378                    include_dot_files.is_empty(),
379                    !ignore_missing_env_vars.is_empty(),
380                )
381                .with_base_dir_opt(base_dir.into_iter().next()),
382            ))
383        } else {
384            excludes.into_iter().for_each(|e| {
385                self.errors.push(
386                    e.attr
387                        .error("You cannot use #[exclude(...)] without #[files(...)]"),
388                )
389            });
390            include_dot_files.into_iter().for_each(|attr| {
391                self.errors
392                    .push(attr.error("You cannot use #[include_dot_files] without #[files(...)]"))
393            });
394            ignore_missing_env_vars.into_iter().for_each(|attr| {
395                self.errors.push(
396                    attr.error("You cannot use #[ignore_missing_env_vars] without #[files(...)]"),
397                )
398            });
399            base_dir.into_iter().for_each(|attr| {
400                self.errors
401                    .push(attr.error(r#"You cannot use #[base_dir = "..."] without #[files(...)]"#))
402            });
403        }
404    }
405}

433trait GlobResolver {
434    fn glob(&self, pattern: &str) -> Result<Vec<PathBuf>, String> {
435        let globs =
436            glob(pattern).map_err(|e| format!("glob failed for whole path `{pattern}` due {e}"))?;
437        globs
438            .into_iter()
439            .map(|p| p.map_err(|e| format!("glob failed for file due {e}")))
440            .map(|r| {
441                r.and_then(|p| {
442                    p.canonicalize()
443                        .map_err(|e| format!("failed to canonicalize {} due {e}", p.display()))
444                })
445            })
446            .collect()
447    }
448}

470impl<'a> ValueListFromFiles<'a> {
471    pub fn to_value_list(
472        &self,
473        files: Vec<(Ident, FilesGlobReferences)>,
474    ) -> Result<Vec<ValueList>, syn::Error> {
475        files
476            .into_iter()
477            .map(|(arg, refs)| {
478                self.file_list_values(refs).map(|values| ValueList {
479                    arg: arg.into_pat(),
480                    values,
481                })
482            })
483            .collect::<Result<Vec<ValueList>, _>>()
484    }
485
486    fn file_list_values(&self, refs: FilesGlobReferences) -> Result<Vec<Value>, syn::Error> {
487        let default_base_dir = self
488            .base_dir
489            .base_dir()
490            .map_err(|msg| refs.glob[0].error(&msg))?;
491
492        let base_dir = match refs.base_dir()? {
493            Some(mut p) => {
494                if p.is_relative() {
495                    p = default_base_dir.join(p);
496                }
497                p.canonicalize().map_err(|e| {
498                    refs.glob[0].error(&format!("Cannot canonicalize base dir due {e}"))
499                })?
500            }
501            None => default_base_dir,
502        };
503
504        let resolved_paths = refs.paths(&base_dir)?;
505        let base_dir = base_dir
506            .into_os_string()
507            .into_string()
508            .map_err(|p| refs.glob[0].error(&format!("Cannot get a valid string from {p:?}")))?;
509
510        let mut values: Vec<(Expr, String)> = vec![];
511        for (attr, abs_path) in self.all_files_path(resolved_paths)? {
512            let relative_path = abs_path
513                .clone()
514                .into_os_string()
515                .into_string()
516                .map(|inner| RelativePath::new(base_dir.as_str()).relative(inner))
517                .map_err(|e| {
518                    attr.error(&format!("Invalid absolute path {}", e.to_string_lossy()))
519                })?;
520
521            if !refs.is_valid(&relative_path) {
522                continue;
523            }
524
525            let path_str = abs_path.to_string_lossy();
526            values.push((
527                parse_quote! {
528                    <::std::path::PathBuf as std::str::FromStr>::from_str(#path_str).unwrap()
529                },
530                render_file_description(&relative_path),
531            ));
532        }
533
534        if values.is_empty() {
535            Err(refs.glob[0].error("No file found"))?;
536        }
537
538        Ok(values
539            .into_iter()
540            .map(|(e, desc)| Value::new(e, Some(desc)))
541            .collect())
542    }
543
544    /// Return the tuples of attribute, file path resolved via glob resolver, sorted by path and without duplications.
545    fn all_files_path<'b>(
546        &self,
547        resolved_paths: Vec<(&'b LitStrAttr, String)>,
548    ) -> Result<Vec<(&'b LitStrAttr, PathBuf)>, syn::Error> {
549        let mut paths = resolved_paths
550            .iter()
551            .map(|(attr, pattern)| {
552                self.g_resolver
553                    .glob(pattern.as_ref())
554                    .map_err(|msg| attr.error(&msg))
555                    .map(|p| (attr, p))
556            })
557            .collect::<Result<Vec<_>, _>>()?
558            .into_iter()
559            .flat_map(|(&attr, inner)| inner.into_iter().map(move |p| (attr, p)))
560            .collect::<Vec<_>>();
561        paths.sort_by(|(_, a), (_, b)| a.cmp(b));
562        paths.dedup_by(|(_, a), (_, b)| a.eq(&b));
563        Ok(paths)
564    }
565}

567fn render_file_description(file: &RelativePath) -> String {
568    let mut description = String::new();
569    for c in file.components() {
570        match c {
571            relative_path::Component::CurDir => continue,
572            relative_path::Component::ParentDir => description.push_str("_UP"),
573            relative_path::Component::Normal(segment) => description.push_str(segment),
574        }
575        description.push('/')
576    }
577    description.pop();
578    description
579}
