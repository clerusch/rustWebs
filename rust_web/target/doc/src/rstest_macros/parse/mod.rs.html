61#[derive(Debug, PartialEq, Clone)]
62pub(crate) enum Attribute {
63    Attr(Ident),
64    Tagged(Ident, Vec<Pat>),
65    Type(Ident, Box<syn::Type>),
66}
67
68impl Parse for Attribute {
69    fn parse(input: ParseStream) -> syn::Result<Self> {
70        if input.peek2(Token![<]) {
71            let tag = input.parse()?;
72            let _open = input.parse::<Token![<]>()?;
73            let inner = input.parse()?;
74            let _close = input.parse::<Token![>]>()?;
75            Ok(Attribute::Type(tag, inner))
76        } else if input.peek2(Token![::]) {
77            let inner = input.parse()?;
78            Ok(Attribute::Attr(inner))
79        } else if input.peek2(token::Paren) {
80            let tag = input.parse()?;
81            let content;
82            let _ = syn::parenthesized!(content in input);
83            let args = Punctuated::<Ident, Token![,]>::parse_terminated(&content)?
84                .into_iter()
85                .map(IntoPat::into_pat)
86                .collect();
87
88            Ok(Attribute::Tagged(tag, args))
89        } else {
90            Ok(Attribute::Attr(input.parse()?))
91        }
92    }
93}

95fn parse_vector_trailing_till_double_comma<T, P>(input: ParseStream) -> syn::Result<Vec<T>>
96where
97    T: Parse,
98    P: syn::token::Token + Parse,
99{
100    Ok(
101        Punctuated::<Option<T>, P>::parse_separated_nonempty_with(input, |input_tokens| {
102            if input_tokens.is_empty() || input_tokens.peek(Token![::]) {
103                Ok(None)
104            } else {
105                T::parse(input_tokens).map(Some)
106            }
107        })?
108        .into_iter()
109        .flatten()
110        .collect(),
111    )
112}

152        }

156impl Parse for Fixture {
157    fn parse(input: ParseStream) -> syn::Result<Self> {
158        let resolve: syn::Path = input.parse()?;
159        if input.peek(Paren) || input.peek(Token![as]) {
160            let positional = if input.peek(Paren) {
161                let content;
162                let _ = syn::parenthesized!(content in input);
163                content.parse()?
164            } else {
165                Default::default()
166            };
167
168            if input.peek(Token![as]) {
169                let _: Token![as] = input.parse()?;
170                let ident: Ident = input.parse()?;
171                Ok(Self::new(ident.into_pat(), resolve, positional))
172            } else {
173                let name = resolve.get_ident().ok_or_else(|| {
174                    syn::Error::new_spanned(
175                        resolve.to_token_stream(),
176                        "Should be an ident".to_string(),
177                    )
178                })?;
179                Ok(Self::new(
180                    name.clone().into_pat(),
181                    name.clone().into(),
182                    positional,
183                ))
184            }
185        } else {
186            Err(syn::Error::new(
187                input.span(),
188                "fixture need arguments or 'as new_name' format",
189            ))
190        }
191    }
192}

263pub(crate) fn extract_argument_attrs<'a, B: 'a + std::fmt::Debug>(
264    node: &mut FnArg,
265    is_valid_attr: fn(&syn::Attribute) -> bool,
266    build: impl Fn(syn::Attribute) -> syn::Result<B> + 'a,
267) -> Box<dyn Iterator<Item = syn::Result<B>> + 'a> {
268    let name = node.maybe_ident().cloned();
269    if name.is_none() {
270        return Box::new(std::iter::empty());
271    }
272
273    if let FnArg::Typed(ref mut arg) = node {
274        // Extract interesting attributes
275        let attrs = std::mem::take(&mut arg.attrs);
276        let (extracted, remain): (Vec<_>, Vec<_>) = attrs.into_iter().partition(is_valid_attr);
277
278        arg.attrs = remain;
279
280        // Parse attrs
281        Box::new(extracted.into_iter().map(build))
282    } else {
283        Box::new(std::iter::empty())
284    }
285}

288/// type
289struct PartialsTypeFunctionExtractor(Result<Vec<(usize, syn::Type)>, ErrorsVec>);
290
291impl PartialsTypeFunctionExtractor {
292    fn take(self) -> Result<Vec<(usize, syn::Type)>, ErrorsVec> {
293        self.0
294    }
295}

297impl Default for PartialsTypeFunctionExtractor {
298    fn default() -> Self {
299        Self(Ok(Vec::default()))
300    }
301}
302
303impl VisitMut for PartialsTypeFunctionExtractor {
304    fn visit_item_fn_mut(&mut self, node: &mut ItemFn) {
305        let attrs = std::mem::take(&mut node.attrs);
306        let (partials, remain): (Vec<_>, Vec<_>) =
307            attrs
308                .into_iter()
309                .partition(|attr| match attr.path().get_ident() {
310                    Some(name) => name
311                        .to_string()
312                        .starts_with(FixtureModifiers::PARTIAL_RET_ATTR),
313                    None => false,
314                });
315
316        node.attrs = remain;
317        let mut errors = ErrorsVec::default();
318        let mut data: Vec<(usize, syn::Type)> = Vec::default();
319        for attr in partials {
320            match attr.parse_args::<syn::Type>() {
321                Ok(t) => {
322                    match attr.path().get_ident().unwrap().to_string()
323                        [FixtureModifiers::PARTIAL_RET_ATTR.len()..]
324                        .parse()
325                    {
326                        Ok(id) => data.push((id, t)),
327                        Err(_) => errors.push(syn::Error::new_spanned(
328                            attr,
329                            "Invalid partial syntax: should be partial_<n_arguments>",
330                        )),
331                    }
332                }
333                Err(e) => errors.push(e),
334            }
335        }
336        self.0 = if errors.len() > 0 {
337            Err(errors)
338        } else {
339            Ok(data)
340        };
341    }
342}

351/// Simple struct used to visit function attributes and extract cases and
352/// eventualy parsing errors
353#[derive(Default)]
354struct CasesFunctionExtractor(Vec<TestCase>, Vec<syn::Error>);
355
356impl VisitMut for CasesFunctionExtractor {
357    fn visit_item_fn_mut(&mut self, node: &mut ItemFn) {
358        let attrs = std::mem::take(&mut node.attrs);
359        let mut attrs_buffer = Default::default();
360        let case: syn::PathSegment = parse_quote! { case };
361        for attr in attrs.into_iter() {
362            if attr_starts_with(&attr, &case) {
363                match attr.parse_args::<Expressions>() {
364                    Ok(expressions) => {
365                        let description =
366                            attr.path().segments.iter().nth(1).map(|p| p.ident.clone());
367                        self.0.push(TestCase {
368                            args: expressions.into(),
369                            attrs: std::mem::take(&mut attrs_buffer),
370                            description,
371                        });
372                    }
373                    Err(err) => self.1.push(err),
374                };
375            } else {
376                attrs_buffer.push(attr)
377            }
378        }
379        node.attrs = std::mem::take(&mut attrs_buffer);
380    }
381}

414/// Simple struct used to visit function args attributes to extract the
415/// excluded ones and eventualy parsing errors
416struct ExcludedTraceAttributesFunctionExtractor(Result<Vec<Pat>, ErrorsVec>);
417impl From<Result<Vec<Pat>, ErrorsVec>> for ExcludedTraceAttributesFunctionExtractor {
418    fn from(inner: Result<Vec<Pat>, ErrorsVec>) -> Self {
419        Self(inner)
420    }
421}
422
423impl ExcludedTraceAttributesFunctionExtractor {
424    pub(crate) fn take(self) -> Result<Vec<Pat>, ErrorsVec> {
425        self.0
426    }
427
428    fn update_error(&mut self, mut errors: ErrorsVec) {
429        match &mut self.0 {
430            Ok(_) => self.0 = Err(errors),
431            Err(err) => err.append(&mut errors),
432        }
433    }
434
435    fn update_excluded(&mut self, value: Pat) {
436        if let Some(inner) = self.0.iter_mut().next() {
437            inner.push(value);
438        }
439    }
440}
441
442impl Default for ExcludedTraceAttributesFunctionExtractor {
443    fn default() -> Self {
444        Self(Ok(Default::default()))
445    }
446}
447
448impl VisitMut for ExcludedTraceAttributesFunctionExtractor {
449    fn visit_fn_arg_mut(&mut self, node: &mut FnArg) {
450        let pat = match node.maybe_pat().cloned() {
451            Some(pat) => pat,
452            None => return,
453        };
454        for r in extract_argument_attrs(node, |a| attr_is(a, "notrace"), |_a| Ok(())) {
455            match r {
456                Ok(_) => self.update_excluded(pat.clone()),
457                Err(err) => self.update_error(err.into()),
458            }
459        }
460
461        syn::visit_mut::visit_fn_arg_mut(self, node);
462    }
463}

471/// Simple struct used to visit function args attributes to check timeout syntax
472struct CheckTimeoutAttributesFunction(Result<(), ErrorsVec>);
473impl From<ErrorsVec> for CheckTimeoutAttributesFunction {
474    fn from(errors: ErrorsVec) -> Self {
475        Self(Err(errors))
476    }
477}
478
479impl CheckTimeoutAttributesFunction {
480    pub(crate) fn take(self) -> Result<(), ErrorsVec> {
481        self.0
482    }
483
484    fn check_if_can_implement_timeous(
485        &self,
486        timeouts: &[&syn::Attribute],
487        asyncness: Option<&Async>,
488    ) -> Option<syn::Error> {
489        if cfg!(feature = "async-timeout") || timeouts.is_empty() {
490            None
491        } else {
492            asyncness.map(|a| {
493                syn::Error::new(
494                    a.span,
495                    "Enable async-timeout feature to use timeout in async tests",
496                )
497            })
498        }
499    }
500}
501
502impl Default for CheckTimeoutAttributesFunction {
503    fn default() -> Self {
504        Self(Ok(()))
505    }
506}
507
508impl VisitMut for CheckTimeoutAttributesFunction {
509    fn visit_item_fn_mut(&mut self, node: &mut ItemFn) {
510        let timeouts = node
511            .attrs
512            .iter()
513            .filter(|&a| attr_is(a, "timeout"))
514            .collect::<Vec<_>>();
515        let mut errors = timeouts
516            .iter()
517            .map(|&attr| attr.parse_args::<syn::Expr>())
518            .filter_map(Result::err)
519            .collect::<Vec<_>>();
520
521        if let Some(e) =
522            self.check_if_can_implement_timeous(timeouts.as_slice(), node.sig.asyncness.as_ref())
523        {
524            errors.push(e);
525        }
526        if !errors.is_empty() {
527            *self = Self(Err(errors.into()));
528        }
529    }
530}
