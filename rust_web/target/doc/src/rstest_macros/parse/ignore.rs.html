7pub(crate) fn extract_ignores(item_fn: &mut ItemFn) -> Result<Vec<Pat>, ErrorsVec> {
8    let mut extractor = JustOnceFnArgAttributeExtractor::from("ignore");
9    extractor.visit_item_fn_mut(item_fn);
10    extractor.take()
11}
