40impl Sqrt2 for Complex<f64> {
41    fn sqrt2_pow(p: i32) -> Complex<f64> {
42        let rt2 = Complex::new(f64::sqrt(2.0), 0.0);
43        if p == 1 {
44            rt2
45        } else {
46            rt2.powi(p)
47        }
48    }
49}

51impl FromPhase for Complex<f64> {
52    fn from_phase(p: impl Into<Phase>) -> Complex<f64> {
53        let p = p.into().to_rational();
54        let exp = (*p.numer() as f64) / (*p.denom() as f64);
55        Complex::new(-1.0, 0.0).powf(exp)
56    }
57
58    fn minus_one() -> Complex<f64> {
59        Self::from_phase(Phase::one())
60    }
61}

95/// Trait that implements conversion of graphs to tensors
96///
97/// This implements a generic method [ToTensor::to_tensor] for any number type that
98/// implements [TensorElem], as well as two convenience methods [ToTensor::to_tensorf]
99/// and [ToTensor::to_tensorf] for [FScalar] and floating-point [Complex] numbers,
100/// respectively.
101pub trait ToTensor {
102    fn to_tensor<A: TensorElem>(&self) -> Tensor<A>;
103
104    /// Shorthand for `to_tensor::<FScalar>()`
105    fn to_tensorf(&self) -> TensorF {
106        self.to_tensor()
107    }
108
109    /// Shorthand for `to_tensor::<Complex<f64>>()`
110    fn to_tensor64(&self) -> Tensor64 {
111        self.to_tensor()
112    }
113}

115pub trait QubitOps<A: TensorElem> {
116    fn ident(q: usize) -> Self;
117    fn delta(q: usize) -> Self;
118    fn cphase(p: impl Into<Phase>, q: usize) -> Self;
119    fn hadamard() -> Self;
120    fn delta_at(&mut self, qs: &[usize]);
121    fn cphase_at(&mut self, p: impl Into<Phase>, qs: &[usize]);
122    fn hadamard_at(&mut self, i: usize);
123
124    /// split into two non-overlapping pieces, where index q=0 and q=1
125    fn slice_qubit_mut(&mut self, q: usize) -> (ArrayViewMut<A, IxDyn>, ArrayViewMut<A, IxDyn>);
126
127    /// contract the last n qubit indices with the first n qubits of other
128    ///
129    /// panics if n is greater than the number of qubits of self or other.
130    fn plug_n_qubits(self, n: usize, other: &Tensor<A>) -> Tensor<A>;
131}

133pub trait CompareTensors {
134    fn scalar_eq(t0: &Self, t1: &Self) -> bool;
135    fn compare(x0: &impl ToTensor, x1: &impl ToTensor) -> bool;
136    fn scalar_compare(x0: &impl ToTensor, x1: &impl ToTensor) -> bool;
137}
138
139impl<A: TensorElem> CompareTensors for Tensor<A> {
140    fn scalar_eq(t0: &Tensor<A>, t1: &Tensor<A>) -> bool {
141        // if dimensions are different, tensors are different
142        if t0.dim() != t1.dim() {
143            return false;
144        }
145
146        // find the first non-zero element of each tensor
147        let a0 = t0.iter().find(|s| !s.is_zero());
148        let a1 = t1.iter().find(|s| !s.is_zero());
149        match (a0, a1) {
150            // if both tensors have a non-zero element, there are 2 cases
151            (Some(b0), Some(b1)) => {
152                // if the non-zero element is equal, tensors should be equal on the nose
153                if b0 == b1 {
154                    t0 == t1
155                }
156                // if they are different, we cross-multiply to check scalar equivalence
157                else {
158                    t0 * *b1 == t1 * *b0
159                }
160            }
161            // all-zero tensors of the same dimension are equal
162            (None, None) => true,
163            // otherwise, one is a zero tensor and the other is non-zero
164            _ => false,
165        }
166    }
167
168    fn scalar_compare(x0: &impl ToTensor, x1: &impl ToTensor) -> bool {
169        let t0 = x0.to_tensor::<A>();
170        let t1 = x1.to_tensor::<A>();
171        Tensor::scalar_eq(&t0, &t1)
172    }
173
174    fn compare(x0: &impl ToTensor, x1: &impl ToTensor) -> bool {
175        x0.to_tensor::<A>() == x1.to_tensor::<A>()
176    }
177}

179impl<A: TensorElem> QubitOps<A> for Tensor<A> {
180    fn slice_qubit_mut(&mut self, q: usize) -> (ArrayViewMut<A, IxDyn>, ArrayViewMut<A, IxDyn>) {
181        let slice0: SliceInfo<_, IxDyn, IxDyn> =
182            SliceInfo::try_from(Vec::from_iter((0..self.ndim()).map(|i| {
183                if i == q {
184                    SliceInfoElem::from(0)
185                } else {
186                    SliceInfoElem::from(..)
187                }
188            })))
189            .unwrap();
190
191        let slice1: SliceInfo<_, IxDyn, IxDyn> =
192            SliceInfo::try_from(Vec::from_iter((0..self.ndim()).map(|i| {
193                if i == q {
194                    SliceInfoElem::from(1)
195                } else {
196                    SliceInfoElem::from(..)
197                }
198            })))
199            .unwrap();
200
201        self.multi_slice_mut((slice0.as_ref(), slice1.as_ref()))
202    }
203
204    fn ident(q: usize) -> Tensor<A> {
205        Tensor::from_shape_fn(vec![2; q * 2], |ix| {
206            if (0..q).all(|i| ix[i] == ix[q + i]) {
207                A::one()
208            } else {
209                A::zero()
210            }
211        })
212    }
213
214    fn delta(q: usize) -> Tensor<A> {
215        Tensor::from_shape_fn(vec![2; q], |ix| {
216            if (0..q).all(|i| ix[i] == 0) || (0..q).all(|i| ix[i] == 1) {
217                A::one()
218            } else {
219                A::zero()
220            }
221        })
222    }
223
224    fn cphase(p: impl Into<Phase>, q: usize) -> Tensor<A> {
225        let mut t = Tensor::ident(q);
226        let qs: Vec<_> = (0..q).collect();
227        t.cphase_at(p, &qs);
228        t
229    }
230
231    fn hadamard() -> Tensor<A> {
232        let n = A::one_over_sqrt2();
233        let minus = A::from_phase(1);
234        array![[n, n], [n, minus * n]].into_dyn()
235    }
236
237    fn delta_at(&mut self, qs: &[usize]) {
238        let mut shape: Vec<usize> = vec![1; self.ndim()];
239        for &q in qs {
240            shape[q] = 2;
241        }
242        let del: Tensor<A> = Tensor::delta(qs.len())
243            .into_shape_with_order(shape)
244            .expect("Bad indices for delta_at");
245        *self *= &del;
246    }
247
248    fn cphase_at(&mut self, p: impl Into<Phase>, qs: &[usize]) {
249        let p = p.into();
250        let mut shape: Vec<usize> = vec![1; self.ndim()];
251        for &q in qs {
252            shape[q] = 2;
253        }
254        let cp: Tensor<A> = Tensor::from_shape_fn(vec![2; qs.len()], |ix| {
255            if (0..qs.len()).all(|i| ix[i] == 1) {
256                A::from_phase(p)
257            } else {
258                A::one()
259            }
260        })
261        .into_shape_with_order(shape)
262        .expect("Bad indices for cphase_at");
263        *self *= &cp;
264    }
265
266    fn hadamard_at(&mut self, q: usize) {
267        let n = A::one_over_sqrt2();
268        let minus = A::from_phase(1); // -1 = e^(i pi)
269
270        // split into two non-overlapping pieces, where index q=0 and q=1
271        let (mut ma, mut mb) = self.slice_qubit_mut(q);
272
273        // iterate over the pieces together and apply a hadamard to each of the
274        // pairs of elements
275        par_azip!((a in &mut ma, b in &mut mb) {
276            let a1 = *a;
277            *a = n * (*a + *b);
278            *b = n * (a1 + minus * *b);
279        });
280    }
281
282    fn plug_n_qubits(self, n: usize, other: &Tensor<A>) -> Tensor<A> {
283        let d1 = self.shape().len();
284        let d2 = other.shape().len();
285        let shape1: Vec<usize> = (0..(d1 + d2 - n))
286            .map(|i| if i < d1 { 2 } else { 1 })
287            .collect();
288        let shape2: Vec<usize> = (0..(d1 + d2 - n))
289            .map(|i| if i < d1 - n { 1 } else { 2 })
290            .collect();
291
292        let t1 = self
293            .into_shared()
294            .into_shape_with_order(shape1)
295            .expect("Invalid tensor reshape");
296        let t1p = t1.broadcast(vec![2; d1 + n]).unwrap();
297        let t2 = other
298            .clone()
299            .into_shared()
300            .into_shape_with_order(shape2)
301            .expect("Invalid tensor reshape");
302        let mut t3 = &t1p * &t2;
303        for _ in 0..n {
304            t3 = t3.sum_axis(Axis(d1 - n));
305        }
306
307        t3
308    }
309}

311impl<G: GraphLike + Clone> ToTensor for G {
312    fn to_tensor<A: TensorElem>(&self) -> Tensor<A> {
313        let mut g = self.clone();
314        g.x_to_z();
315        // H-boxes are not implemented yet
316        for v in g.vertices() {
317            let t = g.vertex_type(v);
318            if t != VType::B && t != VType::Z {
319                panic!("Vertex type currently unsupported: {t:?}");
320            }
321        }
322
323        // initialise the trivial tensor
324        let mut a = Tensor::from_shape_vec(vec![], vec![A::one()]).unwrap();
325        let inp = g.inputs().iter().copied();
326        let mid = g.vertices().filter(|&v| g.vertex_type(v) != VType::B);
327        let outp = g.outputs().iter().copied();
328        let mut vs: Vec<V> = inp.chain(mid.chain(outp)).collect();
329
330        if vs.len() < g.num_vertices() {
331            panic!("All boundary vertices must be an input or an output");
332        }
333
334        vs.reverse();
335        // TODO: pick a good sort order for mid
336
337        let mut indexv: VecDeque<V> = VecDeque::new();
338        let mut seenv: FxHashMap<V, usize> = FxHashMap::default();
339
340        // let mut num_had = 0;
341        // let mut i = 1;
342        // let tot = vs.len();
343
344        for v in vs {
345            // println!("contracting {} ({}/{})", v, i, tot);
346            // i += 1;
347            let p = g.phase(v);
348
349            // the stack! call computes the tensor product of a new spider
350            // (1, e^(i pi p)) with the existing tensor 'a'
351            if p.is_zero() {
352                a = stack![Axis(0), a, a];
353            } else {
354                let f = A::from_phase(p);
355                a = stack![Axis(0), a, &a * f];
356            }
357
358            indexv.push_front(v);
359            let mut deg_v = 0;
360
361            for (w, et) in g.incident_edges(v) {
362                if let Some(deg_w) = seenv.get_mut(&w) {
363                    deg_v += 1;
364                    *deg_w += 1;
365
366                    let wi = indexv
367                        .iter()
368                        .position(|x| *x == w)
369                        .expect("w should be in indexv");
370
371                    if et == EType::N {
372                        a.delta_at(&[0, wi]);
373                    } else {
374                        a.cphase_at(1, &[0, wi]);
375                        // TODO incorporate with cphase_at
376                        a *= A::one_over_sqrt2();
377                        // num_had += 1;
378                    }
379
380                    if g.vertex_type(w) != VType::B && g.degree(w) == *deg_w {
381                        a = a.sum_axis(Axis(wi));
382                        indexv.remove(wi);
383                    }
384                }
385            }
386
387            if g.vertex_type(v) != VType::B && g.degree(v) == deg_v {
388                a = a.sum_axis(Axis(0));
389                indexv.remove(0);
390            }
391
392            seenv.insert(v, deg_v);
393        }
394
395        let s = A::from(*g.scalar()); // * A::sqrt2_pow(-num_had);
396        a * s
397    }
398}

400impl ToTensor for Circuit {
401    fn to_tensor<A: TensorElem>(&self) -> Tensor<A> {
402        use crate::gate::GType::*;
403        let q = self.num_qubits();
404
405        // start with the identity matrix
406        let mut a = Tensor::ident(q);
407
408        // since we are applying the gates to the input indices, this actually
409        // computes the transpose of the circuit, but all the gates are self-
410        // transposed, so we can get the circuit itself if we just reverse the order.
411        for g in self.gates.iter().rev() {
412            match g.t {
413                ZPhase => a.cphase_at(g.phase, &g.qs),
414                Z | CZ | CCZ => a.cphase_at(1, &g.qs),
415                S => a.cphase_at(Rational64::new(1, 2), &g.qs),
416                T => a.cphase_at(Rational64::new(1, 4), &g.qs),
417                Sdg => a.cphase_at(Rational64::new(-1, 2), &g.qs),
418                Tdg => a.cphase_at(Rational64::new(-1, 4), &g.qs),
419                HAD => a.hadamard_at(g.qs[0]),
420                NOT => {
421                    a.hadamard_at(g.qs[0]);
422                    a.cphase_at(Rational64::one(), &g.qs);
423                    a.hadamard_at(g.qs[0]);
424                }
425                XPhase => {
426                    a.hadamard_at(g.qs[0]);
427                    a.cphase_at(g.phase, &g.qs);
428                    a.hadamard_at(g.qs[0]);
429                }
430                CNOT => {
431                    a.hadamard_at(g.qs[1]);
432                    a.cphase_at(Rational64::one(), &g.qs);
433                    a.hadamard_at(g.qs[1]);
434                }
435                TOFF => {
436                    a.hadamard_at(g.qs[2]);
437                    a.cphase_at(Rational64::one(), &g.qs);
438                    a.hadamard_at(g.qs[2]);
439                }
440                SWAP => a.swap_axes(g.qs[0], g.qs[1]),
441                // n.b. these are pyzx-specific gates
442                XCX => {
443                    a.hadamard_at(g.qs[0]);
444                    a.hadamard_at(g.qs[1]);
445                    a.cphase_at(g.phase, &g.qs);
446                    a.hadamard_at(g.qs[0]);
447                    a.hadamard_at(g.qs[1]);
448                }
449                // TODO: these "gates" are not implemented yet
450                ParityPhase => {
451                    panic!("Unsupported gate: ParityPhase")
452                }
453                InitAncilla => {
454                    panic!("Unsupported gate: InitAncilla")
455                }
456                PostSelect => {
457                    panic!("Unsupported gate: PostSelect")
458                }
459                UnknownGate => {} // unknown gates are quietly ignored
460            }
461        }
462        a
463    }
464}
