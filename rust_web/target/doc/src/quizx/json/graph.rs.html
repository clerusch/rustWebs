32impl JsonGraph {
33    /// Encode a graph using the json representation.
34    pub fn from_graph(graph: &impl GraphLike) -> Result<Self, JsonError> {
35        let mut wire_vertices = HashMap::new();
36        let mut node_vertices = HashMap::new();
37        let mut undir_edges = HashMap::new();
38
39        // The encoding requires unique string names for vertices and edges.
40        let mut vertex_name_gen = (0..).map(|i| format!("v{i}"));
41        let mut bound_name_gen = (0..).map(|i| format!("b{i}"));
42        let mut edge_name_gen = (0..).map(|i| format!("e{i}"));
43
44        let mut v_names: HashMap<V, VertexName> = HashMap::new();
45
46        for v in graph.vertices() {
47            let typ = graph.vertex_type(v);
48            let coord = graph.coord(v);
49            let v_name = match typ {
50                VType::B => bound_name_gen.next(),
51                _ => vertex_name_gen.next(),
52            }
53            .unwrap();
54            v_names.insert(v, v_name.clone());
55
56            if typ == VType::B {
57                let input = graph.inputs().iter().position(|&i| i == v);
58                let output = graph.outputs().iter().position(|&o| o == v);
59                assert!(
60                    input.is_some() || output.is_some(),
61                    "Boundary vertex is not an input nor output."
62                );
63                let attrs = VertexAttrs {
64                    annotation: VertexAnnotations {
65                        boundary: true,
66                        coord: (coord.x, coord.y),
67                        input,
68                        output,
69                        ..Default::default()
70                    },
71                    ..Default::default()
72                };
73
74                wire_vertices.insert(v_name, attrs);
75            } else {
76                let phase = graph.phase(v);
77                // Encode zero-phases as empty strings by default. If the vertex
78                // is a Hadamard node, encode "1" as empty strings instead.
79                let phase_options = PhaseOptions {
80                    ignore_value: Some(match typ == VType::H {
81                        true => Phase::one(),
82                        false => Phase::zero(),
83                    }),
84                    ..Default::default()
85                };
86                let value = JsonPhase::from_phase(phase, phase_options);
87                let mut attrs = VertexAttrs {
88                    annotation: VertexAnnotations {
89                        coord: (coord.x, coord.y),
90                        ..Default::default()
91                    },
92                    data: VertexData {
93                        typ,
94                        value,
95                        ..Default::default()
96                    },
97                };
98
99                if typ == VType::ZBox {
100                    // Data for ZBox vertices is not currently supported, so write the default.
101                    attrs.annotation.label = Some("1".to_string());
102                }
103
104                node_vertices.insert(v_name, attrs);
105            };
106        }
107
108        for (src, tgt, typ) in graph.edges() {
109            match typ {
110                EType::N | EType::Wio => {
111                    let attr = EdgeAttrs {
112                        src: v_names[&src].clone(),
113                        tgt: v_names[&tgt].clone(),
114                        typ,
115                    };
116                    undir_edges.insert(edge_name_gen.next().unwrap(), attr);
117                }
118                EType::H => {
119                    // Encoded as a Hadamard node and two simple edges.
120                    let h_name = vertex_name_gen.next().unwrap();
121                    let coord = avg_coord(graph.coord(src), graph.coord(tgt));
122                    node_vertices.insert(
123                        h_name.clone(),
124                        VertexAttrs {
125                            annotation: VertexAnnotations {
126                                coord: (coord.x, coord.y),
127                                ..Default::default()
128                            },
129                            data: VertexData {
130                                typ: VType::H,
131                                is_edge: true,
132                                ..Default::default()
133                            },
134                        },
135                    );
136                    undir_edges.insert(
137                        edge_name_gen.next().unwrap(),
138                        EdgeAttrs {
139                            src: v_names[&src].clone(),
140                            tgt: h_name.clone(),
141                            ..Default::default()
142                        },
143                    );
144                    undir_edges.insert(
145                        edge_name_gen.next().unwrap(),
146                        EdgeAttrs {
147                            src: h_name,
148                            tgt: v_names[&tgt].clone(),
149                            ..Default::default()
150                        },
151                    );
152                }
153            };
154        }
155
156        let scalar = graph.scalar();
157        let scalar = (!scalar.is_one()).then(|| JsonScalar::from(scalar));
158
159        Ok(Self {
160            wire_vertices,
161            node_vertices,
162            undir_edges,
163            variable_types: Default::default(),
164            scalar,
165        })
166    }

168    /// Decode a graph from the json representation.
169    pub fn to_graph<G: GraphLike>(&self) -> Result<G, JsonError> {
170        let mut graph = G::new();
171
172        if !self.variable_types.is_empty() {
173            unimplemented!("Variables are not currently supported.");
174        }
175
176        let mut names: HashMap<VertexName, V> = HashMap::new();
177
178        // Map used to track auxiliary Hadamard nodes that should be decoded as Hadamard edges.
179        // Stores the neighbor nodes of the Hadamard node, and the coordinate of the Hadamard node.
180        let mut hadamards: HashMap<&str, (Vec<V>, Coord)> = HashMap::new();
181
182        for (name, attrs) in &self.node_vertices {
183            let coord = Coord {
184                x: attrs.annotation.coord.0,
185                y: attrs.annotation.coord.1,
186            };
187            if attrs.data.typ == VType::H && attrs.data.is_edge {
188                // A virtual hadamard edge.
189                hadamards.insert(name, (vec![], coord));
190                continue;
191            }
192
193            let phase = attrs
194                .data
195                .value
196                .to_phase()
197                .map_err(|_| JsonError::InvalidNodePhase {
198                    name: name.to_string(),
199                    phase: attrs.data.value.0.clone(),
200                })?;
201            let phase = match (phase, attrs.data.typ) {
202                (Some(r), _) => r,
203                // The phase defaults to one for Hadamard nodes,
204                (None, VType::H) => Rational64::one().into(),
205                // and zero for all others.
206                (None, _) => Rational64::zero().into(),
207            };
208            let v = graph.add_vertex_with_data(VData {
209                ty: attrs.data.typ,
210                qubit: coord.qubit(),
211                row: coord.row(),
212                phase,
213            });
214            names.insert(name.to_string(), v);
215        }
216
217        // Insert the boundary nodes, and collect the input and output vectors.
218        let mut inputs: BTreeMap<usize, &str> = BTreeMap::new();
219        let mut outputs: BTreeMap<usize, &str> = BTreeMap::new();
220        for (name, attrs) in &self.wire_vertices {
221            let coord = Coord {
222                x: attrs.annotation.coord.0,
223                y: attrs.annotation.coord.1,
224            };
225            let v = graph.add_vertex_with_data(VData {
226                ty: VType::B,
227                qubit: coord.qubit(),
228                row: coord.row(),
229                phase: Phase::zero(),
230            });
231            names.insert(name.to_string(), v);
232            if let Some(input) = attrs.annotation.input {
233                inputs.insert(input, name);
234            }
235            if let Some(output) = attrs.annotation.output {
236                outputs.insert(output, name);
237            }
238        }
239        graph.set_inputs(inputs.into_values().map(|name| names[name]).collect());
240        graph.set_outputs(outputs.into_values().map(|name| names[name]).collect());
241
242        // Insert the edges.
243        for attrs in self.undir_edges.values() {
244            let src = || names[&attrs.src];
245            let tgt = || names[&attrs.tgt];
246
247            match (
248                hadamards.get(attrs.src.as_str()),
249                hadamards.get(attrs.tgt.as_str()),
250            ) {
251                (Some((_, src_coord)), Some((_, tgt_coord))) => {
252                    // Both ends are virtual Hadamard nodes.
253                    //
254                    // Not sure how this is possible, but pyzx supports it.
255                    let new_coord = avg_coord(*src_coord, *tgt_coord);
256                    let v = graph.add_vertex_with_data(VData {
257                        ty: VType::Z,
258                        qubit: new_coord.qubit(),
259                        row: new_coord.row(),
260                        phase: Phase::zero(),
261                    });
262                    let name = format!("v{}", graph.num_vertices());
263                    names.insert(name, v);
264                    hadamards.get_mut(attrs.src.as_str()).unwrap().0.push(v);
265                    hadamards.get_mut(attrs.tgt.as_str()).unwrap().0.push(v);
266                    continue;
267                }
268                (Some(_), None) => {
269                    hadamards.get_mut(attrs.src.as_str()).unwrap().0.push(tgt());
270                    continue;
271                }
272                (None, Some(_)) => {
273                    hadamards.get_mut(attrs.tgt.as_str()).unwrap().0.push(src());
274                    continue;
275                }
276                _ => {}
277            }
278
279            graph.add_edge_smart(src(), tgt(), attrs.typ);
280        }
281
282        // Add the Hadamard edges.
283        for (neighbors, _) in hadamards.values() {
284            if neighbors.len() != 2 {
285                panic!("Virtual Hadamard node has wrong number of neighbors.");
286            }
287            let (src, tgt) = (neighbors[0], neighbors[1]);
288            graph.add_edge_smart(src, tgt, EType::H);
289        }
290
291        // Set the scalar.
292        if let Some(scalar) = &self.scalar {
293            *graph.scalar_mut() = FScalar::try_from(scalar)?;
294        }
295
296        Ok(graph)
297    }
298}

300/// Returns the average of two coordinates, as a pair of f64.
301///
302/// Rounds the result to 3 decimal places.
303fn avg_coord(a: Coord, b: Coord) -> Coord {
304    Coord {
305        x: ((a.x + b.x) * 1000.).round() / 2000.,
306        y: ((a.y + b.y) * 1000.).round() / 2000.,
307    }
