17//! # Basic ZX-calculus rules
18//!
19//! These rules always come in triples of functions. For a rule X,
20//! there is a function `check_X(&g, ...) -> bool` which checks
21//! whether a rule is applicable at the given vertex or vertices,
22//! `X_unchecked(&mut g, ...)` applies the rule without doing any
23//! checking, and `X(&mut g, ...) -> bool` is the composition of the
24//! first two.
25//!
26//! Note calling `X_unchecked` is allowed to make unsound ZX-diagram
27//! transformations, or even panic, if `check_X` doesn't return true.
28

37/// Define a checked rule that takes 1 vertex
38macro_rules! checked_rule1 {
39    ( $check:ident, $unchecked:ident, $name:ident ) => {
40        /// A checked implementation of the rule
41        ///
42        /// See e.g. [spider_fusion] for an example.
43        pub fn $name(g: &mut impl GraphLike, v: V) -> bool {
44            if $check(g, v) {
45                $unchecked(g, v);
46                true
47            } else {
48                false
49            }
50        }
51    };
52}
53
54/// Define a checked rule that takes 2 vertices
55macro_rules! checked_rule2 {
56    ( $check:ident, $unchecked:ident, $name:ident ) => {
57        /// A checked implementation of the rule
58        ///
59        /// See e.g. [spider_fusion] for an example.
60        pub fn $name(g: &mut impl GraphLike, v0: V, v1: V) -> bool {
61            if $check(g, v0, v1) {
62                $unchecked(g, v0, v1);
63                true
64            } else {
65                false
66            }
67        }
68    };
69}

71/// Check [spider_fusion_unchecked] applies
72///
73/// Both vertices must be Z or X, have the same type, and be connected
74/// by a normal (i.e. non-Hadamard) edge.
75///
76/// ```
77/// # use quizx::graph::*;
78/// # use quizx::vec_graph::Graph;
79/// # use quizx::basic_rules::check_spider_fusion;
80/// let mut g = Graph::new();
81/// let v0 = g.add_vertex(VType::Z);
82/// let v1 = g.add_vertex(VType::Z);
83/// let v2 = g.add_vertex(VType::X);
84/// g.add_edge(v0, v1);
85/// g.add_edge(v1, v2);
86///
87/// assert!(check_spider_fusion(&g, v0, v1));
88/// assert!(!check_spider_fusion(&g, v1, v2));
89/// ```
90pub fn check_spider_fusion(g: &impl GraphLike, v0: V, v1: V) -> bool {
91    v0 != v1
92        && g.edge_type_opt(v0, v1) == Some(EType::N)
93        && ((g.vertex_type(v0) == VType::Z && g.vertex_type(v1) == VType::Z)
94            || (g.vertex_type(v0) == VType::X && g.vertex_type(v1) == VType::X))
95}

97/// Apply spider fusion
98///
99/// Note the first vertex is preserved by the fusion, and the second
100/// is deleted.
101///
102/// ```
103/// # use quizx::graph::*;
104/// # use quizx::tensor::ToTensor;
105/// # use quizx::vec_graph::Graph;
106/// # use quizx::basic_rules::spider_fusion_unchecked;
107/// let mut g = Graph::new();
108/// let v0 = g.add_vertex(VType::Z);
109/// let v1 = g.add_vertex(VType::Z);
110/// let v2 = g.add_vertex(VType::X);
111/// g.add_edge(v0, v1);
112/// g.add_edge(v1, v2);
113///
114/// let h = g.clone();
115/// spider_fusion_unchecked(&mut g, v0, v1);
116/// assert_eq!(g.to_tensorf(), h.to_tensorf());
117///
118/// let h = g.clone();
119/// spider_fusion_unchecked(&mut g, v0, v2); // oops! Not a valid spider fusion
120/// assert_ne!(g.to_tensorf(), h.to_tensorf());
121/// ```
122pub fn spider_fusion_unchecked(g: &mut impl GraphLike, v0: V, v1: V) {
123    for (v, et) in Vec::from_iter(g.incident_edges(v1)) {
124        if v != v0 {
125            g.add_edge_smart(v0, v, et);
126        }
127    }
128
129    g.add_to_phase(v0, g.phase(v1));
130    g.remove_vertex(v1);
131}

133/// A checked implementation of the rule
134///
135/// Note the first vertex is preserved by the fusion, and the second
136/// is deleted.
137///
138/// ```
139/// # use quizx::graph::*;
140/// # use quizx::tensor::ToTensor;
141/// # use quizx::vec_graph::Graph;
142/// # use quizx::basic_rules::spider_fusion;
143/// let mut g = Graph::new();
144/// let v0 = g.add_vertex(VType::Z);
145/// let v1 = g.add_vertex(VType::Z);
146/// let v2 = g.add_vertex(VType::X);
147/// g.add_edge(v0, v1);
148/// g.add_edge(v1, v2);
149///
150/// let h = g.clone();
151/// let success = spider_fusion(&mut g, v0, v1);
152/// assert!(success);
153/// assert_eq!(g.to_tensorf(), h.to_tensorf());
154///
155/// let h = g.clone();
156/// let success = spider_fusion(&mut g, v0, v2); // should fail
157/// assert!(!success);
158/// assert_eq!(g, h); // g is unchanged
159/// ```
160pub fn spider_fusion(g: &mut impl GraphLike, v0: V, v1: V) -> bool {
161    if check_spider_fusion(g, v0, v1) {
162        spider_fusion_unchecked(g, v0, v1);
163        true
164    } else {
165        false
166    }
167}

169/// Check [pi_copy_unchecked] applies
170pub fn check_pi_copy(g: &impl GraphLike, v: V) -> bool {
171    let vt = g.vertex_type(v);
172    // Find the opposite color of this spider, or fail
173    // if it is not a colored spider.
174    let ovt = match vt {
175        VType::Z => VType::X,
176        VType::X => VType::Z,
177        _ => return false,
178    };
179
180    // No pi-copy on empty spiders.
181    if g.degree(v) == 0 {
182        return false;
183    }
184
185    for neighbor in g.neighbors(v) {
186        // Every neighbor must be the same color and connected by
187        // a hadamard edge or the opposite color and connected by a normal edge.
188        match g.edge_type(v, neighbor) {
189            EType::N if g.vertex_type(neighbor) != ovt => return false,
190            EType::H if g.vertex_type(neighbor) != vt => return false,
191            _ => (),
192        }
193    }
194
195    true
196}

198/// Apply a pi-copy
199///
200/// This will flip the phase of a spider by adding a pi phase
201/// to all its neighbours, assuming that they are either the
202/// same color and connected by a Hadamard edge, or the opposite
203/// color and connected by a normal edge. In particular, this means
204/// that this rule can always be applied when the graph is in gh form.
205pub fn pi_copy_unchecked(g: &mut impl GraphLike, v: V) {
206    // Flip the phase of this node
207    let phase = g.phase(v);
208    g.scalar_mut().mul_phase(phase);
209    g.set_phase(v, -phase);
210
211    // Push a pi to all the surrounding nodes
212    for neighbor in g.neighbor_vec(v) {
213        g.add_to_phase(neighbor, 1);
214    }
215}
216
217checked_rule1!(check_pi_copy, pi_copy_unchecked, pi_copy);

220pub fn check_remove_id(g: &impl GraphLike, v: V) -> bool {
221    let vt = g.vertex_type(v);
222
223    (vt == VType::Z || vt == VType::X) && g.phase(v).is_zero() && g.degree(v) == 2
224}

226/// Remove an arity-2 spider with phase 0
227///
228/// Removes the spider and connects its two neighbors. The type
229/// of the resulting edge is the parity of the types of
230/// original 2 edges, namely: {N,N} -> N, {N,H} -> H, and
231/// {H, H} -> N.
232pub fn remove_id_unchecked(g: &mut impl GraphLike, v: V) {
233    let nhd: Vec<(V, EType)> = g.incident_edges(v).collect();
234    let new_et = match (nhd[0].1, nhd[1].1) {
235        (EType::N, EType::N) => EType::N,
236        (EType::N, EType::H) => EType::H,
237        (EType::H, EType::N) => EType::H,
238        (EType::H, EType::H) => EType::N,
239        (EType::Wio, _) | (_, EType::Wio) => unimplemented!("W nodes not supported"),
240    };
241    g.add_edge_smart(nhd[0].0, nhd[1].0, new_et);
242    g.remove_vertex(v);
243}
244
245checked_rule1!(check_remove_id, remove_id_unchecked, remove_id);

253/// Change the color of a Z or X spider
254///
255/// All of the neighboring edge types are toggled, i.e. N -> H,
256/// H -> N.
257pub fn color_change_unchecked(g: &mut impl GraphLike, v: V) {
258    let vt = g.vertex_type(v);
259    g.set_vertex_type(v, if vt == VType::X { VType::Z } else { VType::X });
260    for w in Vec::from_iter(g.neighbors(v)) {
261        g.toggle_edge_type(v, w);
262    }
263}

267/// Check [local_comp_unchecked] applies
268///
269/// The vertex must be Z, have a phase pi/2 or -pi/2, and be
270/// surrounded by H-edges connected to other Z spiders.
271pub fn check_local_comp(g: &impl GraphLike, v: V) -> bool {
272    g.vertex_type(v) == VType::Z
273        && g.phase(v).is_proper_clifford()
274        && g.incident_edges(v)
275            .all(|(v0, et)| g.vertex_type(v0) == VType::Z && et == EType::H)
276}

278/// Apply a local complementation
279///
280/// This is the version that deletes the targeted vertex. In
281/// other words, it is an N-ary generalization of the Euler
282/// decomposition rule.
283pub fn local_comp_unchecked(g: &mut impl GraphLike, v: V) {
284    let p = g.phase(v);
285
286    // add a totally connected graph of the nhd of v
287    let ns: Vec<V> = g.neighbors(v).collect();
288    for i in 0..ns.len() {
289        g.add_to_phase(ns[i], -p);
290        for j in (i + 1)..ns.len() {
291            g.add_edge_smart(ns[i], ns[j], EType::H);
292        }
293    }
294    g.remove_vertex(v);
295
296    let x = ns.len() as i32;
297    g.scalar_mut().mul_sqrt2_pow(((x - 1) * (x - 2)) / 2);
298    g.scalar_mut()
299        .mul_phase(Rational64::new(*p.to_rational().numer(), 4));
300}
301
302checked_rule1!(check_local_comp, local_comp_unchecked, local_comp);

304/// Check [pivot_unchecked] applies
305///
306/// Both vertices must be Z, have a phase 0 or pi, and be
307/// surrounded by H-edges connected to other Z spiders.
308pub fn check_pivot(g: &impl GraphLike, v0: V, v1: V) -> bool {
309    g.vertex_type(v0) == VType::Z
310        && g.vertex_type(v1) == VType::Z
311        && g.edge_type_opt(v0, v1) == Some(EType::H)
312        && g.phase(v0).is_pauli()
313        && g.phase(v1).is_pauli()
314        && g.incident_edges(v0)
315            .all(|(w, et)| g.vertex_type(w) == VType::Z && et == EType::H)
316        && g.incident_edges(v1)
317            .all(|(w, et)| g.vertex_type(w) == VType::Z && et == EType::H)
318}

320/// Apply pivoting to a pair of vertices
321///
322/// This is the version that deletes both vertices, so it is
323/// effectively a generalised version of the strong complementarity
324/// rule.
325pub fn pivot_unchecked(g: &mut impl GraphLike, v0: V, v1: V) {
326    let p0 = g.phase(v0);
327    let p1 = g.phase(v1);
328
329    // add a complete bipartite graph between the neighbors of v0
330    // and the neighbors of v1
331    let ns0: Vec<V> = g.neighbors(v0).collect();
332    let ns1: Vec<V> = g.neighbors(v1).collect();
333    // let mut z: i32 = 0; // the number of neighbors of v0 and v1
334    for &n0 in &ns0 {
335        g.add_to_phase(n0, p1);
336        for &n1 in &ns1 {
337            if n0 != v1 && n1 != v0 {
338                // unlike PyZX, add_edge_smart handles self-loops
339                g.add_edge_smart(n0, n1, EType::H);
340            }
341        }
342    }
343
344    for &n1 in &ns1 {
345        g.add_to_phase(n1, p0);
346    }
347
348    g.remove_vertex(v0);
349    g.remove_vertex(v1);
350
351    let x = ns0.len() as i32; // the number of neighbors of v0
352    let y = ns1.len() as i32; // the number of neighbors of v1
353    g.scalar_mut().mul_sqrt2_pow((x - 2) * (y - 2));
354
355    if !p0.is_zero() && !p1.is_zero() {
356        g.scalar_mut().mul_phase(Rational64::new(1, 1));
357    }
358}
359
360checked_rule2!(check_pivot, pivot_unchecked, pivot);

362/// Insert an identity spider so v is no longer adjancent to boundary b
363///
364/// If b is not a boundary, this is a noop. The new vertex will be connected
365/// to v by a Hadamard edge.
366fn unfuse_boundary(g: &mut impl GraphLike, v: V, b: V) {
367    if g.vertex_type(b) != VType::B {
368        return;
369    }
370    let vd = VData {
371        ty: VType::Z,
372        phase: Phase::zero(),
373        row: g.row(v),
374        qubit: g.qubit(v),
375    };
376    let v1 = g.add_vertex_with_data(vd);
377    g.add_edge_with_type(v, v1, EType::H);
378    g.add_edge_with_type(v1, b, g.edge_type(v, b).opposite());
379    g.remove_edge(v, b);
380}

382/// Unfuse a non-Pauli phase as a degree-1 phase gadget
383///
384/// If the vertex already has a Pauli phase, this is a noop.
385fn unfuse_gadget(g: &mut impl GraphLike, v: V) {
386    if g.phase(v).is_pauli() {
387        return;
388    }
389    let vd1 = VData {
390        ty: VType::Z,
391        phase: Phase::zero(),
392        row: g.row(v),
393        qubit: -1.0,
394    };
395
396    let vd2 = VData {
397        ty: VType::Z,
398        phase: g.phase(v),
399        row: g.row(v),
400        qubit: -2.0,
401    };
402    let v1 = g.add_vertex_with_data(vd1);
403    let v2 = g.add_vertex_with_data(vd2);
404    g.set_phase(v, Phase::zero());
405    g.add_edge_with_type(v, v1, EType::H);
406    g.add_edge_with_type(v1, v2, EType::H);
407}

409/// Check gen_pivot applies
410///
411/// This checks that we can create a valid matching for the pivot rule
412/// by gadgetizing any non-Pauli phases and inserting identities to make
413/// both vertices interior. Note that repeatedly applying `gen_pivot` with
414/// this checker will not always terminate.
415pub fn check_gen_pivot(g: &impl GraphLike, v0: V, v1: V) -> bool {
416    if v0 == v1 {
417        return false;
418    }
419    if g.edge_type_opt(v0, v1) != Some(EType::H) {
420        return false;
421    }
422
423    for &v in &[v0, v1] {
424        if g.vertex_type(v) != VType::Z {
425            return false;
426        }
427        for (w, et) in g.incident_edges(v) {
428            let t = g.vertex_type(w);
429            if !((t == VType::Z && et == EType::H) || t == VType::B) {
430                return false;
431            }
432        }
433    }
434
435    true
436}

438// check that a vertex is interior, has phase 0 or pi, and is not
439// a phase gadget
440fn is_interior_pauli(g: &impl GraphLike, v: V) -> bool {
441    g.phase(v).is_pauli()
442        && g.neighbors(v)
443            .all(|n| g.vertex_type(n) == VType::Z && g.degree(n) > 1)
444}
445
446// check that a vertex is interior, has phase 0 or pi, and is not
447// a phase gadget
448fn is_boundary_pauli(g: &impl GraphLike, v: V) -> bool {
449    g.phase(v).is_pauli() && g.neighbors(v).any(|n| g.vertex_type(n) == VType::B)
450}

452/// Check gen_pivot applies and at least one vertex is interior Pauli
453///
454/// Unlike `check_gen_pivot`, this guarantees applying `gen_pivot` will
455/// strictly decrease the number of interior Pauli vertices, hence it
456/// will terminate.
457pub fn check_gen_pivot_reduce(g: &impl GraphLike, v0: V, v1: V) -> bool {
458    check_gen_pivot(g, v0, v1) && (is_interior_pauli(g, v0) || is_interior_pauli(g, v1))
459}

461pub fn check_boundary_pivot(g: &impl GraphLike, v0: V, v1: V) -> bool {
462    check_gen_pivot(g, v0, v1) && is_boundary_pauli(g, v0)
463}

465/// Generic version of the pivot rule
466///
467/// This version of the pivoting rule allows either of the vertices
468/// to have non-Pauli phases and/or be connected to boundaries. To handle
469/// these situations, some spiders are first unfused, such that interior
470/// non-Pauli spiders produce phase gadgets and boundary non-Pauli spiders
471/// produce phase gates on inputs/outputs.
472pub fn gen_pivot_unchecked(g: &mut impl GraphLike, v0: V, v1: V) {
473    let nhd0 = g.neighbor_vec(v0);
474    unfuse_gadget(g, v0);
475    for &n in &nhd0 {
476        unfuse_boundary(g, v0, n);
477    }
478
479    let nhd1 = g.neighbor_vec(v1);
480    unfuse_gadget(g, v1);
481    for &n in &nhd1 {
482        unfuse_boundary(g, v1, n);
483    }
484
485    pivot_unchecked(g, v0, v1);
486
487    // for &n in nhd0.iter().chain(nhd1.iter()) {
488    //      if g.contains_vertex(n) {
489    //          if remove_id(g, n) { println!("REMOVED EXTRA: {}", n); }
490    //      }
491    //  }
492}

497pub fn check_gadget_fusion(g: &impl GraphLike, v0: V, v1: V) -> bool {
498    if v0 == v1 {
499        return false;
500    }
501
502    let vs = [v0, v1];
503    let mut nhd = [FxHashSet::default(), FxHashSet::default()];
504
505    for i in 0..2 {
506        let mut found_gphase = false;
507        for (n, et) in g.incident_edges(vs[i]) {
508            if et != EType::H {
509                return false;
510            }
511            if g.vertex_type(n) != VType::Z {
512                return false;
513            }
514            if g.degree(n) == 1 {
515                if found_gphase {
516                    return false;
517                } else {
518                    found_gphase = true;
519                }
520            } else {
521                nhd[i].insert(n);
522            }
523        }
524
525        if !found_gphase {
526            return false;
527        }
528    }
529
530    nhd[0] == nhd[1]
531}

533pub fn gadget_fusion_unchecked(g: &mut impl GraphLike, v0: V, v1: V) {
534    let gphase0 = g
535        .neighbors(v0)
536        .find(|&n| g.degree(n) == 1)
537        .expect("v0 isn't a gadget");
538    let gphase1 = g
539        .neighbors(v1)
540        .find(|&n| g.degree(n) == 1)
541        .expect("v1 isn't a gadget");
542    g.add_to_phase(gphase0, g.phase(gphase1));
543    g.remove_vertex(v1);
544    g.remove_vertex(gphase1);
545
546    let d = g.degree(v0) as i32;
547    g.scalar_mut().mul_sqrt2_pow(2 - d);
548}

550checked_rule2!(check_gadget_fusion, gadget_fusion_unchecked, gadget_fusion);

566pub fn check_remove_pair(g: &impl GraphLike, v0: V, v1: V) -> bool {
567    let t0 = g.vertex_type(v0);
568    let t1 = g.vertex_type(v1);
569
570    g.neighbors(v0).len() == 1
571        && g.neighbors(v1).len() == 1
572        && (t0 == VType::Z || t0 == VType::X)
573        && (t1 == VType::Z || t1 == VType::X)
574        && g.connected(v0, v1)
575}

577/// Remove an isolated Z or X vertex and add it as a global scalar
578pub fn remove_pair_unchecked(g: &mut impl GraphLike, v0: V, v1: V) {
579    let t0 = g.vertex_type(v0);
580    let t1 = g.vertex_type(v1);
581    let et = g.edge_type(v0, v1);
582    let p0 = g.phase(v0);
583    let p1 = g.phase(v1);
584
585    // same color
586    if (t0 == t1 && et == EType::N) || (t0 != t1 && et == EType::H) {
587        g.scalar_mut().mul_one_plus_phase(p0 + p1);
588    // different colors
589    } else {
590        let p2 = Phase::one() + p0 + p1;
591        *g.scalar_mut() *= FScalar::one()
592            + FScalar::from_phase(p0)
593            + FScalar::from_phase(p1)
594            + FScalar::from_phase(p2);
595        g.scalar_mut().mul_sqrt2_pow(-1);
596    }
597
598    g.remove_vertex(v0);
599    g.remove_vertex(v1);
600}
601
602checked_rule2!(check_remove_pair, remove_pair_unchecked, remove_pair);

654    #[test]
655    fn spider_fusion_smart() {
656        // a spider fusion that creates parallel edges, which get
657        // removed by complementarity
658        let mut g = Graph::new();
659        let vs = [
660            g.add_vertex(VType::B),
661            g.add_vertex(VType::B),
662            g.add_vertex(VType::Z),
663            g.add_vertex(VType::Z),
664            g.add_vertex(VType::X),
665            g.add_vertex(VType::B),
666            g.add_vertex(VType::B),
667        ];
668
669        g.set_phase(vs[2], Rational64::new(1, 2));
670        g.set_phase(vs[3], Rational64::new(1, 4));
671
672        g.add_edge(vs[0], vs[2]);
673        g.add_edge(vs[1], vs[2]);
674        g.add_edge(vs[2], vs[3]);
675        g.add_edge(vs[2], vs[4]);
676        g.add_edge(vs[3], vs[4]);
677        g.add_edge(vs[3], vs[5]);
678        g.add_edge(vs[4], vs[6]);
679
680        g.set_inputs(vec![vs[0], vs[1]]);
681        g.set_outputs(vec![vs[5], vs[6]]);
682
683        assert_eq!(g.num_vertices(), 7);
684        assert_eq!(g.num_edges(), 7);
685        assert_eq!(g.degree(vs[2]), 4);
686        assert_eq!(g.degree(vs[4]), 3);
687        assert!(g.scalar().is_one());
688
689        let h = g.clone();
690        let success = spider_fusion(&mut g, vs[2], vs[3]);
691        assert!(success, "Spider fusion should match");
692
693        assert_eq!(g.num_vertices(), 6);
694        assert_eq!(g.num_edges(), 4);
695        assert_eq!(g.degree(vs[2]), 3);
696        assert_eq!(g.degree(vs[4]), 1);
697        assert_eq!(*g.scalar(), FScalar::sqrt2_pow(-2));
698
699        let tg = g.to_tensorf();
700        let th = h.to_tensorf();
701        println!("\n\ntg =\n{tg}");
702        println!("\n\nth =\n{th}");
703        assert_eq!(tg, th);
704
705        assert_eq!(g.phase(vs[2]), Rational64::new(3, 4).into());
706    }

719        g.add_edge_with_type(0, 3, EType::H);
720        g.add_edge_with_type(0, 4, EType::H);
721
722        let b1 = g.add_vertex(VType::B);
723        let b2 = g.add_vertex(VType::B);
724        let b3 = g.add_vertex(VType::B);
725        let b4 = g.add_vertex(VType::B);
726        g.add_edge_with_type(b1, 1, EType::H);
727        g.add_edge_with_type(b2, 2, EType::H);
728        g.add_edge_with_type(b3, 3, EType::H);
729        g.add_edge_with_type(b4, 4, EType::H);
730        g.set_inputs(vec![b1, b2]);
731        g.set_outputs(vec![b3, b4]);
732
733        assert_eq!(g.num_vertices(), 9);
734        assert_eq!(g.num_edges(), 8);
735
736        let h = g.clone();
737        let success = local_comp(&mut g, 0);
738        assert!(success, "Local comp should match");
739
740        assert_eq!(g.num_vertices(), 8);
741        assert_eq!(g.num_edges(), 10);
742
743        let tg = g.to_tensorf();
744        let th = h.to_tensorf();
745        println!("\n\ntg =\n{tg}");
746        println!("\n\nth =\n{th}");
747        assert_eq!(tg, th);
748
749        for i in 1..5 {
750            assert_eq!(g.phase(i), Rational64::new(-1, 2).into());
751        }
752
753        assert_eq!(
754            *g.scalar(),
755            FScalar::sqrt2_pow((4 - 1) * (4 - 2) / 2) * FScalar::from_phase(Rational64::new(1, 4))
756        );
757
758        let h = g.clone();
759        let fail = local_comp(&mut g, 1);
760        assert!(!fail, "Local comp should not match");
761        assert_eq!(g, h);
762    }

920    fn gadget_fusion_1() {
921        // fuse gadgets of various sizes
922        for n in 1..5 {
923            let mut graph = Graph::new();
924            let bs: Vec<_> = (0..n).map(|_| graph.add_vertex(VType::B)).collect();
925            let vs: Vec<_> = (0..n).map(|_| graph.add_vertex(VType::Z)).collect();
926            let gs: Vec<_> = (0..2).map(|_| graph.add_vertex(VType::Z)).collect();
927            let ps: Vec<_> = (0..2).map(|_| graph.add_vertex(VType::Z)).collect();
928            graph.set_inputs(bs.clone());
929
930            for i in 0..n {
931                graph.add_edge(bs[i], vs[i]);
932            }
933            for (&g, &p) in gs.iter().zip(ps.iter()) {
934                graph.add_edge_with_type(g, p, EType::H);
935                for &v in &vs {
936                    graph.add_edge_with_type(v, g, EType::H);
937                }
938            }
939
940            graph.set_phase(ps[0], Rational64::new(1, 4));
941            graph.set_phase(ps[1], Rational64::new(1, 2));
942
943            let h = graph.clone();
944
945            assert!(gadget_fusion(&mut graph, gs[0], gs[1]));
946            assert!(graph
947                .find_vertex(|v| graph.phase(v) == Rational64::new(3, 4).into())
948                .is_some());
949            assert!(graph
950                .find_vertex(|v| graph.phase(v) == Rational64::new(1, 4).into())
951                .is_none());
952            assert!(graph
953                .find_vertex(|v| graph.phase(v) == Rational64::new(1, 2).into())
954                .is_none());
955            // println!("{}", g.to_tensorf());
956            // println!("{}", h.to_tensorf());
957            // println!("g = {} * \n {} \n\n", g.scalar(), g.to_dot());
958            // println!("h = {} * \n {} \n\n", h.scalar(), h.to_dot());
959            assert_eq!(graph.to_tensorf(), h.to_tensorf());
960        }
961    }
