17//! Json encoding for interoperability with pyzx and zxlang using the .qgraph format.
18//!
19//! # Examples
20//!
21//! ```rust
22//! # use quizx::graph::{EType, VType, GraphLike};
23//! # use quizx::vec_graph::Graph;
24//! # use quizx::tensor::ToTensor;
25//! // Define a graph with 4 vertices and 3 edges.
26//! let mut g = Graph::new();
27//! let vs = vec![
28//!     g.add_vertex(VType::B),
29//!     g.add_vertex(VType::Z),
30//!     g.add_vertex(VType::X),
31//!     g.add_vertex(VType::B),
32//! ];
33//! g.set_inputs(vec![vs[0]]);
34//! g.set_outputs(vec![vs[3]]);
35//! g.add_edge(vs[0], vs[1]);
36//! g.add_edge_with_type(vs[1], vs[2], EType::H);
37//! g.add_edge(vs[2], vs[3]);
38//!
39//! // Encode the graph in qgraph format.
40//! let json = quizx::json::encode_graph(&g).unwrap();
41//!
42//! // Decode the graph from the qgraph string.
43//! let g2 = quizx::json::decode_graph::<Graph>(&json).unwrap();
44//!
45//! assert_eq!(g.to_tensorf(), g2.to_tensorf());
46//! ```

62/// Returns the json-encoded representation of a graph.
63pub fn encode_graph(graph: &impl crate::graph::GraphLike) -> Result<String, JsonError> {
64    let jg = JsonGraph::from_graph(graph)?;
65    let s = serde_json::to_string(&jg)?;
66    Ok(s)
67}
68
69/// Writes the json-encoded representation of a graph to a file.
70pub fn write_graph(graph: &impl crate::graph::GraphLike, filename: &Path) -> Result<(), JsonError> {
71    let jg = JsonGraph::from_graph(graph)?;
72    let file = std::fs::File::create(filename).unwrap();
73    let writer = std::io::BufWriter::new(file);
74    serde_json::to_writer(writer, &jg)?;
75    Ok(())
76}
77
78/// Reads a graph from its json-encoded representation.
79pub fn decode_graph<G: GraphLike>(s: &str) -> Result<G, JsonError> {
80    let jg: JsonGraph = serde_json::from_str(s)?;
81    jg.to_graph()
82}
83
84/// Reads a graph from a json-encoded file.
85pub fn read_graph<G: GraphLike>(filename: &Path) -> Result<G, JsonError> {
86    let file = std::fs::File::open(filename).unwrap();
87    let reader = std::io::BufReader::new(file);
88    let jg: JsonGraph = serde_json::from_reader(reader)?;
89    jg.to_graph()
90}

92/// Identifier for an encoded vertex.
93type VertexName = String;
94/// Identifier for an encoded edge.
95type EdgeName = String;

97/// The json-encoded format for pyzx and zxlang graphs.
98#[derive(Serialize, Deserialize, Debug, Default, Clone)]
99pub struct JsonGraph {
100    /// Wire vertices of the graph.
101    #[serde(default)]
102    wire_vertices: HashMap<VertexName, VertexAttrs>,
103    /// Node vertices of the graph.
104    #[serde(default)]
105    node_vertices: HashMap<VertexName, VertexAttrs>,
106    /// Undirected edges between node vertices.
107    #[serde(default)]
108    undir_edges: HashMap<EdgeName, EdgeAttrs>,
109    /// Types of the variables in the graph.
110    ///
111    /// Currently ignored by quizx.
112    #[serde(default)]
113    variable_types: HashMap<String, String>,
114    /// The graph scalar.
115    ///
116    /// pyzx encodes this as a json-encoded string instead of directly embedding
117    /// the dictionary.
118    #[serde(skip_serializing_if = "is_default")]
119    #[serde(default)]
120    #[serde(deserialize_with = "deserialize_scalar")]
121    #[serde(serialize_with = "serialize_scalar")]
122    scalar: Option<JsonScalar>,
123}

125/// Attributes for a vertex in the json-encoded graph.
126#[derive(Serialize, Deserialize, Debug, Default, Clone)]
127struct VertexAttrs {
128    #[serde(default)]
129    annotation: VertexAnnotations,
130    #[serde(default)]
131    data: VertexData,
132}
133
134/// Data for a vertex in the json-encoded graph.
135#[derive(Serialize, Deserialize, Debug, Default, Clone, PartialEq, Eq)]
136struct VertexData {
137    /// The vertex type.
138    #[serde(rename = "type")]
139    typ: VType,
140    /// The vertex phase.
141    #[serde(skip_serializing_if = "is_default")]
142    #[serde(default)]
143    value: JsonPhase,
144    /// A flag marking grounded nodes.
145    #[serde(skip_serializing_if = "is_default")]
146    #[serde(default)]
147    ground: bool,
148    /// Hadamard wires are encoded as nodes with this flag set,
149    /// so that they can be recovered during decoding.
150    ///
151    /// Note that in the pyzx encoder, this is either the string "true" or "false".
152    /// So we need to deserialize it into a bool manually.
153    #[serde(skip_serializing_if = "is_default")]
154    #[serde(default)]
155    #[serde(deserialize_with = "deserialize_bool")]
156    #[serde(serialize_with = "serialize_bool")]
157    is_edge: bool,
158}

160/// The annotations of a vertex in the json-encoded graph.
161#[derive(Serialize, Deserialize, Debug, Default, Clone, PartialEq)]
162struct VertexAnnotations {
163    /// This is a boundary vertex.
164    #[serde(skip_serializing_if = "is_default")]
165    #[serde(default)]
166    boundary: bool,
167    /// Coordinates of the vertex, for rendering purposes.
168    #[serde(default)]
169    coord: (f64, f64),
170    /// The input number for the vertex.
171    ///
172    /// Note that in the pyzx encoder, this is either a number indicating the order in the input list, or
173    /// (in older versions) a boolean flag indicating whether the vertex is an input.
174    /// So we need to deserialize it manually.
175    #[serde(skip_serializing_if = "is_default")]
176    #[serde(default)]
177    #[serde(deserialize_with = "deserialize_io_attribute")]
178    input: Option<usize>,
179    /// The output number for the vertex.
180    #[serde(skip_serializing_if = "is_default")]
181    #[serde(default)]
182    #[serde(deserialize_with = "deserialize_io_attribute")]
183    output: Option<usize>,
184    /// A box label.
185    #[serde(skip_serializing_if = "is_default")]
186    #[serde(default)]
187    label: Option<String>,
188    /// Other arbitrary annotations associated with the vertex.
189    #[serde(flatten)]
190    other: HashMap<String, String>,
191}

193/// Attributes for an edge in the json-encoded graph.
194#[derive(Serialize, Deserialize, Debug, Default, Clone)]
195struct EdgeAttrs {

197    src: VertexName,

201    #[serde(skip_serializing_if = "is_default")]

209/// Encoded as string-formatted rational or a floating point number.

211#[derive(Debug, Default, Clone, PartialEq, Eq, PartialOrd, Ord, Serialize, Deserialize)]

215/// Global scalars in a graph
216#[derive(Serialize, Deserialize, Debug, Default, Clone, PartialEq)]
217pub struct JsonScalar {
218    /// Stores the power of sqrt(2).
219    #[serde(default)]
220    power2: i32,
221    /// Stores complex phase of the number, in half turns.
222    #[serde(default)]
223    phase: JsonPhase,
224    /// A floating point factor for the scalar.
225    #[serde(default)]
226    #[serde(skip_serializing_if = "is_default")]
227    floatfactor: f64,
228    /// Stores a list of legless spiders, by their phases.
229    #[serde(default)]
230    #[serde(skip_serializing_if = "is_default")]
231    phasenodes: Vec<JsonPhase>,
232    /// Whether this scalar is zero.
233    #[serde(default)]
234    #[serde(skip_serializing_if = "is_default")]
235    is_zero: bool,
236    /// Whether this represents an unknown scalar value.
237    #[serde(default)]
238    #[serde(skip_serializing_if = "is_default")]
239    is_unknown: bool,
240}

242/// Helper method to skip serialization of default values in serde.
243///
244/// ```skip
245/// use serde::Serialize;
246///
247/// #[derive(Serialize)]
248/// struct MyStruct {
249///     #[serde(skip_serializing_if = "is_default")]
250///     field: i32,
251/// }
252/// ```
253///
254/// From https://github.com/serde-rs/serde/issues/818.
255pub(crate) fn is_default<T: Default + PartialEq>(t: &T) -> bool {
256    *t == Default::default()
257}

259/// Serialize a boolean to a string field.
260fn serialize_bool<S>(b: &bool, serializer: S) -> Result<S::Ok, S::Error>
261where
262    S: serde::Serializer,
263{
264    match *b {
265        true => serializer.serialize_str("true"),
266        false => serializer.serialize_str("false"),
267    }
268}
269
270/// Deserialize a boolean from a string field.
271fn deserialize_bool<'de, D>(deserializer: D) -> Result<bool, D::Error>
272where
273    D: de::Deserializer<'de>,
274{
275    let s: &str = de::Deserialize::deserialize(deserializer)?;
276
277    match s {
278        "true" => Ok(true),
279        "false" => Ok(false),
280        _ => Err(de::Error::unknown_variant(s, &["true", "false"])),
281    }
282}

284/// Deserialize the input/output attribute of a vertex.
285///
286/// This is either a number indicating the order in the input list, or
287/// (in older versions) a boolean flag indicating whether the vertex is an input.
288fn deserialize_io_attribute<'de, D>(deserializer: D) -> Result<Option<usize>, D::Error>
289where
290    D: de::Deserializer<'de>,
291{
292    let val: serde_json::Value = de::Deserialize::deserialize(deserializer)?;
293
294    match val {
295        serde_json::Value::Number(n) => Ok(Some(n.as_u64().unwrap() as usize)),
296        serde_json::Value::Bool(b) => Ok(b.then_some(0)),
297        _ => Err(de::Error::invalid_value(
298            de::Unexpected::Str(&val.to_string()),
299            &"a number or a boolean",
300        )),
301    }
302}
303
304/// Deserialize a scalar from a string field.
305fn deserialize_scalar<'de, D>(deserializer: D) -> Result<Option<JsonScalar>, D::Error>
306where
307    D: de::Deserializer<'de>,
308{
309    let s: String = de::Deserialize::deserialize(deserializer)?;
310    if s.is_empty() {
311        return Ok(None);
312    }
313    let scalar = serde_json::from_str(&s).map_err(|_| {
314        de::Error::invalid_value(
315            de::Unexpected::Str(&s),
316            &"a json-encoded string representing a scalar",
317        )
318    })?;
319    Ok(Some(scalar))
320}
321
322/// Serialize a scalar to a string field.
323fn serialize_scalar<S>(scalar: &Option<JsonScalar>, serializer: S) -> Result<S::Ok, S::Error>
324where
325    S: serde::Serializer,
326{
327    if scalar.is_none() {
328        return serializer.serialize_str("");
329    }
330    let s = serde_json::to_string(scalar.as_ref().unwrap()).map_err(serde::ser::Error::custom)?;
331    serializer.serialize_str(&s)
332}

334/// An error that can occur when decoding a json graph.
335#[derive(Debug, Display, Error, From)]
336#[non_exhaustive]
337pub enum JsonError {
338    /// Found an invalid phase value.
339    #[display("Got an invalid phase value {phase}")]
340    InvalidPhase { phase: String },
341    /// Found an invalid phase value in a node definition.
342    #[display("Got an invalid phase value {phase} for node {name}")]
343    InvalidNodePhase { name: String, phase: String },
344    /// Some other serde error.
345    #[from]
346    SerdeError(serde_json::Error),
347}

388    //const TEST_JSON_SIMPLE: &str = include_str!("../../test_files/simple-graph.qgraph");
389    const TEST_JSON_4Q_UNITARY: &str = include_str!("../../test_files/4-qubit-unitary.qgraph");

391    #[rstest]
392    fn json_roundtrip(simple_graph: (Graph, Vec<V>)) -> Result<(), JsonError> {
393        let (g, _) = simple_graph;
394        let jg = JsonGraph::from_graph(&g)?;
395        let s = serde_json::to_string(&jg).unwrap();
396
397        let g2: Graph = decode_graph(&s).unwrap();
398
399        assert_eq!(g.num_vertices(), g2.num_vertices());
400        assert_eq!(g.num_edges(), g2.num_edges());
401
402        let inp1 = g.inputs();
403        let inp2 = g2.inputs();
404
405        assert_eq!(inp1.len(), inp2.len());
406        for (v1, v2) in inp1.iter().zip(inp2.iter()) {
407            assert_eq!(g.vertex_type(*v1), g2.vertex_type(*v2));
408
409            let ns1 = g.neighbors(*v1);
410            let ns2 = g2.neighbors(*v2);
411            for (n1, n2) in ns1.zip(ns2) {
412                assert_eq!(g.vertex_type(n1), g2.vertex_type(n2));
413            }
414        }
415
416        Ok(())
417    }

423        let g: Graph = decode_graph(json).unwrap();
