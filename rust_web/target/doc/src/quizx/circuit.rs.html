34#[derive(PartialEq, Eq, Clone, Copy, Debug)]
35pub struct CircuitStats {
36    pub qubits: usize,
37    pub total: usize,
38    pub oneq: usize,
39    pub twoq: usize,
40    pub moreq: usize,
41    pub cliff: usize,
42    pub non_cliff: usize,

45impl CircuitStats {
46    pub fn make(c: &Circuit) -> Self {
47        let mut s = CircuitStats {
48            qubits: c.num_qubits(),
49            total: c.num_gates(),
50            oneq: 0,
51            twoq: 0,
52            moreq: 0,
53            cliff: 0,
54            non_cliff: 0,
55        };
56        for g in &c.gates {
57            match g.qs.len() {
58                1 => {
59                    s.oneq += 1;
60                }
61                2 => {
62                    s.twoq += 1;
63                }
64                _ => {
65                    s.moreq += 1;
66                }
67            }
68
69            match g.t {
70                NOT | Z | S | Sdg | CNOT | CZ | SWAP | HAD => {
71                    s.cliff += 1;
72                }
73                ZPhase | XPhase => {
74                    if g.phase.is_clifford() {
75                        s.cliff += 1;
76                    } else {
77                        s.non_cliff += 1;
78                    }
79                }
80                _ => {
81                    s.non_cliff += 1;
82                }
83            }
84        }
85        s
86    }
87
88    pub fn into_array(self) -> [usize; 7] {
89        [
90            self.qubits,
91            self.total,
92            self.oneq,
93            self.twoq,
94            self.moreq,
95            self.cliff,
96            self.non_cliff,
97        ]
98    }
99}

178    fn from_qasm_parser(read: impl FnOnce(&mut openqasm::Parser)) -> Result<Circuit, String> {
179        let mut cache = openqasm::SourceCache::new();
180        let mut parser = openqasm::Parser::new(&mut cache)
181            .with_file_policy(openqasm::parser::FilePolicy::Ignore);
182        read(&mut parser);
183        parser.parse_source::<String>(
184            "
185            opaque rz(phase) q;
186            opaque rx(phase) q;
187            opaque x q;
188            opaque z q;
189            opaque s q;
190            opaque t q;
191            opaque sdg q;
192            opaque tdg q;
193            opaque h q;
194            opaque cx a, b;
195            opaque cz a, b;
196            opaque ccx a, b, c;
197            opaque ccz a, b, c;
198            opaque swap a, b;
199            opaque xcx a, b;
200            opaque init_anc a;
201            opaque post_sel a;
202        "
203            .to_string(),
204            None,
205        );
206
207        let program = parser.done().to_errors().map_err(|e| e.to_string())?;
208        program
209            .type_check()
210            .to_errors()
211            .map_err(|e| e.to_string())?;
212
213        let mut writer = CircuitWriter {
214            circuit: Circuit::new(0),
215        };
216        let mut linearize = openqasm::Linearize::new(&mut writer, usize::MAX);
217        linearize
218            .visit_program(&program)
219            .to_errors()
220            .map_err(|e| e.to_string())?;
221
222        Ok(writer.circuit)
223    }
224
225    pub fn from_qasm(source: &str) -> Result<Circuit, String> {
226        Circuit::from_qasm_parser(|parser| parser.parse_source::<String>(source.to_string(), None))
227    }

233    /// returns a copy of the circuit, decomposed into 1- and 2-qubit Clifford +
234    /// phase gates.
235    pub fn to_basic_gates(&self) -> Circuit {
236        // calculate the space needed in advance
237        let sz = self.gates.iter().map(|g| g.num_basic_gates()).sum();
238        let mut c = Circuit {
239            gates: VecDeque::with_capacity(sz),
240            nqubits: self.nqubits,
241        };
242        for g in &self.gates {
243            g.push_basic_gates(&mut c);
244        }
245
246        c
247    }

249    pub fn to_graph_with_options<G: GraphLike>(&self, postselect: bool) -> G {
250        let mut graph = G::new();
251        let mut qs = Vec::with_capacity(self.nqubits);
252        let mut inputs = Vec::with_capacity(self.nqubits);
253
254        // we start counting rows from 1, to allow coordinate
255        // (0,0) to mean "no coordinate"
256        for i in 0..self.nqubits {
257            let v = graph.add_vertex_with_data(VData {
258                ty: VType::B,
259                phase: Phase::zero(),
260                qubit: i as f64,
261                row: 1.0,
262            });
263            qs.push(Some(v));
264            inputs.push(v);
265        }
266
267        graph.set_inputs(inputs);
268
269        for g in &self.gates {
270            g.add_to_graph(&mut graph, &mut qs, postselect);
271        }
272
273        let last_row = qs
274            .iter()
275            .fold(None, |r, &v| match (r, v.map(|v1| graph.row(v1))) {
276                (Some(r), Some(r1)) => Some(if r < r1 { r1 } else { r }),
277                (Some(r), None) => Some(r),
278                (None, Some(r1)) => Some(r1),
279                (None, None) => None,
280            })
281            .unwrap_or(0.0);
282
283        let mut outputs = Vec::with_capacity(self.nqubits);
284        for (i, &q) in qs.iter().enumerate() {
285            if let Some(v0) = q {
286                let v = graph.add_vertex_with_data(VData {
287                    ty: VType::B,
288                    phase: Phase::zero(),
289                    qubit: i as f64,
290                    row: last_row + 1.0,
291                });
292                graph.add_edge(v0, v);
293                outputs.push(v);
294            }
295        }
296
297        graph.set_outputs(outputs);
298        graph
299    }
300
301    pub fn to_graph<G: GraphLike>(&self) -> G {
302        self.to_graph_with_options(false)
303    }

322impl std::ops::Add<Circuit> for Circuit {
323    type Output = Circuit;
324    fn add(mut self, mut rhs: Circuit) -> Self::Output {
325        if self.num_qubits() != rhs.num_qubits() {
326            panic!("Cannot append circuits with different numbers of qubits");
327        }
328        self.gates.append(&mut rhs.gates);
329        self
330    }
331}

364/// A circuit can pretend to be a matrix, where row/column operations correspond
365/// to appending or prepending CNOT gates.
366///
367/// For example, we can synthesise a CNOT circuit corresponding to the parity
368/// matrix `m` as follows:
369///
370/// ```
371/// use quizx::circuit::Circuit;
372/// use quizx::linalg::*;
373/// let mut c = Circuit::new(3); // c|b> = |id * b>
374/// let mut m = Mat2::new(vec![vec![1,1,1], vec![0,1,1], vec![0,0,1]]);
375/// m.gauss_x(true, 1, &mut c);  // c|b> = |m^-1 * b>
376/// c.reverse();                 // c|b> = |m * b>
377///
378/// ```
379impl RowOps for Circuit {
380    fn row_add(&mut self, r0: usize, r1: usize) {
381        self.push_back(Gate::new(GType::CNOT, vec![r1, r0]));
382    }
383
384    fn row_swap(&mut self, r0: usize, r1: usize) {
385        self.push_back(Gate::new(GType::SWAP, vec![r0, r1]));
386    }
387}

389struct CircuitWriter {
390    circuit: Circuit,
391}
392
393#[derive(Debug)]
394#[allow(clippy::enum_variant_names)]
395enum CircuitWriterError {
396    UnitaryNotSupported,
397    BarrierNotSupported,
398    ResetNotSupported,
399    MeasureNotSupported,
400    ConditionalNotSupported,
401}
402
403impl std::fmt::Display for CircuitWriterError {
404    fn fmt(&self, f: &mut std::fmt::Formatter) -> std::fmt::Result {
405        match self {
406            CircuitWriterError::UnitaryNotSupported => {
407                write!(f, "arbitrary unitaries are not supported")
408            }
409            CircuitWriterError::BarrierNotSupported => write!(f, "barriers are not supported"),
410            CircuitWriterError::ResetNotSupported => write!(f, "resets are not supported"),
411            CircuitWriterError::MeasureNotSupported => write!(f, "measurements are not supported"),
412            CircuitWriterError::ConditionalNotSupported => {
413                write!(f, "conditionals are not supported")
414            }
415        }
416    }
417}
418
419impl std::error::Error for CircuitWriterError {}
420
421impl openqasm::GateWriter for &mut CircuitWriter {
422    type Error = CircuitWriterError;
423
424    fn initialize(&mut self, qubits: &[Symbol], _: &[Symbol]) -> Result<(), Self::Error> {
425        self.circuit = Circuit::new(qubits.len());
426        Ok(())
427    }
428
429    fn write_cx(&mut self, a: usize, b: usize) -> Result<(), Self::Error> {
430        self.circuit.push(Gate::new(GType::CNOT, vec![a, b]));
431        Ok(())
432    }
433
434    fn write_opaque(
435        &mut self,
436        name: &Symbol,
437        params: &[Value],
438        regs: &[usize],
439    ) -> Result<(), Self::Error> {
440        fn param_to_phase(value: Value) -> Phase {
441            if value.a.is_zero() {
442                Rational64::new(*value.b.numer(), *value.b.denom()).into()
443            } else {
444                let a = *value.a.numer() as f32 / *value.a.denom() as f32;
445                let mut r =
446                    Rational64::approximate_float(a / std::f32::consts::PI).unwrap_or(0.into());
447                r += Rational64::new(*value.b.numer(), *value.b.denom());
448                Phase::new(r)
449            }
450        }
451
452        let mut g = Gate::from_qasm_name(name.as_str());
453        g.qs.extend_from_slice(regs);
454        if !params.is_empty() {
455            g.phase = param_to_phase(params[0]);
456        }
457
458        self.circuit.push(g);
459
460        Ok(())
461    }
462
463    fn write_u(&mut self, _: Value, _: Value, _: Value, _: usize) -> Result<(), Self::Error> {
464        Err(CircuitWriterError::UnitaryNotSupported)
465    }
466
467    fn write_barrier(&mut self, _: &[usize]) -> Result<(), Self::Error> {
468        Err(CircuitWriterError::BarrierNotSupported)
469    }
470
471    fn write_reset(&mut self, _: usize) -> Result<(), Self::Error> {
472        Err(CircuitWriterError::ResetNotSupported)
473    }
474
475    fn write_measure(&mut self, _: usize, _: usize) -> Result<(), Self::Error> {
476        Err(CircuitWriterError::MeasureNotSupported)
477    }
478
479    fn start_conditional(&mut self, _: usize, _: usize, _: u64) -> Result<(), Self::Error> {
480        Err(CircuitWriterError::ConditionalNotSupported)
481    }
482
483    fn end_conditional(&mut self) -> Result<(), Self::Error> {
484        Err(CircuitWriterError::ConditionalNotSupported)
485    }
486}
