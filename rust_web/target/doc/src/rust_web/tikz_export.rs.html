1use quizx::hash_graph::*;
2use std::fs::File;
3use std::io::{Write, Result};
4
5/// Exports a graph to a TikZ file for LaTeX visualization
6#[allow(dead_code)] // Remove once used
7pub fn export_to_tikz(g: &Graph, filename: &str) -> Result<()> {
8    let mut file = File::create(filename)?;
9
10    writeln!(file, "\\documentclass{{standalone}}")?;
11    writeln!(file, "\\usepackage{{tikz}}")?;
12    writeln!(file, "\\begin{{document}}")?;
13    writeln!(file, "\\begin{{tikzpicture}}[scale=1]")?;
14
15    // Define styles for X, Z and boundary spiders
16    writeln!(file, "\\tikzstyle{{xspider}}=[draw,circle,fill=red!20]")?;
17    writeln!(file, "\\tikzstyle{{zspider}}=[draw,circle,fill=green!20]")?;
18    writeln!(file, "\\tikzstyle{{boundary}}=[draw,circle,fill=black!20]")?;
19
20    let mut positions = vec![];
21
22    // Assign simple horizontal positions to each vertex
23    for (i, v) in g.vertices().enumerate() {
24        let x = i as f64 * 1.5; // horizontal spacing
25        let (style, label) = match g.vertex_type(v) {
26            VType::X => {
27                let phase = g.phase(v);
28                let phase_str = if phase.to_string() == "0" {
29                    String::from("")
30                } else {
31                    format!("{}π", phase)
32                };
33                ("xspider", phase_str)
34            },
35            VType::Z => {
36                let phase = g.phase(v);
37                let phase_str = if phase.to_string() == "0" {
38                    String::from("")
39                } else {
40                    format!("{}π", phase)
41                };
42                ("zspider", phase_str)
43            },
44            _ => ("boundary", String::from("B")),
45        };
46        
47        writeln!(
48            file,
49            "\\node[{}] (v{}) at ({},0) {{{}}};",
50            style, v, x, label
51        )?;
52        positions.push((v, x));
53    }
54
55    // Draw edges
56    for (v0, v1, _) in g.edges() {
57        writeln!(file, "\\draw (v{}) -- (v{});", v0, v1)?;
58    }
59
60    writeln!(file, "\\end{{tikzpicture}}")?;
61    writeln!(file, "\\end{{document}}")?;
62    
63    Ok(())
64}
