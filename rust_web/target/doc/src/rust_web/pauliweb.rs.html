1use std::collections::HashMap;
2use serde::{Serialize, Deserialize};
3
4#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash, Serialize, Deserialize)]
5pub enum Pauli {
6    X,
7    Y,
8    Z,
9}
10
11/// Represents a Pauli web in a ZX diagram
12#[derive(Debug, Default, Clone)]
13pub struct PauliWeb {
14    /// Maps edge (from, to) to Pauli operator
15    /// Note: from < to to ensure consistent ordering
16    pub edge_operators: HashMap<(usize, usize), Pauli>,
17}
18
19impl PauliWeb {
20    /// Create a new empty PauliWeb
21    pub fn new() -> Self {
22        Self::default()
23    }
24
25    /// Set the Pauli operator for an edge between two nodes
26    pub fn set_edge(&mut self, from: usize, to: usize, pauli: Pauli) {
27        self.edge_operators.insert((from.min(to), from.max(to)), pauli);
28    }
29
30    /// Get the Pauli operator for an edge between two nodes
31    pub fn get_edge(&self, from: usize, to: usize) -> Option<Pauli> {
32        self.edge_operators.get(&(from.min(to), from.max(to))).copied()
33    }
34
35    /// Get the color to use when drawing an edge
36    pub fn get_edge_color(&self, from: usize, to: usize) -> Option<&'static str> {
37        self.get_edge(from, to).map(|pauli| match pauli {
38            Pauli::X => "red",
39            Pauli::Y => "blue",
40            Pauli::Z => "green",
41        })
42    }
43}

45#[cfg(test)]
46mod tests {
47    use super::*;
48
49    #[test]
50    fn test_new_pauliweb() {
51        let pw = PauliWeb::new();
52        assert!(pw.edge_operators.is_empty());
53    }
54
55    #[test]
56    fn test_set_and_get_edge() {
57        let mut pw = PauliWeb::new();
58        
59        // Test setting and getting an edge
60        pw.set_edge(1, 2, Pauli::X);
61        assert_eq!(pw.get_edge(1, 2), Some(Pauli::X));
62        assert_eq!(pw.get_edge(2, 1), Some(Pauli::X)); // Should work in both directions
63        
64        // Test updating an edge
65        pw.set_edge(1, 2, Pauli::Z);
66        assert_eq!(pw.get_edge(1, 2), Some(Pauli::Z));
67        
68        // Test non-existent edge
69        assert_eq!(pw.get_edge(1, 3), None);
70    }
71
72    #[test]
73    fn test_get_edge_color() {
74        let mut pw = PauliWeb::new();
75        
76        // Test colors for each Pauli operator
77        pw.set_edge(1, 2, Pauli::X);
78        pw.set_edge(2, 3, Pauli::Y);
79        pw.set_edge(3, 4, Pauli::Z);
80        
81        assert_eq!(pw.get_edge_color(1, 2), Some("red"));
82        assert_eq!(pw.get_edge_color(2, 3), Some("blue"));
83        assert_eq!(pw.get_edge_color(3, 4), Some("green"));
84        assert_eq!(pw.get_edge_color(4, 5), None); // Non-existent edge
85    }
86
87    #[test]
88    fn test_edge_ordering() {
89        let mut pw = PauliWeb::new();
90        
91        // Test that edge ordering doesn't matter for get/set
92        pw.set_edge(2, 1, Pauli::X);
93        assert_eq!(pw.get_edge(1, 2), Some(Pauli::X));
94        
95        // Test that updating with different order works
96        pw.set_edge(1, 2, Pauli::Z);
97        assert_eq!(pw.get_edge(2, 1), Some(Pauli::Z));
98    }
